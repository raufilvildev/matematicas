\chapter{Introduction to Measure Theory}

\section{Measurable spaces}

\begin{definition}
  Let $X$ be a set. A \emph{$\sigma$-algebra on $X$} is a collection $\Sigma\subseteq\mathcal{P}\pr{X}$ of subsets in $X$ satisfying the following conditions:

  \begin{itemize}
    \item $\varnothing\in\Sigma$.
    \item If $A\in\Sigma$, then   $A^c:=X\setminus A\in\Sigma$.
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of sets in $\Sigma$, then $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma$.
  \end{itemize}

  The ordered pair $\pr{X,\Sigma}$ is called a \emph{measurable space}. The elements of $\Sigma$ are called \emph{$\Sigma$-measurable sets}.
\end{definition}

\begin{example}
  Let $X$ be a set. Some examples of $\sigma$-algebras on $X$ are the following:

  \begin{itemize}
    \item Trivial (or indiscrete) $\sigma$-algebra: $\Sigma:=\set{\varnothing,X}$.
    \item Discrete $\sigma$-algebra: $\Sigma:=\mathcal{P}\pr{X}$.
    \item Countable-cocountable $\sigma$-algebra: $\Sigma:=\set{A\middle|A\subseteq X,A\text{ or }A^c\text{ is countable}}$.
  \end{itemize}
\end{example}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space. The following properties hold:

  \begin{enumerate}
    \item $X\in\Sigma$.
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of sets in $\Sigma$, then $\bigcap_{n=1}^{\infty}{A_n}\in\Sigma$.
    \item If $A_1,\dots,A_n\in\Sigma$ with $n\in\mathbb{N}$, then $\bigcup_{k=1}^{n}{A_k},\bigcap_{k=1}^{n}{A_k}\in\Sigma$.
    \item If $A,B\in\Sigma$, then $A\setminus B\in\Sigma$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item This is immediate, since $X:=\varnothing^c$.
      \item For each $n\in\mathbb{N}$, since $A_n\in\Sigma$, we have $A_n^c\in\Sigma$. Then $\bigcup_{n=1}^{\infty}{A_n^c}\in\Sigma$. By De Morgan's law, $\bigcap_{n=1}^{\infty}{A_n}=\pr{\bigcup_{n=1}^{\infty}{A_n^c}}^c\in\Sigma$.
      \item Given $m\in\mathbb{N}$, we define:
      
      \begin{equation}
        A'_m=\begin{cases}
          A_m & \text{if $m\leq n$}\\
          \varnothing & \text{if $m>n$}
        \end{cases}\hspace{0.5cm}A''_m=\begin{cases}
          A_m & \text{if $m\leq n$}\\
          X & \text{if $m>n$}
        \end{cases}
      \end{equation}

      Then $\set{A'_m}_{m\in\mathbb{N}},\set{A''_m}_{m\in\mathbb{N}}\subseteq\Sigma$. Therefore, $\bigcup_{k=1}^{n}{A_k}=\bigcup_{m=1}^{\infty}{A'_m},\bigcap_{k=1}^{n}{A_k}=\bigcap_{m=1}^{\infty}{A''_m}\in\Sigma$.
      \item Since $B\in\Sigma$, we have $B^c\in\Sigma$. Therefore, $A\setminus B=A\cap B^c\in\Sigma$.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\mathscr{S}$ be a family of $\sigma$-algebras on a set $X$. Then $\Sigma:=\bigcap{\mathscr{S}}\subseteq\mathcal{P}\pr{X}$ is a $\sigma$-algebra on $X$.
  \begin{proof}[\rm\textbf{Proof}]
    Since $\varnothing\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$, we have $\varnothing\in\Sigma$. 
    
    Let $A\in\Sigma$. Then $A\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Thus $A^c\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Therefore, $A^c\in\Sigma$. 
    
    Let $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ be a countable collection of sets in $\Sigma$. Then $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of sets in $\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Thus $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Therefore $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma$.

    In conclusion, $\Sigma$ is a $\sigma$-algebra on $X$.
  \end{proof}
\end{proposition}

\begin{definition}
  Let $X$ be a set and $\mathcal{S}\subseteq\mathcal{P}\pr{X}$ a collection of subsets in $X$. We define the \emph{$\sigma$-algebra generated by $\mathcal{S}$ on $X$} as follows:

  \begin{equation}
    \sigma\pr{X,\mathcal{S}}:=\bigcap\set{\Sigma\middle|\Sigma\subseteq\mathcal{P}\pr{X},\mathcal{S}\subseteq\Sigma, \Sigma\text{ is a $\sigma$-algebra on $X$}}.
  \end{equation}
\end{definition}

\begin{remark}
  If $\pr{X,\Sigma}$ is a measurable space and $\mathcal{S}\subseteq\Sigma$, then $\sigma\pr{X,\mathcal{S}}\subseteq\Sigma$ by the definition of $\sigma\pr{X,\mathcal{S}}$.
\end{remark}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. The \emph{Borel $\sigma$-algebra on $\pr{X,\mathcal{T}}$} is defined by $\mathscr{B}\pr{X,\mathcal{T}}:=\sigma\pr{X,\mathcal{T}}$. The ordered pair $\pr{X,\mathscr{B}\pr{X,\mathcal{T}}}$ is called a \emph{Borel measurable space}.

  If $\pr{X,\mathcal{T}}=\pr{\mathbb{K}^n,\mathcal{T}_e\pr{\mathbb{K}^n}}$ where $\mathbb{K}\in\set{\mathbb{R},\mathbb{C}}$ and $n\in\mathbb{N}$, we write $\mathscr{B}\pr{\mathbb{K}^n}:=\mathscr{B}\pr{\mathbb{K}^n,\mathcal{T}_e\pr{\mathbb{K}^n}}$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space and $S\subseteq X$. The collection $\Sigma_S:=\set{A\cap S\middle|A\in\Sigma}\subseteq\mathcal{P}\pr{S}$ is a $\sigma$-algebra on $S$ called the \emph{$\sigma$-algebra induced by $S$ on $\pr{X,\Sigma}$}.
  \begin{proof}[\rm\textbf{Proof}]
    Since $\varnothing=\varnothing\cap S$, we have $\varnothing\in\Sigma_S$. 
    
    Let $A'\in\Sigma_S$. Then there exists $A\in\Sigma$ such that $A'=A\cap S$. Thus $S\setminus A'=S\setminus\pr{A\cap S}=S\cap\pr{A\cap S}^c=S\cap A^c$. Since $A^c\in\Sigma$, we have $S\setminus A'\in\Sigma_S$. 
    
    Let $\set{A'_n}_{n\in\mathbb{N}}\subseteq\Sigma_S$ be a countable collection of sets in $\Sigma_S$. Then there exists a countable collection of sets $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ in $\Sigma$ such that $A'_n=A_n\cap S$. Since $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma$, we have $\bigcup_{n=1}^{\infty}{A'_n}=\bigcup_{n=1}^{\infty}{\pr{A_n\cap S}}=\pr{\bigcup_{n=1}^{\infty}{A_n}}\cap S\in\Sigma_S$.

    Therefore, $\Sigma_S$ is a $\sigma$-algebra on $S$.
  \end{proof}
\end{proposition}

\section{The extended real line}

Let $\overline{\mathbb{R}}:=\mathbb{R}\cup\set{-\infty,+\infty}$. The arithmetic of $\mathbb{R}$ can be partially extended to $\overline{\mathbb{R}}$ as follows:

\begin{equation}
  \begin{split}
    a+\pr{\pm\infty} & = \pr{\pm\infty}+a=\pm\infty,\forall a\in\mathbb{R},\\
    a\cdot\pr{\pm\infty} & = \pr{\pm\infty}\cdot a=\pm\infty,\forall a\in\pbr{0,+\infty},\\
    -\pr{\pm\infty} & = \mp\infty,\\
    a\cdot\pr{\pm\infty} & = \pr{\pm\infty}\cdot a=\mp\infty,\forall a\in\bpr{-\infty,0},\\
    \pr{\pm\infty}^{-1} & = 0,\\
    \frac{a}{\pm\infty} & = 0,\forall a\in\mathbb{R},\\
    \frac{\pm\infty}{a} & = \pm\infty,\forall a\in\pr{0,+\infty},\\
    \frac{\pm\infty}{a} & = \mp\infty,\forall a\in\pr{-\infty,0},\\
  \end{split}
\end{equation}

Furthermore, we can extend the order relation of $\mathbb{R}$ to $\overline{\mathbb{R}}$ by setting $-\infty<x<+\infty$ for all $x\in\mathbb{R}$ and $-\infty<+\infty$. The ordered pair $\pr{\overline{\mathbb{R}},\leq}$ is a totally ordered set. 

Given $a,b\in\mathbb{R}$, we define the following subsets of $\overline{\mathbb{R}}$:

\begin{equation}
  \bpr{-\infty,b}=\set{x\middle|x\in\overline{\mathbb{R}},x<b},\hspace{0.5cm}\pbr{a,+\infty}=\set{x\middle|x\in\overline{\mathbb{R}},x>a},\hspace{0.5cm}\br{-\infty,+\infty}:=\overline{\mathbb{R}}.
\end{equation}

The order relation on $\overline{\mathbb{R}}$ allows us to define the order topology $\mathcal{T}_e\pr{\overline{\mathbb{R}}}:=\mathcal{T}\pr{\mathcal{B}}$ where $\mathcal{B}$ is the following basis for the topology:

\begin{equation}
  \mathcal{B}:=\set{\pr{a,b}\middle|a,b\in\mathbb{R},a\leq b}\cup\set{\pbr{a,+\infty}\middle|a\in\mathbb{R}}\cup\set{\bpr{-\infty,b}\middle|b\in\mathbb{R}}.
\end{equation}

Observe that $\mathcal{T}_e\pr{\mathbb{R}}\subseteq\mathcal{T}_e\pr{\overline{\mathbb{R}}}$.

\section{Measure spaces}

\begin{definition}
  Let $\pr{X,\Sigma}$ be a measurable space. A \emph{(positive) measure on $\pr{X,\Sigma}$} is a map $\mu:\Sigma\rightarrow\br{0,+\infty}$ satisfying the following conditions:

  \begin{itemize}
    \item Measure of the empty set: $\mu\pr{\varnothing}=0$.
    \item $\sigma$-additivity: If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of pairwise disjoint $\Sigma$-measurable sets, then $\mu\pr{\bigcup_{n=1}^{\infty}{A_n}}=\sum_{n=1}^{\infty}{\mu\pr{A_n}}$.
  \end{itemize}

  The triple $\pr{X,\Sigma,\mu}$ is called a \emph{measure space}.
\end{definition}

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space. We say:

  \begin{itemize}
    \item \emph{$\mu$ is finite} if $\mu\pr{X}<+\infty$. If $\mu\pr{X}=1$, we say that $\mu$ is a \emph{probability measure}. In this context, we say that $X$ is the \emph{sample space}, $\Sigma$ is the \emph{event space}, $\pr{X,\Sigma}$ is the \emph{probabilizable space} and $\pr{X,\Sigma,\mu}$ is the \emph{probability space}. 
    \item \emph{$\mu$ is $\sigma$-finite} if there exists a countable collection $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ of sets in $\Sigma$ with $\mu\pr{A_n}<+\infty$ for all $n\in\mathbb{N}$ such that $X=\bigcup_{n=1}^{\infty}{A_n}$.
    \item \emph{$\mu$ is complete} if for all $A,B\subseteq X$, the condition $B\in\Sigma$ and $\mu\pr{B}=0$ implies that $A\in\Sigma$.
  \end{itemize}
\end{definition}

\begin{example}
  Let $\pr{X,\Sigma}$ be a measurable space. Some examples of measures on $\pr{X,\Sigma}$ are the following:

  \begin{itemize}
    \item Dirac measure: Given $x\in X$, we define $\delta:\Sigma\rightarrow\br{0,+\infty}$ as $\delta_x\pr{A}=1$ if $x\in A$ and $\delta_x\pr{A}=0$ if $x\notin A$.
    \item Cardinality measure: The map $\text{card}:A\in\Sigma\mapsto\text{card}\pr{A}\in\br{0,+\infty}$ is a measure on $\pr{X,\Sigma}$.
    \item Probability measure induced by a finite measure: If $\mu:\Sigma\rightarrow\br{0,+\infty}$ is a finite measure on $\pr{X,\Sigma}$ with $\mu\pr{X}>0$, then $\nu:\Sigma\rightarrow\br{0,+\infty}$ defined by $\nu\pr{A}:=\frac{\mu\pr{A}}{\mu\pr{X}}$ is a probability measure on $\pr{X,\Sigma}$.
  \end{itemize}
\end{example}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space. The following properties hold:

  \begin{enumerate}
    \item If $\nu:\Sigma\rightarrow\br{0,+\infty}$ is a measure on $\pr{X,\Sigma}$ and $\alpha,\beta\in\bpr{0,+\infty}$, then $\alpha\cdot\mu+\beta\cdot\nu$ is a measure on $\pr{X,\Sigma}$ where $0\cdot\pr{+\infty}=\pr{+\infty}\cdot 0=0$ in this context.
    \item If $S\in\Sigma$, then $\rest{\mu}{\Sigma_S}$ is a measure on $\pr{S,\Sigma_S}$.
    \item Finite additivity: If $A_1,A_2,\dots,A_n\in\Sigma$ with $n\in\mathbb{N}$ are pairwise disjoint $\Sigma$-measurable sets, then $\mu\pr{\bigcup_{k=1}^{n}{A_k}}=\sum_{k=1}^{n}{\mu\pr{A_k}}$.
    \item Monotonicity: If $A,B\in\Sigma$ are two $\Sigma$-measurable sets such that $A\subseteq B$, then $\mu\pr{A}\leq\mu\pr{B}$.
    \item Measure of the difference of sets: If $A,B\in\Sigma$ are two $\Sigma$-measurable sets such that $A\subseteq B$ and $\mu\pr{B}<+\infty$, then $\mu\pr{B\setminus A}=\mu\pr{B}-\mu\pr{A}$.
    \item $\sigma$-subadditivity: If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of $\Sigma$-measurable sets in $\Sigma$, then $\mu\pr{\bigcup_{n=1}^{\infty}{A_n}}\leq\sum_{n=1}^{\infty}{\mu\pr{A_n}}$.
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of subsets in $\Sigma$ such that $A_n\subseteq A_{n+1}$ for all $n\in\mathbb{N}$, then $\mu\pr{\bigcup_{n=1}^{\infty}{A_n}}=\lim_{n\to\infty}{\mu\pr{A_n}}$. 
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of subsets in $\Sigma$ such that $\mu\pr{A_1}<+\infty$ and $A_{n+1}\subseteq A_n$ for all $n\in\mathbb{N}$, then $\mu\pr{\bigcap_{n=1}^{\infty}{A_n}}=\lim_{n\to\infty}{\mu\pr{A_n}}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Firstly, $\alpha\cdot\mu\pr{\varnothing}+\beta\cdot\nu\pr{\varnothing}=0$.
      
      Let $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ be a countable collection of pairwise disjoint $\Sigma$-measurable sets. Then:

      \begin{equation}
        \begin{split}
          \alpha\cdot\mu\pr{\bigcup_{n=1}^{\infty}{A_n}}+\beta\cdot\nu\pr{\bigcup_{n=1}^{\infty}{A_n}} & =\alpha\cdot\sum_{n=1}^{\infty}{\mu\pr{A_n}}+\beta\cdot\sum_{n=1}^{\infty}{\nu\pr{A_n}}\\
          & =\sum_{n=1}^{\infty}{\pr{\alpha\cdot\mu\pr{A_n}+\beta\cdot\nu\pr{A_n}}}.
        \end{split}
      \end{equation}

      Therefore, $\alpha\cdot\mu+\beta\cdot\nu$ is a measure on $\pr{X,\Sigma}$.
      \item Trivial, since $\Sigma_S\subseteq\Sigma$.
      \item Let $A_m=\varnothing$ for all $m\in\mathbb{N}$ with $m>n$. Then $\set{A_m}_{m\in\mathbb{N}}\subseteq\Sigma$ is a collection of pairwise disjoint $\Sigma$-measurable sets and:
      
      \begin{equation}
        \begin{split}
          \mu\pr{\bigcup_{k=1}^{n}{A_k}} & =\mu\pr{\bigcup_{m=1}^{\infty}{A_m}}=\sum_{m=1}^{\infty}{\mu\pr{A_m}}=\sum_{k=1}^{n}{\mu\pr{A_k}}+\sum_{m=n+1}^{\infty}{\mu\pr{\varnothing}}\\
          & =\sum_{k=1}^{n}{\mu\pr{A_k}}.
        \end{split}
      \end{equation}

      \item Since $A\subseteq B$, $B=\pr{B\setminus A}\cup A$ and $\pr{B\setminus A}\cap A=\varnothing$, we have:
      
      \begin{equation}
        \mu\pr{B}=\mu\pr{B\setminus A}+\mu\pr{B\cap A}+\mu\pr{A}\geq 0 + \mu\pr{A}=\mu\pr{A}.
      \end{equation}

      \item Since $A\subseteq B$, $B=\pr{B\setminus A}\cup A$ and $\pr{B\setminus A}\cap A=\varnothing$, we have $\mu\pr{B}=\mu\pr{B\setminus A}+\mu\pr{B\cap A}+\mu\pr{A}$. By monotonicity, $\mu\pr{A}<+\infty$. Therefore, we can isolate $\mu\pr{B\setminus A}$ obtaining $\mu\pr{B\setminus A}=\mu\pr{B}-\mu\pr{A}$.
      \item Let $B_1:=A_1$ and $B_{n+1}=A_{n+1}\setminus\pr{A_1\cup\dots\cup A_n}$ for all $n\in\mathbb{N}$. Then $\set{B_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of pairwise disjoint $\Sigma$-measurable sets, $B_n\subseteq A_n$ for all $n\in\mathbb{N}$, $\bigcup_{n=1}^{\infty}{A_n}=\bigcup_{n=1}^{\infty}{B_n}$ and:
      
      \begin{equation}
        \mu\pr{\bigcup_{n=1}^{\infty}{A_n}}=\mu\pr{\bigcup_{n=1}^{\infty}{B_n}}=\sum_{n=1}^{\infty}{\mu\pr{B_n}}\leq\sum_{n=1}^{\infty}{\mu\pr{A_n}}.
      \end{equation}

      \item Let $B_1:=A_1$ and $B_{n+1}=A_{n+1}\setminus\pr{A_1\cup\dots\cup A_n}$ for all $n\in\mathbb{N}$. Then $\set{B_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of pairwise disjoint $\Sigma$-measurable sets, $B_n\subseteq A_n$ for all $n\in\mathbb{N}$, $\bigcup_{n=1}^{\infty}{A_n}=\bigcup_{n=1}^{\infty}{B_n}$, $A_n=\bigcup_{k=1}^{n}{B_k}$ for all $n\in\mathbb{N}$ and:
      
      \begin{equation}
        \begin{split}
          \mu\pr{\bigcup_{n=1}^{\infty}{A_n}} & =\mu\pr{\bigcup_{n=1}^{\infty}{B_n}}=\sum_{n=1}^{\infty}{\mu\pr{B_n}}=\lim_{n\to\infty}{\sum_{k=1}^{n}{\mu\pr{B_k}}}=\lim_{n\to\infty}{\mu\pr{\bigcup_{k=1}^{n}{B_k}}}\\
          & =\lim_{n\to\infty}{\mu\pr{A_n}}.
        \end{split}
      \end{equation}

      \item Let $B_n:=A_1\setminus A_n$ for all $n\in\mathbb{N}$. Since $\mu\pr{A_1}<+\infty$, we have $\mu\pr{B_n}=\mu\pr{A_1}-\mu\pr{A_n}$ for all $n\in\mathbb{N}$. Furthermore, $\set{B_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of $\Sigma$-measurable sets such that $B_n\subseteq B_{n+1}$ for all $n\in\mathbb{N}$. Additionally:
      
      \begin{equation}
        \mu\pr{\bigcup_{n=1}^{\infty}{\pr{A_1\setminus A_n}}}=\mu\pr{A_1\setminus\bigcap_{n=1}^{\infty}{A_n}}=\mu\pr{A_1}-\mu\pr{\bigcap_{n=1}{A_n}}.
      \end{equation}
      
      Thus:
      

      \begin{equation}
        \begin{split}
          \mu\pr{A_1}-\mu\pr{\bigcap_{n=1}{A_n}} & =\mu\pr{\bigcup_{n=1}^{\infty}{B_n}}=\lim_{n\to\infty}{\mu\pr{B_n}}=\lim_{n\to\infty}{\pr{\mu\pr{A_1}-\mu\pr{A_n}}}\\
          & =\mu\pr{A_1}-\lim_{n\to\infty}{\mu\pr{A_n}}.
        \end{split}
      \end{equation}

      Therefore $\mu\pr{\bigcap_{n=1}^{\infty}{A_n}}=\lim_{n\to\infty}{\mu\pr{A_n}}$.
    \end{enumerate}
  \end{proof}
\end{proposition}

\section{Measurable functions}

\begin{definition}
  Let $\pr{X,\Sigma_1}$ and $\pr{Y,\Sigma_2}$ be two measurable spaces and $Y'\subseteq Y$. We say a map $f:X\rightarrow Y'$ is \emph{$\pr{\Sigma_1,\Sigma_2}$-measurable} if $f^{-1}\pr{B}\in\Sigma_1$ for all $B\in\Sigma_2$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\Sigma_1}$ be a measurable space, $Y$ a set, $Y'\subseteq Y$, $\mathcal{S}\subseteq\mathcal{P}\pr{Y}$ a collection of subsets in $Y$, $\Sigma_2:=\sigma\pr{Y,\mathcal{S}}\subseteq\mathcal{P}\pr{Y}$ and $f:X\rightarrow Y'$ a map. Then $f$ is $\pr{\Sigma_1,\Sigma_2}$-measurable if and only if $f^{-1}\pr{S}\in\Sigma_1$ for all $S\in\mathcal{S}$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Clearly $f^{-1}\pr{S}\in\Sigma_1$ for all $S\in\mathcal{S}$, because $\mathcal{S}\subseteq\sigma\pr{Y,\mathcal{S}}=\Sigma_2$.
      \item[\boxed{\Leftarrow}] Let $\Sigma_2':=\set{B\middle|B\in\Sigma_2,f^{-1}\pr{B}\in\Sigma_1}$. Observe that $\mathcal{S}\subseteq\Sigma_2'$. Let's prove that $\Sigma_2'$ is a $\sigma$-algebra on $Y$.
      
      Clearly $\varnothing\in\Sigma_2'$, because $f^{-1}\pr{\varnothing}=\varnothing\in\Sigma_1$. Let $B\in\Sigma_2$. Then $f^{-1}\pr{B}\in\Sigma_1$. Thus $f^{-1}\pr{B^c}=\pr{f^{-1}\pr{B}}^c\in\Sigma_1$. Consequently, $B^c\in\Sigma_2'$.

      Let $\set{B_n}_{n\in\mathbb{N}}\subseteq\Sigma_2'$ be a countable collection of sets in $\Sigma_2'$. Then $f^{-1}\pr{B_n}\in\Sigma_1$ for all $n\in\mathbb{N}$. Thus $f^{-1}\pr{\bigcup_{n=1}^{\infty}{B_n}}=\bigcup_{n=1}^{\infty}{f^{-1}\pr{B_n}}\in\Sigma_1$. We deduce that $\bigcup_{n=1}^{\infty}{B_n}\in\Sigma_2'$.

      Therefore, $\Sigma_2'$ is a $\sigma$-algebra on $Y$. Since $\Sigma_2'$ contains $\mathcal{S}$, $\Sigma_2=\sigma\pr{Y,\mathcal{S}}\subseteq\Sigma_2'$ which means $f^{-1}\pr{B}\in\Sigma_1$ for all $B\in\Sigma_2$. In conclusion, $f$ is $\pr{\Sigma_1,\Sigma_2}$-measurable.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\Sigma_1}$ be a measurable space, $\pr{Y,\mathcal{T}}$ a topological space with a countable basis $\mathcal{B}\subseteq\mathcal{T}$, $\Sigma_2:=\mathscr{B}\pr{Y,\mathcal{T}}\subseteq\mathcal{P}\pr{Y}$ and $Y'\subseteq Y$. Then a map $f:X\rightarrow Y'$ is $\pr{\Sigma_1,\Sigma_2}$-measurable if and only if $f^{-1}\pr{B}\in\Sigma_1$ for all $B\in\mathcal{B}$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Clearly $f^{-1}\pr{B}\in\Sigma_1$ for all $B\in\mathcal{B}$, because $\mathcal{B}\subseteq\mathcal{T}\subseteq\mathscr{B}\pr{Y,\mathcal{T}}$.
      \item[\boxed{\Leftarrow}] Let $G\in\mathcal{T}$. Then there exists a countable collection of basis elements $\mathcal{B}'\subseteq\mathcal{B}$ such that $G=\bigcup_{B'\in\mathcal{B}'}{B'}$. Then $f^{-1}\pr{B'}\in\Sigma_1$ for all $B'\in\mathcal{B}'$. Thus $f^{-1}\pr{G}=f^{-1}\pr{\bigcup_{B'\in\mathcal{B}'}{B'}}=\bigcup_{B'\in\mathcal{B}'}{f^{-1}\pr{B'}}\in\Sigma_1$. Therefore $f$ is $\pr{\Sigma_1,\Sigma_2}$-measurable.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\Sigma_1}$, $\pr{Y,\Sigma_2}$ and $\pr{Z,\Sigma_3}$ be three measurable spaces, $Y'\subseteq Y$, $Z'\subseteq Z$, $f:X\rightarrow Y'$ a $\pr{\Sigma_1,\Sigma_2}$-measurable map and $g:Y\rightarrow Z'$ a $\pr{\Sigma_2,\Sigma_3}$-measurable map. Then $g\circ f$ is a $\pr{\Sigma_1,\Sigma_3}$-measurable map.
  \begin{proof}[\rm\textbf{Proof}]
    Let $C\in\Sigma_3$. Since $g$ is $\pr{\Sigma_2,\Sigma_3}$-measurable, we have $g^{-1}\pr{C}\in\Sigma_2$. As $f$ is $\pr{\Sigma_1,\Sigma_2}$-measurable, we have $\pr{g\circ f}^{-1}\pr{C}=f^{-1}\pr{g^{-1}\pr{C}}\in\Sigma_1$. Therefore, $g\circ f$ is $\pr{\Sigma_1,\Sigma_3}$-measurable.
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}_1}$ and $\pr{Y,\mathcal{T}_2}$ be two topological spaces, and $\Sigma\subseteq\mathcal{P}\pr{X}$ a $\sigma$-algebra on $X$ such that $\mathcal{T}_1\subseteq\Sigma$. If $f:X\rightarrow Y$ is a continuous map, then $f$ is $\pr{\Sigma,\mathscr{B}\pr{Y,\mathcal{T}_2}}$-measurable.
  \begin{proof}[\rm\textbf{Proof}]
    Let $G\in\mathcal{T}_2$. Then $f^{-1}\pr{G}\in\mathcal{T}_1\subseteq\Sigma$, since $f$ is continuous. Therefore $f$ is $\pr{\Sigma,\mathscr{B}\pr{Y,\mathcal{T}_2}}$-measurable.
  \end{proof}
\end{proposition}

\begin{definition}
  Let $\pr{X,\Sigma}$ be a measurable space and $Y'\subseteq\overline{\mathbb{R}}$. A function $f:X\rightarrow Y'$ is said to be \emph{$\Sigma$-measurable} if $f^{-1}\pr{G}\in\Sigma$ for every open set $G\in\mathcal{T}_e\pr{\overline{\mathbb{R}}}$, or, equivalently, if $f$ is $\pr{\Sigma,\mathscr{B}\pr{\mathbb{\overline{R}},\mathcal{T}_e\pr{\overline{\mathbb{R}}}}}$-measurable.
\end{definition}

Let $X$ be a set, $Y'\subseteq\overline{\mathbb{R}}$ and $f,g:X\rightarrow Y'$ two functions. Given $a,b\in\overline{\mathbb{R}}$, $\thicksim\,\in\set{=,\neq,<,>,\leq,\geq}$ and $\thicksim_1,\thicksim_2\,\in\set{<,\leq}$, we define the following sets:

\begin{equation}
    \begin{split}
      \set{f\thicksim a} & =\set{x\in X\middle| f\pr{x}\thicksim a},\\
      \set{a\thicksim f} & =\set{x\in X\middle| a\thicksim f\pr{x}},\\
      \set{f\thicksim g} & =\set{x\in X\middle| f\pr{x}\thicksim g\pr{x}},\\
      \set{a\thicksim_1 f\thicksim_2 b} & =\set{x\in X\middle| a\thicksim_1 f\pr{x},f\pr{x}\thicksim_2 b}.
    \end{split}
\end{equation}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a function. The following are equivalent:

\begin{enumerate}
    \item $f$ is $\Sigma$-measurable.
    \item For all $a\in\mathbb{R}$, $\set{f<a}\in\Sigma$.
    \item For all $a\in\mathbb{R}$, $\set{f\leq a}\in\Sigma$.
    \item For all $a\in\mathbb{R}$, $\set{f>a}\in\Sigma$.
    \item For all $a\in\mathbb{R}$, $\set{f\geq a}\in\Sigma$.
\end{enumerate}
\begin{proof}[\rm\textbf{Proof}]
\hfill
\begin{itemize}
    \item[\boxed{1\Rightarrow2}] Let $a\in\mathbb{R}$ be fixed. Let's observe that $\set{f<a}=f^{-1}\pr{\bpr{-\infty,a}}$. Since $\bpr{-\infty,a}\in\mathcal{T}_e\pr{\overline{\mathbb{R}}}$, then $\set{f<a}\in\Sigma$.
    \item[\boxed{2\Rightarrow3}] Let $a\in\mathbb{R}$ be fixed. Firstly, $\set{f\leq a}=\bigcap_{n=1}^{\infty}{\set{f<a+\frac{1}{n}}}$. We know that $\set{f<a+\frac{1}{n}}\in\Sigma$ for all $n\in\mathbb{N}$. Therefore $\set{f\leq a}\in\Sigma$.
    \item[\boxed{3\Rightarrow4}] It suffices to observe that $\set{f>a}=\pr{\set{f\leq a}}^c$ for all $a\in\mathbb{R}$.
    \item[\boxed{4\Rightarrow5}] Let $a\in\mathbb{R}$ be fixed. Firstly, $\set{f\geq a}=\bigcap_{n=1}^{\infty}{\set{f>a-\frac{1}{n}}}$. Since $\set{f>a-\frac{1}{n}}\in\Sigma$ for all $n\in\mathbb{N}$, then $\set{f\geq a}\in\Sigma$.
    \item[\boxed{5\Rightarrow2}] It is enough to note that $\set{f<a}=\pr{\set{f\geq a}}^c$ for all $a\in\mathbb{R}$.
    \item[\boxed{2\Rightarrow1}] Let $G\in\mathcal{T}_e\pr{\overline{\mathbb{R}}}$. Then there exists a countable collection $\set{I_n}_{n\in\mathbb{N}}\subseteq\mathcal{T}_e\pr{\overline{\mathbb{R}}}$ of non-empty open intervals such that $G=\bigcup_{n=1}^{\infty}{I_n}$. We have proved that properties 2, 3, 4 and 5 are equivalent. Then $\set{a<f<b}=\set{a<f}\cap\set{f<b}\in\Sigma$ for all $a,b\in\mathbb{R}$ with $a<b$. Thus $f^{-1}\pr{I_n}\in\Sigma$ for all $n\in\mathbb{N}$. Consequently, $f^{-1}\pr{G}=f^{-1}\pr{\bigcup_{n=1}^{\infty}{I_n}}=\bigcup_{n=1}^{\infty}{f^{-1}\pr{I_n}}\in\Sigma$. Therefore, $f$ is $\Sigma$-measurable.
\end{itemize}
\end{proof}
\end{proposition}

\begin{corollary}
\rm Let $\pr{X,\Sigma}$ be a measurable space, $a,b\in\overline{\mathbb{R}}$, $\thicksim,\thicksim_1,\thicksim_2\,\in\set{<,\leq}$ and $f:X\rightarrow\overline{\mathbb{R}}$ a $\Sigma$-measurable function. Then $\set{f=a},\set{f\neq a},\set{f\thicksim +\infty},\set{-\infty\thicksim f},\set{a\thicksim_1 f\thicksim_2 b}\in\Sigma$.
\begin{proof}[\rm\textbf{Proof}]
It is enough to note that:

\begin{equation}
  \begin{split}
    \set{f=a} & =\set{f\leq a}\cap\set{f\geq a},\\
    \set{f\neq a} & =\set{f=a}^c,\\
    \set{f=+\infty} & =\bigcap_{n=1}^{\infty}{\set{f>n}},\\
    \set{f=-\infty} & =\bigcap_{n=1}^{\infty}{\set{f<-n}},\\
    \set{f<+\infty} & =\bigcup_{n=1}^{\infty}{\set{f<n}},\\
    \set{f>-\infty} & =\bigcup_{n=1}^{\infty}{\set{f>-n}},\\
    \set{f\leq +\infty} & =\set{f\geq -\infty}=\set{-\infty\leq f\leq +\infty}=X,\\
    \set{a\thicksim_1 f\thicksim_2 b} & =\set{a\thicksim_1 f}\cap\set{f\thicksim b}.
  \end{split}
\end{equation}
\end{proof}
\end{corollary}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space, $Y',Y''\subseteq\overline{\mathbb{R}}$ and, $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two $\Sigma$-measurable functions. Then $\set{f=g},\set{f\neq g},\set{f<g},\set{f\leq g},\set{f>g},\set{f\geq g}\in\Sigma$.

  \begin{proof}[\rm\textbf{Proof}]
    Given $x\in X$, it is very easy to prove that $f\pr{x}<g\pr{x}$ if and only if there exists $q\in\mathbb{Q}$ such that $f\pr{x}<q<g\pr{x}$. Given $a\in\mathbb{R}$, using that $\mathbb{Q}$ is a countable set and, $f$ and $g$ are $\Sigma$-measurable, we have:

    \begin{equation}
      \set{f<g}=\bigcup_{q\in\mathbb{Q}}{\set{f<q<g}}=\bigcup_{q\in\mathbb{Q}}{\pr{\set{f<q}\cap\set{q<g}}}\in\Sigma.
    \end{equation}

    For the rest of sets, it is enough to observe that:

    \begin{equation}
      \begin{split}
        \set{f\leq g}=\bigcap_{n=1}^{\infty}{\set{f<g+\frac{1}{n}}}\in\Sigma,\\
        \set{f>g}=\set{g<f}\in\Sigma,\\
        \set{f\geq g}=\set{g\leq f}\in\Sigma,\\
        \set{f=g}=\set{f\leq g}\cap\set{f\geq g}\in\Sigma,\\
        \set{f\neq g}=\set{f=g}^c\in\Sigma.
      \end{split}
    \end{equation}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space, $Y',Y'',Y'''\subseteq\overline{\mathbb{R}}$, $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two $\Sigma$-measurable functions, and $\set{f_n}_{n\in\mathbb{N}}\subseteq\pr{Y'''}^X$ a sequence of $\Sigma$-measurable functions. The following properties hold: % chktex 23

  \begin{enumerate}
    \item If $A\in\Sigma$, then $\rest{f}{A}:A\rightarrow Y'$ is $\Sigma_A$-measurable.
    \item $\alpha\cdot f:X\rightarrow\overline{\mathbb{R}}$ is $\Sigma$-measurable, where $0\cdot\pr{\pm\infty}=\pr{\pm\infty}\cdot 0=0$ in this context.
    \item If $\set{f=+\infty}\cap\set{g=-\infty},\set{f=-\infty}\cap\set{g=+\infty}=\varnothing$, then $f+g:X\rightarrow\overline{\mathbb{R}}$ is $\Sigma$-measurable.
    \item $\max\set{f,g}, \min\set{f,g}, f^+, f^-, \bars{f}:X\rightarrow\overline{\mathbb{R}}$ are $\Sigma$-measurable.
    \item If $\set{\bars{f}=+\infty}\cap\set{g=0},\set{f=0}\cap\set{\bars{g}=+\infty}=\varnothing$, then $f\cdot g:X\rightarrow\overline{\mathbb{R}}$ is $\Sigma$-measurable.
    \item If $\set{\bars{f}=+\infty}\cap\set{\bars{g}=+\infty},\set{g=0}=\varnothing$, then $\frac{f}{g}:X\rightarrow\overline{\mathbb{R}}$ is $\Sigma$-measurable.
    \item $\sup_{n\in\mathbb{N}}{f_n}, \inf_{n\in\mathbb{N}}{f_n}, \liminf_{n\to\infty}{f_n}, \limsup_{n\to\infty}{f_n}:X\rightarrow\overline{\mathbb{R}}$ are $\Sigma$-measurable.
    \item If $L:=\set{x\middle|x\in X,\exists\lim_{n\to\infty}{f_n}\in\overline{\mathbb{R}}}$, then $L\in\Sigma$ and $\lim_{n\to\infty}{f_n}:L\rightarrow\overline{\mathbb{R}}$ is $\Sigma_L$-measurable.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $a\in\mathbb{R}$. Since $A\in\Sigma$,  $\set{\rest{f}{A}<a}=\set{f<a}\cap A\in\Sigma_A$. Therefore, $\rest{f}{A}$ is $\Sigma_A$-measurable.
      \item Let $a\in\mathbb{R}$. If $\alpha=0$, then $\alpha\cdot f\equiv 0$ and, $\set{\alpha\cdot f\leq a}=\varnothing\in\Sigma$ if $a<0$ and $\set{\alpha\cdot f\leq a}=X\in\Sigma$ if $a\geq 0$. Suppose $\alpha\neq 0$. Then $\set{\alpha\cdot f\leq a}=\set{f\leq\frac{a}{\alpha}}\in\Sigma$ if $\alpha>0$ and $\set{\alpha\cdot f\leq a}=\set{f\geq\frac{a}{\alpha}}\in\Sigma$ if $\alpha<0$. Therefore $\alpha\cdot f$ is $\Sigma$-measurable.
      \item Let $a\in\mathbb{R}$. Let $b,c\in\overline{\mathbb{R}}$ such that if $b=-\infty$ then $c\neq+\infty$ and if $b=+\infty$ then $c\neq-\infty$. Let's first prove that $b+c<a$ if and only if $b<a-c$. 
      
      If $b,c\in\mathbb{R}$, the proof is trivial. If $b=-\infty$, then $c\in\mathbb{R}\cup\set{-\infty}$, $a-c\in\mathbb{R}\cup\set{+\infty}$, $b+c=-\infty<a$ and $b=-\infty<a-c$. If $b=+\infty$, then $c\in\mathbb{R}\cup\set{+\infty}$, $a-c\in\mathbb{R}\cup\set{-\infty}$, $b+c=+\infty\not<a$ and $b=+\infty\not<a-c$. This proves that $b+c<a$ if and only if $b<a-c$.
      
      Given $x\in X$, we have $f\pr{x}+g\pr{x}<a$ if and only if $f\pr{x}<a-g\pr{x}$. As $\mathbb{Q}$ is dense in $\mathbb{R}$, the inequality $f\pr{x}<a-g\pr{x}$ is equivalent to the existence of $q\in\mathbb{Q}$ such that $f\pr{x}<q<a-g\pr{x}$. In other words, $f\pr{x}+g\pr{x}<a$ if and only if there exists $q\in\mathbb{Q}$ such that $f\pr{x}<q$ and $g\pr{x}<a-q$. Since $\mathbb{Q}$ is a countable set, we have:

      \begin{equation}
        \begin{split}
          \set{f+g<a} & =\set{f<a-g}=\bigcup_{q\in\mathbb{Q}}{\set{f<q<a-g}}\\
          & =\bigcup_{q\in\mathbb{Q}}{\pr{\set{f<q}\cap\set{g<a-q}}}\in\Sigma.
        \end{split}
      \end{equation}

      Therefore, $f+g$ is $\Sigma$-measurable.
      \item Let $a\in\mathbb{R}$. Then:
      
      \begin{equation}
        \set{\max\set{f,g}\leq a}=\set{f\leq a}\cap\set{g\leq a}\in\Sigma.
      \end{equation}

      Thus $\max\set{f,g}$ is $\Sigma$-measurable. For the rest of functions, it is enough to notice that:

      \begin{equation}
        \begin{split}
          \min\set{f,g} & =-\max\set{-f,-g},\\
          f^+ & =\max\set{f,0},\\
          f^- & =\max\set{-f,0},\\
          \bars{f} & =f^++f^-.
        \end{split}
      \end{equation}

      \item Let $a\in\mathbb{R}$. Suppose that $f$ and $g$ are non-negative $\Sigma$-measurable functions. If $a\leq 0$, then $\set{f\cdot g<a}=\varnothing$. Suppose that $a>0$. Given $x\in X$, by the density of $\mathbb{Q}$ in $\mathbb{R}$, it is easy to check that $f\pr{x}\cdot g\pr{x}<a$ if and only if there exists $q\in\mathbb{Q}$ with $q>0$ such that $f\pr{x}<q$, $g\pr{x}<\frac{a}{q}$. Then:
      
      \begin{equation}
        \set{f\cdot g<a}=\bigcup_{q\in\mathbb{Q},q>0}{\pr{\set{f<q}\cap\set{g<\frac{a}{q}}}}\in\Sigma.
      \end{equation}

      Therefore $f\cdot g$ is $\Sigma$-measurable. If $f$ and $g$ are $\Sigma$-measurable functions in general, we can see that:

      \begin{equation}
        f\cdot g=\pr{f^+-f^-}\cdot\pr{g^+-g^-}=f^+\cdot g^+-f^+\cdot g^--f^-\cdot g^++f^-\cdot g^-.
      \end{equation}

      In conclusion, $f\cdot g$ is $\Sigma$-measurable.
      \item Since $\frac{f}{g}=f\cdot\frac{1}{g}$, it suffices to show that $\frac{1}{g}$ is $\Sigma$-measurable. Let $a\in\mathbb{R}$. Then:
      
      \begin{equation}
        \set{\frac{1}{g}<a}=\begin{cases}
          \set{g<0}\cap\set{g>\frac{1}{a}} & \text{if $a<0$}\\
          \set{g<0}\cap\set{g\neq-\infty} & \text{if $a=0$}\\
          \set{g<0}\cup\set{g>\frac{1}{a}} & \text{if $a>0$}
        \end{cases}
      \end{equation}

      In any case, $\set{\frac{1}{g}<a}\in\Sigma$. Consequently, $\frac{1}{g}$ is $\Sigma$-measurable. Therefore $\frac{f}{g}$ is $\Sigma$-measurable.
      \item Let $a\in\mathbb{R}$. Given $x\in X$, we have that $\sup_{n\in\mathbb{N}}{f_n\pr{x}}\leq a$ if and only if $f_n\pr{x}\leq a$ for all $n\in\mathbb{N}$. Then:
      
      \begin{equation}
        \set{\sup_{n\in\mathbb{N}}{f_n}\leq a}=\bigcap_{n=1}^{\infty}{\set{f_n\leq a}}\in\Sigma.
      \end{equation}

      For the rest of functions, it is enough to notice that:

      \begin{equation}
        \begin{split}
          \inf_{n\in\mathbb{N}}{f_n} & =-\sup_{n\in\mathbb{N}}{\pr{-f_n}},\\
          \liminf_{n\to\infty}{f_n} & =\sup_{n\in\mathbb{N}}{\inf_{m\in\mathbb{N},m\geq n}{f_m}},\\
          \limsup_{n\to\infty}{f_n} & =\inf_{n\in\mathbb{N}}{\sup_{m\in\mathbb{N},m\geq n}{f_m}}.
        \end{split}
      \end{equation}
      \item Since $L=\set{\liminf_{n\to\infty}{f_n}=\limsup_{n\to\infty}{f_n}}$ and $\liminf_{n\to\infty}{f_n}$ and $\limsup_{n\to\infty}{f_n}$ are $\Sigma$-measurable, then $L\in\Sigma$. The function $\lim_{n\to\infty}{f_n}$ is well-defined on $L$ and equal to $\liminf_{n\to\infty}{f_n}=\limsup_{n\to\infty}{f_n}$ on $L$. Therefore, $\lim_{n\to\infty}{f_n}$ is $\Sigma_L$-measurable.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{definition}
  Let $\pr{X,\Sigma}$ be a measurable space and $Y'\subseteq\mathbb{C}$. A function $f:X\rightarrow Y'$ is said to be $\Sigma$-measurable if $f^{-1}\pr{G}\in\Sigma$ for every open set $G\in\mathcal{T}_e\pr{\mathbb{C}}$, or, equivalently, if $f$ is $\pr{\Sigma,\mathscr{B}\pr{\mathbb{C}}}$-measurable.
\end{definition}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space, $Y'\subseteq\mathbb{C}$ and $f:X\rightarrow Y'$ a map. Then $f$ is $\Sigma$-measurable if and only if $\Re\pr{f}$ and $\Im\pr{f}$ are real $\Sigma$-measurable functions.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Let $a\in\mathbb{R}$. Then:
      
      \begin{equation}
        \begin{split}
          \set{\Re\pr{f}<a} & =f^{-1}\pr{\pr{-\infty,a}+\mathbb{R}\cdot i},\\
          \set{\Im\pr{f}<a} & =f^{-1}\pr{\mathbb{R}+\pr{-\infty,a}\cdot i},
        \end{split}
      \end{equation}

      where $\pr{-\infty,a}+\mathbb{R}\cdot i:=\set{z\in\mathbb{C}\middle|\Re\pr{z}<a}$ and $\mathbb{R}+\pr{-\infty,a}\cdot i:=\set{z\in\mathbb{C}\middle|\Im\pr{z}<a}$. We know that $\pr{-\infty,a}+\mathbb{R}\cdot i,\mathbb{R}+\pr{-\infty,a}\cdot i\in\mathcal{T}_e\pr{\mathbb{C}}$. Since, $f$ is $\Sigma$-measurable, we have $\set{\Re\pr{f}<a},\set{\Im\pr{f}<a}\in\Sigma$. Therefore, $\Re\pr{f}$ and $\Im\pr{f}$ are real $\Sigma$-measurable functions.
      \item[\boxed{\Leftarrow}] Let $a,b,c,d\in\mathbb{R}$ with $a<b$ and $c<d$, define the following set:
      
      \begin{equation}
        \pr{a,b}+\pr{c,d}\cdot i:=\set{z\in\mathbb{C}\middle|a<\Re\pr{z}<b,c<\Im\pr{z}<d}.
      \end{equation}
      
      Then:

      \begin{equation}
        f^{-1}\pr{\pr{a,b}+\pr{c,d}\cdot i}=\Re\pr{f}^{-1}\pr{\pr{a,b}}\cap\Im\pr{f}^{-1}\pr{\pr{c,d}}.
      \end{equation}

      Since $\Re\pr{f}$ and $\Im\pr{f}$ are $\Sigma$-measurable functions, $f^{-1}\pr{\pr{a,b}+\pr{c,d}\cdot i}\in\Sigma$. Therefore, $f$ is a $\Sigma$-measurable function.
    \end{itemize}
  \end{proof}
\end{proposition}


\begin{corollary}
  \rm Let $\pr{X,\Sigma}$ be a measurable space, $Y',Y'',Y'''\subseteq\mathbb{C}$,  $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two $\Sigma$-measurable functions, and $\set{f_n}_{n\in\mathbb{N}}\subseteq\pr{Y'''}^X$ a sequence of $\Sigma$-measurable functions. The following properties hold: % chktex 23

  \begin{enumerate}
    \item If $A\in\Sigma$, then $\rest{f}{A}$ is $\Sigma_A$-measurable.
    \item If $\alpha\in\mathbb{C}$, the functions $\overline{f},\alpha\cdot f,f+g,f\cdot g:X\rightarrow\mathbb{C}$ are $\Sigma$-measurable. Furthermore, if $\set{g=0}=\varnothing$, then $\frac{f}{g}:X\rightarrow\mathbb{C}$ is $\Sigma$-measurable.
    \item $\bars{f}:X\rightarrow\mathbb{R}$ is $\Sigma$-measurable.
    \item If $L:=\set{x\middle|x\in X,\exists\lim_{n\to\infty}{f_n}\in\mathbb{C}}$, then $L\in\Sigma$ and $\lim_{n\to\infty}{f_n}:L\rightarrow\mathbb{C}$ is $\Sigma_L$-measurable.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $u,v:X\rightarrow\mathbb{R}$ be the functions such that $f=u+v\cdot i$. Since $f$ is $\Sigma$-measurable, then $u$ and $v$ are $\Sigma$-measurable. Using that $A\in\Sigma$, we deduce that $\rest{u}{A}$ and $\rest{v}{A}$ are $\Sigma_A$-measurable. Therefore, $\rest{f}{A}=\rest{u}{A}+\rest{v}{A}\cdot i$ is $\Sigma_A$-measurable.
      \item Let $a,b\in\mathbb{R}$ and $u,v,u',v':X\rightarrow\mathbb{R}$ such that $\alpha=a+b\cdot i$, $f=u+v\cdot i$ and $g=u'+v'\cdot i$. Then:
      
      \begin{equation}
        \begin{split}
          \overline{f} & =u-v\cdot i=u+\pr{-v}\cdot i,\\
          \alpha\cdot f & =\pr{a+b\cdot i}\cdot\pr{u+v\cdot i}=\pr{a\cdot u-b\cdot v}+\pr{a\cdot v+b\cdot u}\cdot i,\\
          f+g &=\pr{u+v\cdot i}+\pr{u'+v'\cdot i}=\pr{u+u'}+\pr{v+v'}\cdot i,\\
          f\cdot g & =\pr{u+v\cdot i}\cdot\pr{u'+v'\cdot i}=\pr{u\cdot u'-v\cdot v'}+\pr{u\cdot v'+v\cdot u'}\cdot i.
        \end{split}
      \end{equation}

      If $\set{g=0}=\varnothing$, we also have:

      \begin{equation}
        \begin{split}
          \frac{f}{g} & =\frac{f\cdot\overline{g}}{\bars{g}^2}=\frac{\pr{u\cdot u'+v\cdot v'}+\pr{-u\cdot v'+v\cdot u'}\cdot i}{\pr{u'}^2+\pr{v'}^2}\\
          &=\frac{u\cdot u'+v\cdot v'}{\pr{u'}^2+\pr{v'}^2}+\frac{-u\cdot v'+v\cdot u'}{\pr{u'}^2+\pr{v'}^2}\cdot i.
        \end{split}
      \end{equation}
      
      Since $f$ is $\Sigma$-measurable, then $u$, $v$, $u'$ and $v'$ are $\Sigma$-measurable. Then the real and imaginary parts of the functions above are $\Sigma_L$-measurable and, consequently, the above functions themselves are $\Sigma_L$-measurable.

      \item Let $a\in\mathbb{R}$ and $u,v:X\rightarrow\mathbb{R}$ such that $f=u+v\cdot i$. Since $f$ is $\Sigma$-measurable, then $u$ and $v$ are $\Sigma$-measurable. If $a\leq 0$, then $\set{\bars{f}<a}=\varnothing$. If $a>0$, then $\set{\bars{f}<a}=\set{\bars{f}^2<a^2}=\set{u^2+v^2<a^2}\in\Sigma$, because $u$ and $v$ are $\Sigma$-measurable. Therefore, $\bars{f}$ is $\Sigma$-measurable.
      \item For each $n\in\mathbb{N}$, let $u_n,v_n:X\rightarrow\mathbb{R}$ be the functions such that $f_n=u_n+v_n\cdot i$. Given $x\in X$, we have that $\lim_{n\to\infty}{f_n\pr{x}}$ exists in $\mathbb{C}$ if and only if the limits $\lim_{n\to\infty}{u_n\pr{x}}$ and $\lim_{n\to\infty}{v_n\pr{x}}$ exist in $\mathbb{R}$. Furthermore, for all $x\in L$, we have that $\lim_{n\to\infty}{f_n\pr{x}}=\lim_{n\to\infty}{u_n\pr{x}}+\pr{\lim_{n\to\infty}{v_n\pr{x}}}\cdot i$. Let $f:L\rightarrow\mathbb{C}$ and $u,v:L\rightarrow\mathbb{R}$ be the functions such that $f:=\lim_{n\to\infty}{f_n}$, $u:=\lim_{n\to\infty}{u_n}$ and $v:=\lim_{n\to\infty}{v_n}$. Then:
      
      \begin{equation}
        \begin{split}
          L & =\set{x\middle|x\in X,\exists\lim_{n\to\infty}{u_n}\in\mathbb{R}}\cap\set{x\middle|x\in X,\exists\lim_{n\to\infty}{v_n}\in\mathbb{R}}\\
          & =\set{x\middle|x\in X,\liminf_{n\to\infty}{u_n}=\limsup_{n\to\infty}{u_n}}\cap u^{-1}\pr{\mathbb{R}}\\
          & \cap\set{x\middle|x\in X,\liminf_{n\to\infty}{v_n}=\limsup_{n\to\infty}{v_n}}\cap v^{-1}\pr{\mathbb{R}}.
        \end{split}
      \end{equation}

      Since $u_n$ and $v_n$ are $\Sigma$-measurable for all $n\in\mathbb{N}$, then $L\in\Sigma$ and $u$ and $v$ are $\Sigma_L$-measurable. Therefore, $f=u+v\cdot i$ is $\Sigma_L$-measurable.
    \end{enumerate}
  \end{proof}
\end{corollary}

\subsection{Simple Function Approximation Theorem}

\begin{definition}
  Let $X$ be a set and $A\subseteq X$. We define the \textit{characteristic function of $A$} as the map $\chi_A:X\rightarrow\set{0,1}$ given by $\chi_A\pr{x}=1$ if $x\in A$ and $\chi_A\pr{x}=0$ if $x\notin A$.
\end{definition}

\begin{proposition}
  Let $X$ be a set. The following properties holds:
  \begin{enumerate}
    \item If $A,B\subseteq X$, Then $\chi_{A\cap B}\pr{x}=\chi_A\pr{x}\cdot\chi_B\pr{x}$ and $\chi_{A\cup B}\pr{x}=\chi_A\pr{x}+\chi_B\pr{x}-\chi_{A\cap B}\pr{x}$. Particulary, if $A$ and $B$ are disjoint, then $\chi_{A\cup B}=\chi_A+\chi_B$.
    \item If $A_1,A_2,\dots,A_n\subseteq X$ with $n\in\mathbb{N}$ are pairwise disjoint sets, then $\chi_{\bigcup_{k=1}^{n}{A_k}}=\sum_{k=1}^{n}{\chi_{A_k}}$.
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\mathcal{P}\pr{X}$ is a countable collection of pairwise disjoint subsets of $X$, then $\chi_{\bigcup_{n=1}^{\infty}{A_n}}=\sum_{n=1}^{\infty}{\chi_{A_n}}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $x\in X$. We can distinguish three cases:
      \begin{itemize}
        \item If $x\notin A\cup B$, then $\chi_{A\cap B}\pr{x}=\chi_{A\cup B}\pr{x}=\chi_A\pr{x}=\chi_B\pr{x}=0$, $\chi_A\pr{x}\cdot\chi_B\pr{x}=0$ and $\chi_A\pr{x}+\chi_B\pr{x}-\chi_{A\cap B}\pr{x}=0$.
        \item If $x\in A\setminus B$ (resp. $x\in B\setminus A$), then $\chi_{A\cup B}\pr{x}=1$, $\chi_{A\cap B}\pr{x}=0$, $\chi_A\pr{x}=1$ (resp. $\chi_A\pr{x}=0$), $\chi_B\pr{x}=0$ (resp. $\chi_B\pr{x}=1$), $\chi_A\pr{x}\cdot\chi_B\pr{x}=0$ and $\chi_A\pr{x}+\chi_B\pr{x}-\chi_{A\cap B}\pr{x}=1$.
        \item If $x\in A\cap B$, then $\chi_{A\cap B}\pr{x}=\chi_{A\cup B}\pr{x}=\chi_A\pr{x}=\chi_B\pr{x}=1$, $\chi_A\pr{x}\cdot\chi_B\pr{x}=1$ and $\chi_A\pr{x}+\chi_B\pr{x}-\chi_{A\cap B}\pr{x}=1$.
      \end{itemize}

      Therefore, $\chi_{A\cap B}\pr{x}=\chi_A\pr{x}\cdot\chi_B\pr{x}$ and $\chi_{A\cup B}\pr{x}=\chi_A\pr{x}+\chi_B\pr{x}-\chi_{A\cap B}\pr{x}$.
      \item Let $x\in X$. If $x\notin\bigcup_{k=1}^{n}{A_k}$, then $\chi_{\bigcup_{k=1}^{n}{A_k}}\pr{x}=0$ and $\chi_{A_k}\pr{x}=0$ for all $k\in\set{1,2,\dots, n}$. Then $\sum_{k=1}^{n}{\chi_{A_k}\pr{x}}=0$. If $x\in\bigcup_{k=1}^{n}{A_k}$, there exists $k_0\in\set{1,2,\dots,n}$ such that $x\in A_{k_0}\setminus A_k$ for all $k\in\set{1,2,\dots,n}\setminus\set{k_0}$. Thus $\chi_{\bigcup_{k=1}^{n}{A_k}}\pr{x}=1$, $\chi_{A_{k_0}}\pr{x}=1$ and $\chi_{A_k}\pr{x}=0$ for all $k\in\set{1,2,\dots,n}\setminus\set{k_0}$. Consequently, $\sum_{k=1}^{n}{\chi_{A_k}\pr{x}}=1$. Therefore, $\chi_{\bigcup_{k=1}^{n}{A_k}}=\sum_{k=1}^{n}{\chi_{A_k}}$.
      \item Let $x\in X$. If $x\notin\bigcup_{n=1}^{\infty}{A_k}$, then $\chi_{\bigcup_{n=1}^{\infty}{A_n}}\pr{x}=0$ and $\chi_{A_n}\pr{x}=0$ for all $n\in\mathbb{N}$. Then $\sum_{n=1}^{\infty}{\chi_{A_n}\pr{x}}=0$. If $x\in\bigcup_{n=1}^{\infty}{A_n}$, there exists $n_0\in\mathbb{N}$ such that $x\in A_{n_0}\setminus A_n$ for all $n\in\mathbb{N}\setminus\set{n_0}$. Thus $\chi_{\bigcup_{n=1}^{\infty}{A_n}}\pr{x}=1$, $\chi_{A_{n_0}}\pr{x}=1$ and $\chi_{A_n}\pr{x}=0$ for all $n\in\mathbb{N}\setminus\set{n_0}$. Consequently, $\sum_{n=1}^{\infty}{\chi_{A_n}\pr{x}}=1$. Therefore, $\chi_{\bigcup_{n=1}^{\infty}{A_n}}=\sum_{n=1}^{\infty}{\chi_{A_n}}$.
    \end{enumerate}
    
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space and $A\subseteq X$. Then $\chi_A$ is $\Sigma$-measurable if and only if $A\in\Sigma$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Since $\chi_A$ is $\Sigma$-measurable, $A=\chi_A^{-1}\pr{\set{1}}\in\Sigma$.
      \item[\boxed{\Leftarrow}] Since $A\in\Sigma$, $A^c\in\Sigma$. Let $a\in\mathbb{R}$. Then:
      
      \begin{equation}
        \set{\chi_A<a}=\begin{cases}
          \varnothing & \text{if $a\leq 0$}\\
          A^c & \text{if $0<a<1$} \\
          A & \text{if $a\geq 1$}
        \end{cases}
      \end{equation}

      In any case, $\set{\chi_A<a}\in\Sigma$. Therefore, $\chi_A$ is $\Sigma$-measurable.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{definition}
  Let $X$ be a set and $Y'\subseteq\mathbb{R}$. A function $\varphi:X\rightarrow Y'$ is said to be \textit{simple} if $\im\pr{\varphi}$ is finite.
\end{definition}

\begin{proposition}
  Let $X$ be a set, $Y\subseteq\mathbb{R}$ and $\varphi:X\rightarrow Y$ a map. Then $\varphi$ is simple if and only if there exists $A_1,A_2,\dots,A_n\subseteq X$ and $a_1,a_2,\dots,a_n\in\mathbb{R}$ with $n\in\mathbb{N}$ such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Since $\im\pr{\varphi}$ is a finite set, we have:
      
      \begin{equation}
        \varphi=\sum_{a\in\im\pr{\varphi}}{a\cdot\chi_{\set{\varphi=a}}}.
      \end{equation}
      \item[\boxed{\Leftarrow}] Let's observe that:

      \begin{equation}
        \im\pr{\varphi}=\set{\sum_{k=1}^{n}{a_k\cdot\delta_k}\middle|\delta_k\in\set{0,1},\forall k\in\set{1,\dots, n}}.
      \end{equation}

      Then $\im\pr{\varphi}$ is a finite set with at most $2^n$ elements. Therefore, $\varphi$ is simple.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space, $Y'\subseteq\mathbb{R}$ and $\varphi:X\rightarrow Y'$ a simple function. The following are equivalent:

  \begin{enumerate}
    \item $\varphi$ is $\Sigma$-measurable.
    \item For all $a\in\mathbb{R}$, $\set{\varphi=a}\in\Sigma$.
    \item There exists $A_1,A_2,\dots,A_n\in\Sigma$ and $a_1,a_2,\dots,a_n$ with $n\in\mathbb{N}$ such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$.

  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Trivial.
      \item[\boxed{2\Rightarrow3}] It suffices to see that $\varphi=\sum_{a\in\im\pr{\varphi}}{a\cdot\chi_{\set{\varphi=a}}}$.
      \item[\boxed{3\Rightarrow1}] Since $A_k\in\Sigma$ for all $k\in\set{1,2,\dots,n}$, $\chi_{A_k}$ is $\Sigma$-measurable for all $k\in\set{1,2,\dots,n}$. Therefore, $\varphi$ is $\Sigma$-measurable.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{theorem}[\textbf{Simple Function Approximation Theorem}]
  \rm Let $\pr{X,\Sigma}$ be a measurable space, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a non-negative $\Sigma$-measurable function. Then there exists a sequence of simple non-negative $\Sigma$-measurable functions $\set{\varphi_n}_{n\in\mathbb{N}}\subseteq\mathbb{R}^X$ such that $0\leq\varphi_n\pr{x}\leq\varphi_{n+1}\pr{x}\leq f\pr{x}$ for all $n\in\mathbb{N}$ and $\lim_{n\to\infty}{\varphi_n\pr{x}}=f\pr{x}$ for all $x\in X$. Furthermore, if $f$ is bounded, then we can construct the sequence of simple non-negative $\Sigma$-measurable functions $\set{\varphi_n}_{n\in\mathbb{N}}\subseteq\mathbb{R}^X$ such that the convergence $\lim_{n\to\infty}{\varphi_n}=f$ is uniform.

  \begin{proof}[\rm\textbf{Proof}]
    Given $n\in\mathbb{N}$, we define $A_{n,k}:=\set{\frac{k-1}{2^n}\leq f<\frac{k}{2^n}}$ for each $k\in\set{1,2,\dots, n\cdot 2^n}$ and $B_n:=\set{f\geq n}$. For each $n\in\mathbb{N}$, we define $\varphi_n:X\rightarrow\mathbb{R}$ as follows:

    \begin{equation}
      \varphi_n:=\sum_{k=1}^{n\cdot 2^n}{\dfrac{k-1}{2^n}}\cdot\chi_{A_{n,k}}+n\cdot\chi_{B_n}.
    \end{equation}

    Clearly $\varphi_n$ is a non-negative simple function for all $n\in\mathbb{N}$. Since $f$ is $\Sigma$-measurable, $A_{n,k},B_n\in\Sigma$ for each $n\in\mathbb{N}$ and $k\in\set{1,2,\dots,n\cdot 2^n}$ and, consequently, $\varphi_n$ would be $\Sigma$-measurable.

    Let's prove that $\varphi_n\pr{x}\leq f\pr{x}$ for all $n\in\mathbb{N}$ and $x\in X$. Let $n\in\mathbb{N}$ and $x\in X$. If $f\pr{x}\geq n$, then $\varphi_n\pr{x}=n\leq f\pr{x}$. If $f\pr{x}<n$, there exists $k\in\set{0,1,\dots,n\cdot 2^n}$ such that $\frac{k-1}{2^n}\leq f\pr{x}<\frac{k}{2^n}$. Then $\varphi_n\pr{x}=\frac{k-1}{2^n}\leq f\pr{x}$. Therefore, $\varphi_n\pr{x}\leq f\pr{x}$ for all $n\in\mathbb{N}$ and $x\in X$.

    Now, we are going to show that $\varphi_n\pr{x}\leq \varphi_{n+1}\pr{x}$ for all $n\in\mathbb{N}$ and $x\in X$. Let $n\in\mathbb{N}$ and $x\in X$. If $f\pr{x}\geq n+1$, then $\varphi_n\pr{x}=n<n+1=\varphi_{n+1}\pr{x}$. If $n\leq f\pr{x}<n+1$, there exists $k\in\set{0,1,\dots,\pr{n+1}\cdot 2^{n+1}}$ such that $\frac{k-1}{2^{n+1}}\leq f\pr{x}<\frac{k}{2^{n+1}}$. Then $\varphi_n\pr{x}=n<\frac{k}{2^{n+1}}$ or, equivalently, $2^{n+1}\cdot n<k$. Thus $2^{n+1}\cdot n\leq k-1$. Consequently, $\varphi_n\pr{x}=n\leq\frac{k-1}{2^{n+1}}=\varphi_{n+1}\pr{x}$. If $f\pr{x}<n$, there exists $k\in\set{0,1,\dots,n\cdot 2^n}$ and $l\in\set{0,1,\dots, \pr{n+1}\cdot 2^{n+1}}$ such that $\varphi_n\pr{x}=\frac{k-1}{2^n}\leq f\pr{x}<\frac{l}{2^{n+1}}$. Then $\frac{k-1}{2^n}<\frac{l}{2^{n+1}}$ or, equivalently, $2\cdot\pr{k-1}<l$. Thus $2\cdot\pr{k-1}\leq l-1$. Consequently, $\varphi_n\pr{x}=\frac{k-1}{2^n}\leq\dfrac{l-1}{2^{n+1}}=\varphi_{n+1}\pr{x}$. Therefore, $\varphi_n\pr{x}\leq \varphi_{n+1}\pr{x}$ for all $n\in\mathbb{N}$ and $x\in X$.

    Since $\varphi_n\pr{x}\leq\varphi_{n+1}\pr{x}\leq f\pr{x}$ for all $n\in\mathbb{N}$ and $x\in X$, there exist $\lim_{n\to\infty}{\varphi_n\pr{x}}\in\overline{\mathbb{R}}$ and $\lim_{n\to\infty}{\varphi_n\pr{x}}\leq f\pr{x}$. Let's prove that $\lim_{n\to\infty}{\varphi_n\pr{x}}=f\pr{x}$. Let $x\in X$. If $f\pr{x}=+\infty$, then $\varphi_n\pr{x}=n$ for all $n\in\mathbb{N}$ and we get $\lim_{n\to\infty}{\varphi_n\pr{x}}=+\infty=f\pr{x}$. If $f\pr{x}<+\infty$, there exists $n_0=n_0\pr{x}\in\mathbb{N}$ such that $f\pr{x}<n_0$. Then, for all $n\in\mathbb{N}$ with $n\geq n_0$, there exists $k=k\pr{n}\in\set{1,2,\dots,n\cdot 2^n}$ such that $\varphi_n\pr{x}=\frac{k-1}{2^n}\leq f\pr{x}<\frac{k}{2^n}$. Then:

    \begin{equation}
      \bars{\varphi_n\pr{x}-f\pr{x}}<\frac{1}{2^n},
    \end{equation}

    for all $n\in\mathbb{N}$ with $n\geq n_0$. Consequently, $\lim_{n\to\infty}{\varphi_n\pr{x}}=f\pr{x}$. Therefore, $\lim_{n\to\infty}{\varphi_n\pr{x}}=f\pr{x}$ for all $x\in X$.

    If $f$ is bounded, we can take $n_0\in\mathbb{N}$ such that $n_0>\sup{f}$. Then, for all $n\in\mathbb{N}$ with $n\geq n_0$ and $x\in X$, we have:

    \begin{equation}
      \bars{\varphi_n\pr{x}-f\pr{x}}<\frac{1}{2^n}.
    \end{equation}

    In this case, $n_0$ does not depend on the election of $x\in X$. Then, for all $n\geq n_0$, we have:

    \begin{equation}
      \sup_{X}{\bars{\varphi_n-f}}\leq\frac{1}{2^n}.
    \end{equation}

    Therefore, $\lim_{n\to\infty}{\sup_{X}{\bars{\varphi_n-f}}}=0$ or, in other words, the convergence $\lim_{n\to\infty}{\varphi_n}=f$ is uniform.
  \end{proof}
\end{theorem}

\begin{corollary}
  Let $\pr{X,\Sigma}$ be a measurable space, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a $\Sigma$-measurable function. Then there exists a sequence of simple $\Sigma$-measurable functions $\set{\varphi_n}_{n\in\mathbb{N}}\subseteq\mathbb{R}^X$ such that $\lim_{n\to\infty}{\varphi_n\pr{x}}=f\pr{x}$ for all $x\in X$. Furthermore, if $f$ is bounded, then we can construct the sequence of simple $\Sigma$-measurable functions $\set{\varphi_n}_{n\in\mathbb{N}}\subseteq\mathbb{R}^X$ such that the convergence $\lim_{n\to\infty}{\varphi_n}=f$ is uniform.
  \begin{proof}[\rm\textbf{Proof}]
    \rm Let $\set{\varphi_{1,n}}_{n\in\mathbb{N}},\set{\varphi_{2,n}}_{n\in\mathbb{N}}\subseteq\mathbb{R}^X$ be two sequences of simple $\Sigma$-measurable functions such that $\lim_{n\to\infty}{\varphi_{1,n}\pr{x}}=f^+\pr{x}$ and $\lim_{n\to\infty}{\varphi_{2,n}\pr{x}}=f^-\pr{x}$ for all $x\in X$. For each $n\in\mathbb{N}$, let $\varphi_n:X\rightarrow\mathbb{R}$ be defined as $\varphi_n:=\varphi_{1,n}-\varphi_{2,n}$. Then $\set{\varphi_n}_{n\in\mathbb{N}}\subseteq\mathbb{R}^X$ is a sequence of simple $\Sigma$-measurable functions such that $\lim_{n\to\infty}{\varphi_n\pr{x}}=\lim_{n\to\infty}{\pr{\varphi_{1,n}\pr{x}-\varphi_{2,n}\pr{x}}}=f^+\pr{x}-f^-\pr{x}=f\pr{x}$ for all $x\in X$.

    If $f$ is bounded, then $f^+$ and $f^-$ are bounded as well. Thus we can suppose that the convergences $\lim_{n\to\infty}{\varphi_{1,n}}=f^+$ and $\lim_{n\to\infty}{\varphi_{2,n}}=f^-$ are uniform. Thus:

    \begin{equation}
      \sup_X{\bars{\varphi_n-f}}\leq\sup_X\pr{\bars{\varphi_{1,n}-f^+}+\bars{\varphi_{2,n}-f^-}}\leq\sup_X\bars{\varphi_{1,n}-f^+}+\sup_X\bars{\varphi_{2,n}-f^-}.
    \end{equation}
    
    Therefore, $\lim_{n\to\infty}{\sup_{X}{\bars{\varphi_n-f}}}=0$ or, in other words, the convergence $\lim_{n\to\infty}{\varphi_n}=f$ is uniform.
  \end{proof}
\end{corollary}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $n,m\in\mathbb{N}$, $A_1,\dots,A_n,B_1,\dots,B_m\in\Sigma$ and $a_1,\dots,a_n,b_1,\dots,b_m\in\mathbb{R}$ such that $\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}=\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}}$. Then $\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k}}=\sum_{l=1}^{m}{b_l\cdot\mu\pr{B_l}}$.

  \begin{proof}[\rm\textbf{Proof}]
    Given $\varepsilon:=\pr{\delta_1,\dots,\delta_n,\delta'_1,\dots,\delta'_m}\in\set{0,1}^{n+m}$, we define:

    \begin{equation}
      C_{\varepsilon}:=\pr{\bigcap_{k=1}^{n}{E_k}}\cap\pr{\bigcap_{l=1}^{m}{F_l}},
    \end{equation}

    where:

    \begin{equation}
      E_k:=\begin{cases}
        A_k^c & \text{if $\delta_k=0$}\\
        A_k & \text{if $\delta_k=1$}
      \end{cases},\hspace{0.5cm}F_l:=\begin{cases}
        B_l^c & \text{if $\delta_l'=0$}\\
        B_l & \text{if $\delta_l'=1$}
      \end{cases}
    \end{equation}

    for each $k\in\set{1,\dots,n}$ and $l\in\set{1,\dots,m}$. It is clear that $\set{C_{\varepsilon}\middle|\varepsilon\in\set{0,1}^{n+m}}\subseteq\Sigma$ is a collection of pairwise disjoint $\Sigma$-measurable sets such that $X=\bigcup_{\varepsilon\in\set{0,1}^{n+m}}{C_{\varepsilon}}$. Furthermore, if $\varepsilon\in\set{0,1}^{n+m}$, then $C_{\varepsilon}\subseteq A_k$ or $C_{\varepsilon}\subseteq A_k^c$ for all $k\in\set{1,\dots,n}$, and, $C_{\varepsilon}\subseteq B_l$ or $C_{\varepsilon}\subseteq B_l^c$ for all $l\in\set{1,\dots,m}$. Thus, for each $\varepsilon\in\set{0,1}^{n+m}$, there exists $\alpha_\varepsilon,\beta_\varepsilon\in\mathbb{R}$ such that:

    \begin{equation}
      \begin{split}
        \alpha_\varepsilon & =\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}\pr{x}}=\sum_{\substack{k=1\\C_{\varepsilon}\subseteq A_k}}^{n}{a_k},\\
        \beta_\varepsilon & =\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}\pr{x}}=\sum_{\substack{l=1\\C_{\varepsilon}\subseteq B_l}}^{m}{b_l},
      \end{split}
    \end{equation}

    for all $x\in C_{\varepsilon}$. Since $\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}=\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}}$, then $\alpha_\varepsilon=\beta_\varepsilon$ for all $\varepsilon\in\set{0,1}^{n+m}$. Thus:

      \allowdisplaybreaks % chktex 1
      \begin{align}
        \sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k}} & =\sum_{k=1}^{n}{\sum_{\varepsilon\in\set{0,1}^{n+m}}{a_k\cdot\mu\pr{A_k\cap C_{\varepsilon}}}}=\sum_{k=1}^{n}{\sum_{\substack{\varepsilon\in\set{0,1}^{n+m}\\C_\varepsilon\subseteq A_k}}{a_k\cdot\mu\pr{C_{\varepsilon}}}}\notag\\
        & =\sum_{\varepsilon\in\set{0,1}^{n+m}}{\alpha_\varepsilon}=\sum_{\varepsilon\in\set{0,1}^{n+m}}{\beta_\varepsilon}=\sum_{l=1}^{m}{\sum_{\substack{\varepsilon\in\set{0,1}^{n+m}\\C_\varepsilon\subseteq B_l}}{b_l\cdot\mu\pr{C_{\varepsilon}}}}\\
        &=\sum_{l=1}^{m}{\sum_{\varepsilon\in\set{0,1}^{n+m}}{b_l\cdot\mu\pr{B_l\cap C_{\varepsilon}}}}=\sum_{l=1}^{m}{b_l\cdot\mu\pr{B_l}}.\notag
      \end{align}
  \end{proof}
\end{proposition}

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\mathbb{R}$ and $\varphi:X\rightarrow Y'$ a simple non-negative $\Sigma$-measurable function. We define the\textit{ (Lebesgue) integral of $\varphi$ over $X$ with respect to $\mu$} as follows:

  \begin{equation}
    \int_{X}{\varphi\,d\mu}:=\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k}}\in\br{0,+\infty},
  \end{equation}

  where $n\in\mathbb{N}$, $a_1,\dots,a_n\in\bpr{0,+\infty}$ and $A_1,\dots,A_n\in\Sigma$ are such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$. In this context, we set $0\cdot\pr{+\infty}=\pr{+\infty}\cdot 0=0$.
\end{definition}

\begin{remark}
  By the previous proposition, if $\pr{X,\Sigma,\mu}$ is a measure space, $Y'\subseteq\mathbb{R}$ and $\varphi:X\rightarrow Y'$ is a simple non-negative $\Sigma$-measurable function, the integral of $\varphi$ over $X$ with respect to $\mu$ is well-defined.
\end{remark}

\begin{remark}
  Given $\pr{X,\Sigma,\mu}$ a measure space, $A\in\Sigma$, $Y'\subseteq\mathbb{R}$ and $\varphi:X\rightarrow Y'$ a simple non-negative $\Sigma$-measurable function. We define the\textit{ (Lebesgue) integral of $\varphi$ over $A$ with respect to $\mu$} as follows:

  \begin{equation}
    \int_{A}{\varphi\,d\mu}:=\int_{A}{\rest{\varphi}{A}\,d\rest{\mu}{\Sigma_A}}.
  \end{equation}

  If $a_1,\dots,a_n\in\bpr{0,+\infty}$ and $A_1,\dots,A_n\in\Sigma$ with $n\in\mathbb{N}$ are such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$, then:

  \begin{equation}
    \begin{split}
      \int_{A}{\varphi\,d\mu} & =\int_{A}{\rest{\varphi}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{\pr{\sum_{k=1}^{n}{a_k\cdot\rest{\chi_{A_k}}{A}}}\,d\rest{\mu}{\Sigma_A}}\\
      & =\int_{A}{\pr{\sum_{k=1}^{n}{a_k\cdot\rest{\chi_{A_k\cap A}}{A}}}\,d\rest{\mu}{\Sigma_A}}=\sum_{k=1}^{n}{a_k\cdot\rest{\mu}{\Sigma_A}\pr{A_k\cap A}}\\
      & =\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k\cap A}}=\int_{X}{\pr{\sum_{k=1}^{n}{a_k\cdot\chi_{A_k\cap A}}}\,d\mu}=\int_{X}{\pr{\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}\cdot\chi_A}}\,d\mu}\\
      & =\int_{X}{\pr{\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}\cdot\chi_A}\,d\mu}=\int_{X}{\varphi\cdot\chi_A\,d\mu}.
    \end{split}
  \end{equation}

  Therefore:

  \begin{equation}
    \int_{A}{\varphi\,d\mu}=\int_{X}{\varphi\cdot\chi_A\,d\mu}=\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k\cap A}}.
  \end{equation}
\end{remark}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$, $\alpha,\beta\geq 0$ and, $\varphi$ and $\psi$ two simple non-negative $\Sigma$-measurable functions. The following properties hold:

  \begin{enumerate}
      \item $\int_{X}{\pr{\alpha\cdot\varphi+\beta\cdot\psi}\,d\mu}=\alpha\cdot\int_{X}{\varphi\,d\mu}+\beta\cdot\int_{X}{\psi\,d\mu}$, where, in this context, $0\cdot\pr{+\infty}=\pr{+\infty}\cdot 0=0$.
      \item If $\varphi\pr{x}\leq\psi\pr{x}$ for all $x\in X$, then $\int_{X}{\varphi\,d\mu}\leq\int_{X}{\psi\,d\mu}$.
      \item If $\mu\pr{A}=0$ or $\varphi\pr{x}=0$ for all $x\in A$, then $\int_{A}{\varphi\,d\mu}=0$.
      \item The map $\nu_{\varphi}:E\in\Sigma\mapsto\int_{E}{\varphi\,d\mu}\in\br{0,+\infty}$ is a measure on $\pr{X,\Sigma}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $a_1,\dots,a_n,b_1,\dots,b_m\in\mathbb{R}$ and $A_1,\dots,A_n,B_1,\dots,B_m\in\Sigma$ with $n,m\in\mathbb{N}$ such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$ and $\psi=\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}}$. Then:
      
      \begin{equation}
        \begin{split}
          \int_{X}{\pr{\alpha\cdot\varphi+\beta\cdot\psi}\,d\mu} & =\int_{X}{\pr{\alpha\cdot\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}+\beta\cdot\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}}}\,d\mu}\\
          & =\int_{X}{\pr{\sum_{k=1}^{n}{\pr{\alpha\cdot a_k}\cdot\chi_{A_k}}+\sum_{l=1}^{m}{\pr{\beta\cdot b_l}\cdot\chi_{B_l}}}\,d\mu}\\
          & =\sum_{k=1}^{n}{\pr{\alpha\cdot a_k}\cdot\mu\pr{A_k}}+\sum_{l=1}^{m}{\pr{\beta\cdot b_l}\cdot\mu\pr{B_l}}\\
          & =\alpha\cdot\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k}}+\beta\cdot\sum_{l=1}^{m}{b_l\cdot\mu\pr{B_l}}\\
          & =\alpha\cdot\int_{X}{\varphi\,d\mu}+\beta\cdot\int_{X}{\psi\,d\mu}.
        \end{split}
      \end{equation}
      \item Let $a_1,\dots,a_n,b_1,\dots,b_m\in\mathbb{R}$ and $A_1,\dots,A_n,B_1,\dots,B_m\in\Sigma$ with $n,m\in\mathbb{N}$ such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$ and $\psi=\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}}$. We can suppose that $A_1,\dots,A_n$ are pairwise disjoint and $B_1,\dots, B_m$ are pairwise disjoint as well. For each $k\in\set{1,\dots,n}$ and $l\in\set{1,\dots,m}$, let $C_{k,l}:=A_k\cap B_l$. Then:
      
      \begin{equation}
        \begin{split}
          \varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}=\sum_{k=1}^{n}{a_k\cdot\pr{\sum_{l=1}^{m}{\chi_{C_{k,l}}}}}=\sum_{k=1}^{n}{\sum_{l=1}^{m}{a_k\cdot\chi_{C_{k,l}}}},\\
          \psi=\sum_{l=1}^{m}{b_l\cdot\chi_{B_l}}=\sum_{l=1}^{m}{b_l\cdot\pr{\sum_{k=1}^{n}{\chi_{C_{k,l}}}}}=\sum_{k=1}^{n}{\sum_{l=1}^{m}{b_l\cdot\chi_{C_{k,l}}}}.
        \end{split}
      \end{equation}

      Given $k\in\set{1,\dots,n}$ and $l\in\set{1,\dots,m}$, if $x\in C_{k,l}$, then $\varphi\pr{x}=a_k\leq b_l=\psi\pr{x}$. Then:

      \begin{equation}
        \int_{X}{\varphi\,d\mu}=\sum_{k=1}^{n}{\sum_{l=1}^{m}{a_k\cdot\mu\pr{C_{k,l}}}}\leq\sum_{k=1}^{n}{\sum_{l=1}^{m}{b_l\cdot\mu\pr{C_{k,l}}}}=\int_{X}{\psi\,d\mu}.
      \end{equation}
      \item Let $a_1,\dots,a_n\in\mathbb{R}$ and $A_1,\dots,A_n\in\Sigma$ with $n\in\mathbb{N}$ such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$. Then:
      
      \begin{equation}
        \int_{A}{\varphi\,d\mu}=\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k\cap A}}.
      \end{equation}

      If $\mu\pr{A}=0$, then $\mu\pr{A_k\cap A}=0$ for all $k\in\set{1,\dots,n}$ by monotonicity and:
      
      \begin{equation}
        \int_{A}{\varphi\,d\mu}=\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k\cap A}}=0.
      \end{equation}

      If $\varphi\pr{x}=0$ for all $x\in A$, then:
      
      \begin{equation}
        \int_{A}{\varphi\,d\mu}=\int_{A}{\rest{\varphi}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{0\,d\rest{\mu}{\Sigma_A}}=0\cdot\rest{\mu}{\Sigma_A}\pr{A}=0.
      \end{equation}
      
      \item Let $a_1,\dots,a_n\in\mathbb{R}$ and $A_1,\dots,A_n\in\Sigma$ with $n\in\mathbb{N}$ such that $\varphi=\sum_{k=1}^{n}{a_k\cdot\chi_{A_k}}$. Firstly, we see that $\nu_{\varphi}$ is well defined, because $0\leq\varphi\pr{x}$ for all $x\in X$ and, consequently, $0=\int_{X}{0\,d\mu}\leq\int_{X}{\varphi\,d\mu}$, or, in other words, $\int_{X}{\varphi\,d\mu}\in\br{0,+\infty}$. Since $\mu\pr{\varnothing}=0$, by the previous property, we have that $\nu_{\varphi}\pr{\varnothing}=0$. Let $\set{A_m}_{m\in\mathbb{N}}\subseteq\Sigma$ be a countable collection of pairwise disjoint $\Sigma$-measurable sets. Then:
      

      \begin{equation}
        \begin{split}
          \nu_\varphi\pr{\bigcup_{m=1}^{\infty}{A_m}}& =\int_{\bigcup_{m=1}^{\infty}{A_m}}{\varphi\,d\mu} =\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k\cap\pr{\bigcup_{m=1}^{\infty}{A_m}}}}\\
          & =\sum_{k=1}^{n}{a_k\cdot\mu\pr{\bigcup_{m=1}^{\infty}{\pr{A_k\cap A_m}}}}=\sum_{k=1}^{n}{a_k\cdot\pr{\sum_{m=1}^{\infty}{\mu\pr{A_k\cap A_m}}}}\\
          & =\sum_{k=1}^{n}{\sum_{m=1}^{\infty}{a_k\cdot\mu\pr{A_k\cap A_m}}}=\sum_{m=1}^{\infty}{\sum_{k=1}^{n}{a_k\cdot\mu\pr{A_k\cap A_m}}}\\
          & =\sum_{m=1}^{\infty}{\int_{A_m}{\varphi\,d\mu}}=\sum_{m=1}^{\infty}{\nu_\varphi\pr{A_m}}.
        \end{split}
      \end{equation}

      Therefore, $\nu_\varphi$ is a measure on $\pr{X,\Sigma}$.
    \end{enumerate}
  \end{proof}
\end{proposition}

We will denote by $\mathcal{S}\pr{X,\Sigma}$ the \textit{set of all simple $\Sigma$-measurable functions in $\mathbb{R}^X$}.

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\overline{\mathbb{R}}$, and $f:X\rightarrow Y'$ a non-negative $\Sigma$-measurable function. We define the \textit{(Lebesgue) integral of $f$ over $X$ with respect to $\mu$} as follows:

  \begin{equation}
    \int_{X}{f\,d\mu}:=\sup\set{\int_{X}{\varphi\,d\mu}\middle|\varphi\in\mathcal{S}\pr{X,\Sigma},0\leq\varphi\pr{x}\leq f\pr{x},\forall x\in X}\in\br{0,+\infty}.
  \end{equation}
\end{definition}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space. The definition of the Lebesgue integral for non-negative $\Sigma$-measurable functions is consistent with the definition that we gave previously for simple non-negative $\Sigma$-measurable functions.
\end{remark}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a non-negative $\Sigma$-measurable function. We define the \textit{(Lebesgue) integral of $f$ over $A$ with respect to $\mu$ as follows:}

  \begin{equation}
    \int_{A}{f\,d\mu}:=\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}.
  \end{equation}

  Let $\varphi\in\mathcal{S}\pr{A,\Sigma_A}$ such that $0\leq\varphi\pr{x}\leq\rest{f}{A}\pr{x}$ for all $x\in A$. Let $\widetilde{\varphi}:X\rightarrow\mathbb{R}$ be defined as $\widetilde{\varphi}\pr{x}=\varphi\pr{x}$ for all $x\in A$ and $\widetilde{\varphi}\pr{x}=0$ if $x\notin A$. Then $0\leq\widetilde{\varphi}\pr{x}\leq f\pr{x}\cdot\chi_A\pr{x}$ for all $x\in X$ and:

  \begin{equation}
    \begin{split}
      \int_{A}{\varphi\,d\left.\mu\right|_{\Sigma_A}} & =\int_{A}{\rest{\widetilde{\varphi}}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{\widetilde{\varphi}\,d\mu}=\int_{A}{\widetilde{\varphi}\,d\mu}+\int_{A^c}{0\,d\mu}=\int_{A}{\widetilde{\varphi}\,d\mu}+\int_{A^c}{\widetilde{\varphi}\,d\mu}\\
      & =\int_{X}{\widetilde{\varphi}\,d\mu}\leq\int_{X}{f\cdot\chi_A\,d\mu}.
    \end{split}
  \end{equation}

  Then:

  \begin{equation}
    \int_{A}{f\,d\mu}=\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}\leq\int_{X}{f\cdot\chi_A\,d\mu}.
  \end{equation}

  Let $\varphi\in\mathcal{S}\pr{X,\Sigma}$ such that $0\leq\varphi\pr{x}\leq f\pr{x}\cdot\chi_A\pr{x}$ for all $x\in X$. Then $\widetilde{\varphi}=\rest{\varphi}{A}\in\mathcal{S}\pr{A,\Sigma_A}$ and $0\leq\widetilde{\varphi}\pr{x}\leq\rest{f}{A}\pr{x}$ for all $x\in A$. Thus:

  \begin{equation}
    \begin{split}
      \int_{X}{\varphi\,d\mu} & =\int_{A}{\varphi\,d\mu}+\int_{A^c}{\varphi\,d\mu}=\int_{A}{\varphi\,d\mu}=\int_{A}{\rest{\varphi}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{\widetilde{\varphi}\,d\rest{\mu}{\Sigma_A}}\\
      & \leq\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{f\,d\mu}.
    \end{split}
  \end{equation}

  Consequently:

  \begin{equation}
    \int_{X}{f\cdot\chi_A\,d\mu}\leq\int_{A}{f\,d\mu}.
  \end{equation}

  Therefore:

  \begin{equation}
    \int_{A}{f\,d\mu}=\int_{X}{f\cdot\chi_A\,d\mu}.
  \end{equation}
\end{remark}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A,B\in\Sigma$, $Y',Y''\subseteq\overline{\mathbb{R}}$, $\alpha\geq 0$ and, $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two non-negative $\Sigma$-measurable functions. The following properties hold:

  \begin{enumerate}
      \item If $f\pr{x}\leq g\pr{x}$ for all $x\in X$, then $\int_{X}{f\,d\mu}\leq\int_{X}{g\,d\mu}$.
      \item If $A\subseteq B$, then $\int_{A}{f\,d\mu}\leq\int_{B}{f\,d\mu}$.
      \item $\int_{X}{\pr{\alpha\cdot f}\,d\mu}=\alpha\cdot\int_{X}{f\,d\mu}$, where, in this context, $0\cdot\pr{+\infty}=\pr{+\infty}\cdot 0=0$.
      \item $\int_{X}{\pr{f+g}\,d\mu}=\int_{X}{f\,d\mu}+\int_{X}{g\,d\mu}$.
      \item If $\mu\pr{A}=0$ or $f\pr{x}=0$ for all $x\in A$, then $\int_{A}{f\,d\mu}=0$.
      \item If $\mu\pr{A}=0$, then $\int_{X}{f\,d\mu}=\int_{A^c}{f\,d\mu}$.
      \item The map $\nu_f:E\in\Sigma\mapsto\int_{E}{f\,d\mu}\in\br{0,+\infty}$ is a measure on $\pr{X,\Sigma}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $\varphi\in\mathcal{S}\pr{X,\Sigma}$ such that $0\leq\varphi\pr{x}\leq f\pr{x}$ for all $x\in X$. Since $f\pr{x}\leq g\pr{x}$ for all $x\in X$, $0\leq\varphi\pr{x}\leq g\pr{x}$ for all $x\in X$. Then:
      
      \begin{equation}
        \int_{X}{\varphi\,d\mu}\leq\int_{X}{g\,d\mu}.
      \end{equation}

      Therefore:

      \begin{equation}
        \int_{X}{f\,d\mu}\leq\int_{X}{g\,d\mu}.
      \end{equation}
      \item If $A\subseteq B$, then $f\pr{x}\cdot\chi_A\pr{x}\leq f\pr{x}\cdot\chi_B\pr{x}$ for all $x\in X$. Then:
      
      \begin{equation}
        \int_{A}{f\,d\mu}=\int_{X}{f\cdot\chi_A\,d\mu}\leq\int_{X}{f\cdot\chi_B\,d\mu}=\int_{B}{f\,d\mu}.
      \end{equation}
      \item If $\alpha=0$, then it is trivial. Suppose $\alpha>0$. Let $\varphi\in\mathcal{S}\pr{X,\Sigma}$ such that $0\leq\varphi\pr{x}\leq\alpha\cdot f\pr{x}$ for all $x\in X$. Then $\alpha^{-1}\cdot\varphi\pr{x}\in\mathcal{S}\pr{X,\Sigma}$ verifies $0\leq\alpha^{-1}\cdot\varphi\pr{x}\leq f\pr{x}$ for all $x\in X$. Thus:
      
      \begin{equation}
        \alpha^{-1}\cdot\int_{X}{\varphi\,d\mu}=\int_{X}{\pr{\alpha^{-1}\cdot\varphi}\,d\mu}\leq\int_{X}{f\,d\mu}.
      \end{equation}

      Consequently:

      \begin{equation}
        \int_{X}{\varphi\,d\mu}\leq\alpha\cdot\int_{X}{f\,d\mu}.
      \end{equation}

      Hence:

      \begin{equation}
        \begin{split}
          \int_{X}{\pr{\alpha\cdot f}\,d\mu}&\leq\alpha\cdot\int_{X}{f\,d\mu}=\alpha\cdot\int_{X}{\pr{\alpha^{-1}\cdot\alpha\cdot f}\,d\mu}\leq\alpha\cdot\pr{\alpha^{-1}\cdot\int_{X}{\pr{\alpha\cdot f}\,d\mu}}\\
          & =\int_{X}{\pr{\alpha\cdot f}\,d\mu}.
        \end{split}
      \end{equation}

      Therefore:

      \begin{equation}
        \int_{X}{\pr{\alpha\cdot f}\,d\mu}=\alpha\cdot\int_{X}{f\,d\mu}.
      \end{equation}

      \item Let $\set{\varphi_n}_{n\in\mathbb{N}},\set{\psi_n}_{n\in\mathbb{N}}\subseteq\mathcal{S}\pr{X,\Sigma}$ be two sequences of simple non-negative $\Sigma$-measurable functions such that $0\leq\varphi_n\pr{x}\leq\varphi_{n+1}\pr{x}\leq f\pr{x}$ and $0\leq\psi_n\pr{x}\leq\psi_{n+1}\pr{x}\leq g\pr{x}$ for all $n\in\mathbb{N}$, and, $\lim_{n\to\infty}{\varphi_n\pr{x}}=f\pr{x}$ and $\lim_{n\to\infty}{\psi_n\pr{x}}=g\pr{x}$ for all $x\in X$. Then $\set{\varphi_n+\psi_n}_{n\in\mathbb{N}}\subseteq\mathcal{S}\pr{X,\Sigma}$ is a sequence of simple non-negative $\Sigma$-measurable functions such that $0\leq\varphi_n\pr{x}+\psi_n\pr{x}\leq f\pr{x}+g\pr{x}$ for all $n\in\mathbb{N}$ and $\lim_{n\to\infty}{\pr{\varphi_n\pr{x}+\psi_n\pr{x}}}=f\pr{x}+g\pr{x}$ for all $x\in X$. By the \hyperref[Monotone Convergence Theorem]{Monotone Convergence Theorem}, we have:
      
      \begin{equation}
        \begin{split}
          \int_{X}{\pr{f+g}\,d\mu} & =\lim_{n\to\infty}{\int_{X}{\pr{\varphi_n+\psi_n}\,d\mu}}=\lim_{n\to\infty}{\pr{\int_{X}{\varphi_n\,d\mu}+\int_{X}{\psi_n\,d\mu}}}\\
          & =\lim_{n\to\infty}{\int_{X}{\varphi_n\,d\mu}}+\lim_{n\to\infty}{\int_{X}{\psi_n\,d\mu}}=\int_{X}{f\,d\mu}+\int_{X}{g\,d\mu}.
        \end{split}
      \end{equation}

      \item If $\mu\pr{A}=0$, then $\int_{A}{\varphi\,d\mu}=0$ for all $\varphi\in\mathcal{S}\pr{X,\Sigma}$. Thus:
      
      \begin{equation}
        \int_{A}{f\,d\mu}=\int_{X}{f\cdot\chi_A\,d\mu}=0.
      \end{equation}

      Suppose $f\pr{x}=0$ for all $x\in A$. Then $\varphi\pr{x}=0$ for all $\varphi\in\mathcal{S}\pr{A,\Sigma_A}$ with $0\leq\varphi\pr{x}\leq\rest{f}{A}\pr{x}$ for all $x\in A$. Thus:
      
      \begin{equation}
        \int_{A}{f\,d\mu}=\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}=0.
      \end{equation}
      \item Since $f=f\cdot\chi_A+f\cdot\chi_{A^c}$, we have:
      
      \begin{equation}
        \begin{split}
          \int_{X}{f\,d\mu} & =\int_{X}{f\cdot\chi_A\,d\mu}+\int_{X}{f\cdot\chi_{A^c}\,d\mu}=\int_{A}{f\,d\mu}+\int_{A^c}{f\,d\mu}=0+\int_{A^c}{f\,d\mu}\\
          & =\int_{A^c}{f\,d\mu}.
        \end{split}
      \end{equation}

      \item Firstly, we see that $\nu_f$ is well defined, because $0\leq f\pr{x}$ for all $x\in X$ and, consequently, $0=\int_{X}{0\,d\mu}\leq\int_{X}{f\,d\mu}$, or, in other words, $\int_{X}{f\,d\mu}\in\br{0,+\infty}$. Since $\mu\pr{\varnothing}=0$, by the previous property, we have that $\nu_f\pr{\varnothing}=0$. Let $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ be a countable collection of pairwise disjoint $\Sigma$-measurable sets. By \hyperref[Beppo Levi's Theorem]{Beppo Levi's Theorem}, we have that:
      
      \begin{equation}
        \begin{split}
          \nu_f\pr{\bigcup_{n=1}^{\infty}{A_n}} &=\int_{\bigcup_{n=1}^{\infty}{A_n}}{f\,d\mu}=\int_{X}{\pr{f\cdot\chi_{\bigcup_{n=1}^{\infty}{A_n}}}\,d\mu}=\int_{X}{\pr{\sum_{n=1}^{\infty}{\pr{f\cdot\chi_{A_n}}}}\,d\mu}\\
          & =\sum_{n=1}^{\infty}{\int_{X}{\pr{f\cdot\chi_{A_n}}\,d\mu}}=\sum_{n=1}^{\infty}{\int_{A_n}{f\,d\mu}}=\sum_{n=1}^{\infty}{\nu_f\pr{A_n}}.
        \end{split}
      \end{equation}

      Therefore, $\nu_f$ is a measure on $\pr{X,\Sigma}$.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$ and $P$ a predicate on $X$. We say that \textit{$P$ holds $\mu$-almost everywhere in $A$} if there exists a $\Sigma$-measurable set $N\in\Sigma$ such that the set $\set{x\middle|x\in A,P\pr{x}\text{ does not hold}}\subseteq N$ and $\mu\pr{N}=0$.
\end{definition}

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\overline{\mathbb{R}}$ and $f,g:X\rightarrow Y'$ two functions. We say that $f$ and $g$ are $\mu$-equivalent if $f=g$ $\mu$-almost everywhere on $X$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$, $Y'\subseteq\overline{\mathbb{R}}$ and $f,g:X\rightarrow Y'$ two functions. The following properties hold:

  \begin{enumerate}
    \item If $\mu$ is complete, $f$ is $\Sigma$-measurable and $f=g$ $\mu$-almost everywhere in $X$, then $g$ is $\Sigma$-measurable.
    \item If $\mu$ is complete and $\set{f_n}_{n\in\mathbb{N}}\subseteq\overline{\mathbb{R}}^X$ is a sequence of $\Sigma$-measurable functions such that, for $\mu$-almost every $x\in X$, there exists $\lim_{n\to\infty}{f_n\pr{x}}=f\pr{x}$, then $f$ is $\Sigma$-measurable.
    \item If $f$ and $g$ are non-negative $\Sigma$-measurable functions such that $f=g$ $\mu$-almost everywhere in $X$, then $\int_{X}{f\,d\mu}=\int_{X}{g\,d\mu}$.
    \item Let $\mathcal{T}_1\subseteq\Sigma$ a topology on $X$, $\pr{Y,\mathcal{T}_2}$ a topological space and $\Sigma':=\mathscr{B}\pr{Y,\mathcal{T}_2}$. If $\mu$ is complete and $h:X\rightarrow Y$ is $\pr{\mathcal{T}_1,\mathcal{T}_2}$-continuous for $\mu$-almost $x\in X$, then $h$ is $\pr{\Sigma,\Sigma'}$-measurable. 
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $N\in\Sigma$ such that $\set{x\middle|x\in X,f\pr{x}\neq g\pr{x}}\subseteq N$ and $\mu\pr{N}=0$. Then $N^c\in\Sigma$ and $f\pr{x}=g\pr{x}$ for all $x\in N^c$. Let $a\in\mathbb{R}$. Then:
      
      \begin{equation}
        \begin{split}
          \set{g<a} & =\set{g<a}\cap\pr{N\cup N^c}=\pr{\set{g<a}\cap N}\cup\pr{\set{g<a}\cap N^c}\\
          & =\pr{\set{g<a}\cap N}\cup\pr{\set{f<a}\cap N^c}.
        \end{split}
      \end{equation}

      Since $\mu$ is a complete measure and $\set{g<a}\cap N\subseteq N$, $\set{g<a}\cap N\in\Sigma$. Furthermore, since $f$ is a $\Sigma$-measurable function, $\set{f<a}\in\Sigma$ and, consequently, $\set{f<a}\cap N^c\in\Sigma$. Thus, $\set{g<a}\in\Sigma$. Therefore, $g$ is $\Sigma$-measurable.

      \item Let $g=\liminf_{n\to\infty}{f_n}$. We know that $g$ is $\Sigma$-measurable and $f=g$ $\mu$-almost everywhere in $X$. Then $f$ is $\Sigma$-measurable.
      \item Let $N:=\set{f\neq g}\in\Sigma$. Then $\mu\pr{N}=0$, $f\pr{x}=g\pr{x}$ for all $x\in N^c$ and:
      
      \begin{equation}
        \begin{split}
          \int_{X}{f\,d\mu} & =\int_{N^c}{f\,d\mu}=\int_{N^c}{\rest{f}{N^c}\,d\rest{\mu}{\Sigma_{N^c}}}=\int_{N^c}{\rest{g}{N^c}\,d\rest{\mu}{\Sigma_{N^c}}}=\int_{N^c}{g\,d\mu}\\
          & =\int_{X}{g\,d\mu}.
        \end{split}
      \end{equation}
      \item Let $N\in\Sigma$ such that $\set{x\middle|x\in X,h\text{ is not $\pr{\mathcal{T}_1,\mathcal{T}_2}$-continuous at $x$}}\subseteq N$ and $\mu\pr{N}=0$, and $G\in\mathcal{T}_2$. Then:
      
      \begin{equation}
        \begin{split}
          h^{-1}\pr{G} & =h^{-1}\pr{G}\cap\pr{N\cup N^c}=\pr{h^{-1}\pr{G}\cap N}\cup\pr{h^{-1}\pr{G}\cap N^c}\\
          & =\pr{h^{-1}\pr{G}\cap N}\cup\pr{\rest{h}{N^c}}^{-1}\pr{G}.
        \end{split}
      \end{equation}
      
      Since $\mu$ is a complete measure and $h^{-1}\pr{G}\cap N\subseteq N$, $h^{-1}\pr{G}\cap N\in\Sigma$. Furthermore, since $\rest{h}{N^c}$ is $\pr{\pr{\mathcal{T}_1}_{N^c},\mathcal{T}_2}$-continuous and $\pr{\mathcal{T}_1}_{N^c}\subseteq\Sigma_{N^c}\subseteq\Sigma$, we have that $\pr{\rest{h}{N^c}}^{-1}\pr{G}\in\Sigma$. Consequently, $h^{-1}\pr{G}\in\Sigma$. Therefore, $h$ is $\pr{\Sigma,\Sigma'}$-measurable.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a non-negative $\Sigma$-measurable function. The following properties hold:

  \begin{enumerate}
    \item Chebyshev's inequality: If $a>0$, then $\mu\pr{\set{f\geq a}}\leq\frac{1}{a}\cdot\int_{X}{f\,d\mu}$.\label{Chebyshev's inequality}
    \item If $\int_{X}{f\,d\mu}=0$, then $f\pr{x}=0$ $\mu$-almost everywhere in $X$.
    \item If $\int_{X}{f\,d\mu}<+\infty$, then $f<+\infty$ $\mu$-almost everywhere in $X$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Since $a\cdot\chi_{\set{f\geq a}}\pr{x}\leq f\pr{x}$ for all $x\in X$, we have:
      
      \begin{equation}
        a\cdot\mu\pr{\set{f\geq a}}=a\cdot\int_{X}{\chi_{\set{f\geq a}}\,d\mu}=\int_{X}{a\cdot\chi_{\set{f\geq a}}\,d\mu}\leq\int_{X}{f\,d\mu}.
      \end{equation}

      Therefore:

      \begin{equation}
        \mu\pr{\set{f\geq a}}\leq\frac{1}{a}\cdot\int_{X}{f\,d\mu}.
      \end{equation}
      \item Let $N:=\set{f>0}=\bigcup_{n=1}^{\infty}{\set{f\geq\frac{1}{n}}}\in\Sigma$. Given $n\in\mathbb{N}$, let $N_n:=\set{f\geq\frac{1}{n}}$. By contradiction, suppose that $\mu\pr{N}>0$. Necessarily, there exists $n_0\in\mathbb{N}$ such that $\mu\pr{N_{n_0}}>0$. By \hyperref[Chebyshev's inequality]{Chebyshev's inequality}, we have:
      
      \begin{equation}
        0<\mu\pr{N_{n_0}}\leq n_0\cdot\int_{X}{f\,d\mu}=0.
      \end{equation}

      We have reached a contradiction. Therefore, $\mu\pr{N}=0$, or, equivalently, $f=0$ $\mu$-almost everywhere in $X$.
      \item Let $N:=\set{f=+\infty}=\bigcap_{n=1}^{\infty}{\set{f>n}}\in\Sigma$. By Chebyshev's inequality, we have:
      
      \begin{equation}
        \mu\pr{\set{f>n}}\leq\frac{1}{n}\cdot\int_{X}{f\,d\mu}.
      \end{equation}

      Since $\int_{X}{f\,d\mu}<+\infty$, we have that $\mu\pr{\set{f>n}}<+\infty$ for all $n\in\mathbb{N}$, $\lim_{n\to\infty}{\frac{1}{n}\cdot\int_{X}{f\,d\mu}}=0$ and, consequently, $\mu\pr{N}=\lim_{n\to\infty}{\mu\pr{\set{f>n}}}=0$. Therefore, $f<+\infty$ $\mu$-almost everywhere in $X$.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a $\Sigma$-measurable function. If $\int_{X}{f^+\,d\mu}$ and $\int_{X}{f^-\,d\mu}$ are not both infinite, we define the \textit{(Lebesgue) integral of $f$ over $X$ with respect to $\mu$} as follows:

  \begin{equation}
    \int_{X}{f\,d\mu}:=\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}.
  \end{equation}

  We say that $f$ is \textit{(Lebesgue) integrable (with respect to $\pr{X,\Sigma,\mu}$)} if $\int_{X}{f^+\,d\mu},\int_{X}{f^-\,d\mu}<+\infty$. 
  
  We define the following sets:

  \begin{equation}
    \begin{split}
      \Leb{X}{\Sigma}{\mu} & :=\set{f\middle|Y'\subseteq\overline{\mathbb{R}},f\in\pr{Y'}^{X}\text{ Lebesgue integrable with respect to $\pr{X,\Sigma,\mu}$}},\\
      \LebR{X}{\Sigma}{\mu} & :=\set{f\middle|Y'\subseteq\mathbb{R},f\in\pr{Y'}^{X}\text{ Lebesgue integrable with respect to $\pr{X,\Sigma,\mu}$}}.
    \end{split}
  \end{equation}
\end{definition}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space. The definition of the Lebesgue integral for $\Sigma$-measurable functions is consistent with the definition that we gave previously for non-negative $\Sigma$-measurable functions.
\end{remark}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$ and $f\in\Leb{X}{\Sigma}{\mu}$. By the next proposition, we know that $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$. We define the \textit{(Lebesgue) integral of $f$ over $A$ with respect to $\mu$ as follows:}

  \begin{equation}
    \int_{A}{f\,d\mu}:=\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}.
  \end{equation}

  Then:

  \begin{equation}
    \begin{split}
      \int_{A}{f\,d\mu} & =\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{\pr{\rest{f}{A}}^+\,d\rest{\mu}{\Sigma_A}}-\int_{A}{\pr{\rest{f}{A}}^-\,d\rest{\mu}{\Sigma_A}}\\
      & =\int_{A}{\rest{f^+}{A}\,d\rest{\mu}{\Sigma_A}}-\int_{A}{\rest{f^-}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{f^+\,d\mu}-\int_{A}{f^-\,d\mu}\\
      & =\int_{X}{f^+\cdot\chi_A\,d\mu}-\int_{X}{f^-\cdot\chi_A\,d\mu}=\int_{X}{\pr{f\cdot\chi_A}^+\,d\mu}-\int_{X}{\pr{f\cdot\chi_A}^-\,d\mu}\\
      & =\int_{X}{f\cdot\chi_A\,d\mu}.
    \end{split}
  \end{equation}

  Therefore:

  \begin{equation}
    \int_{A}{f\,d\mu}=\int_{A}{f^+\,d\mu}-\int_{A}{f^-\,d\mu}=\int_{X}{f\cdot\chi_A\,d\mu}.
  \end{equation}
\end{remark}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $\alpha\in\mathbb{R}$ and $f,g\in\Leb{X}{\Sigma}{\mu}$ two Lebesgue integrable functions on $\pr{X,\Sigma,\mu}$. We define the functions $\alpha\cdot f:X\rightarrow\overline{\mathbb{R}}$ and $f+g:X\rightarrow\overline{\mathbb{R}}$ as follows:

  \begin{equation}
    \begin{split}
      \pr{\alpha\cdot f}\pr{x} & =\begin{cases}
      0 & \text{if $\alpha=0$}\\
      \alpha\cdot f\pr{x} & \text{otherwise}
    \end{cases}\\\\
    \pr{f+g}\pr{x} & =\begin{cases}
      0 & \text{if $\bars{f\pr{x}}=\bars{g\pr{x}}=+\infty$ and $f\pr{x}\cdot g\pr{x}=-\infty$}\\
      f\pr{x}+g\pr{x} & \text{otherwise}
    \end{cases}
    \end{split}
  \end{equation}
\end{remark}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A,B\in\Sigma$, $Y',Y''\subseteq\overline{\mathbb{R}}$, $\alpha\in\mathbb{R}$ and, $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two $\Sigma$-measurable functions. The following properties hold:

  \begin{enumerate}
    \item If $f\in\Leb{X}{\Sigma}{\mu}$, then $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$.
    \item If $f,g\in\Leb{X}{\Sigma}{\mu}$ and $f\leq g$ $\mu$-almost everywhere in $X$, then:
    
    \begin{equation}
      \int_{X}{f\,d\mu}\leq\int_{X}{g\,d\mu}.
    \end{equation}
    \item If $f=0$ $\mu$-almost everywhere in $A$ or $\mu\pr{A}=0$, then $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\int_{A}{f\,d\mu}=0$.
    \item If $\mu\pr{A}=0$ and $\rest{f}{A^c}\in\Leb{A^c}{\Sigma_{A^c}}{\rest{\mu}{\Sigma_{A^c}}}$, then $f\in\Leb{X}{\Sigma}{\mu}$ and:
    
    \begin{equation}
      \int_{X}{f\,d\mu}=\int_{A^c}{f\,d\mu}.
    \end{equation}
    \item If $f\in\Leb{X}{\Sigma}{\mu}$, then $\alpha\cdot f\in\Leb{X}{\Sigma}{\mu}$ and:
    
    \begin{equation}
      \int_{X}{\pr{\alpha\cdot f}\,d\mu}=\alpha\cdot\int_{X}{f\,d\mu}.
    \end{equation}

    In this context, $0\cdot\pr{\pm\infty}=\pr{\pm\infty}\cdot 0=0$.
    \item If $f,g\in\Leb{X}{\Sigma}{\mu}$, then $f+g\in\Leb{X}{\Sigma}{\mu}$ and:
    
    \begin{equation}
      \int_{X}{\pr{f+g}\,d\mu}=\int_{X}{f\,d\mu}+\int_{X}{g\,d\mu}.
    \end{equation}
    \item If $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\rest{f}{B}\in\Leb{B}{\Sigma_B}{\rest{\mu}{\Sigma_B}}$, then $\rest{f}{C}\in\Leb{C}{\Sigma_C}{\rest{\mu}{\Sigma_C}}$ with $C:=A\cup B$ and:
    
    \begin{equation}
      \int_{A\cup B}{f\,d\mu}=\int_{A}{f\,d\mu}+\int_{B}{f\,d\mu}-\int_{A\cap B}{f\,d\mu}
    \end{equation}

    Particularly, if $A$ and $B$ are disjoint, then $\int_{A\cup B}{f\,d\mu}=\int_{A}{f\,d\mu}+\int_{B}{f\,d\mu}$.

    \item If $f\in\Leb{X}{\Sigma}{\mu}$ and $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of pairwise disjoint $\Sigma$-measurable sets, then:

    \begin{equation}
      \int_{\bigcup_{n=1}^{\infty}{A_n}}{f\,d\mu}=\sum_{n=1}^{\infty}{\int_{A_n}{f\,d\mu}}.
    \end{equation}
    
    \item If $f,g\in\Leb{X}{\Sigma}{\mu}$, then $\max\set{f,g},\min\set{f,g}\in\Leb{X}{\Sigma}{\mu}$.
    \item If $f\in\Leb{X}{\Sigma}{\mu}$, then $\bars{f}\in\Leb{X}{\Sigma}{\mu}$. Consequently, $\bars{f}<+\infty$ $\mu$-almost everywhere in $X$. Furthermore:
    
    \begin{equation}
      \bars{\int_{X}{f\,d\mu}}\leq\int_{X}{\bars{f}\,d\mu}.
    \end{equation}
    \item If $\mu\pr{A}<+\infty$ and $f$ is bounded on $A$, then $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and:
    
    \begin{equation}
      \bars{\int_{A}{f\,d\mu}}\leq\mu\pr{A}\cdot\sup_A\bars{f}.
    \end{equation}
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Since $f\in\Leb{X}{\Sigma}{\mu}$, we have $\int_{X}{f^+\,d\mu},\int_{X}{f^-\,d\mu}<+\infty$. Furthermore, since $A\subseteq X$, we have $\int_{A}{\rest{f^\pm}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{f^\pm\,d\mu}\leq\int_{X}{f^\pm\,d\mu}<+\infty$. Therefore, $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$.
      \item Let $N:=\set{f>g}\in\Sigma$. Then $\mu\pr{N}=0$. Observe that the function $\max\set{\cdot,0}:\mathbb{R}\rightarrow\mathbb{R}$ is increasing and the function $-\min\set{\cdot,0}:\mathbb{R}\rightarrow\mathbb{R}$ is decreasing. Then $f^+\pr{x}\leq g^+\pr{x}$ and $f^-\pr{x}\geq g^-\pr{x}$ for all $x\in N^c$. Integrating:

    \begin{equation}
        \begin{split}
          \int_{X}{f^+\,d\mu} & =\int_{N^c}{f^+\,d\mu}\leq\int_{N^c}{g^+\,d\mu}=\int_{X}{g^+\,d\mu},\\\\
          \int_{X}{f^-\,d\mu} & =\int_{N^c}{f^-\,d\mu}\geq\int_{N^c}{g^-\,d\mu}=\int_{X}{g^-\,d\mu},
        \end{split}
    \end{equation}

    Therefore:

    \begin{equation}
        \int_{X}{f\,d\mu}=\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}\leq\int_{X}{g^+\,d\mu}-\int_{X}{g^-\,d\mu}=\int_{X}{g\,d\mu}.
    \end{equation}
      \item If $f=0$ $\mu$-almost everywhere in $A$, then $f^+,f^-=0$ $\mu$-almost everywhere in $A$. Let $N:=\set{x\middle|x\in A,f\pr{x}\neq 0}$. Then $\mu\pr{N}=0$ and $f\pr{x}=0$ for all $x\in A\setminus N$. In that case, $\int_{A}{f^\pm\,d\mu}=\int_{A\setminus N}{f^\pm\,d\mu}=\int_{A\setminus N}{0\,d\mu}=0$. If $\mu\pr{A}=0$, then $\int_{A}{f^\pm\,d\mu}=0$. 
      
      In any case, $\int_{A}{f^\pm\,d\mu}=0$. Therefore, $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and:
      
      \begin{equation}
        \int_{A}{f\,d\mu}=\int_{A}{f^+\,d\mu}-\int_{A}{f^-\,d\mu}=0-0=0.
      \end{equation}

      \item Since $\mu\pr{A}=0$, then $\int_{X}{f^\pm\,d\mu}=\int_{A^c}{f^\pm\,d\mu}<+\infty$. Therefore, $f\in\Leb{X}{\Sigma}{\mu}$ and:
      
      \begin{equation}
        \int_{X}{f\,d\mu}=\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}=\int_{A^c}{f^+\,d\mu}-\int_{A^c}{f^-\,d\mu}=\int_{A^c}{f\,d\mu}.
      \end{equation}
      \item If $\alpha=0$, then it is trivial. If $\alpha>0$, then $\pr{\alpha\cdot f}^\pm=\alpha\cdot f^\pm$, and:
      
      \begin{equation}
        \int_{X}{\pr{\alpha\cdot f}^\pm\,d\mu}=\int_{X}{\pr{\alpha\cdot f^\pm}\,d\mu}=\alpha\cdot\int_{X}{f^\pm\,d\mu}<+\infty
      \end{equation}

      Thus $\alpha\cdot f\in\Leb{X}{\Sigma}{\mu}$ and:

      \begin{equation}
        \begin{split}
          \int_{X}{\pr{\alpha\cdot f}\,d\mu} & =\int_{X}{\pr{\alpha\cdot f}^+\,d\mu}-\int_{X}{\pr{\alpha\cdot f}^-\,d\mu}=\int_{X}{\pr{\alpha\cdot f^+}\,d\mu}-\int_{X}{\pr{\alpha\cdot f^-}\,d\mu}\\
          & =\alpha\cdot\int_{X}{f^+\,d\mu}-\alpha\cdot\int_{X}{f^-\,d\mu}=\alpha\cdot\pr{\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}}\\
          & =\alpha\cdot\int_{X}{f\,d\mu}.
        \end{split}
      \end{equation}

      If $\alpha<0$, then $\pr{\alpha\cdot f}^{\pm}=-\alpha\cdot f^\mp$ and:
      
      \begin{equation}
        \int_{X}{\pr{\alpha\cdot f}^{\pm}\,d\mu}=\int_{X}{\pr{\pr{-\alpha}\cdot f^\mp}\,d\mu}=\pr{-\alpha}\cdot\int_{X}{f^\mp\,d\mu}<+\infty.
      \end{equation}

      Thus:

      \begin{equation}
        \begin{split}
          \int_{X}{\pr{\alpha\cdot f}\,d\mu} & =\int_{X}{\pr{\alpha\cdot f}^+\,d\mu}-\int_{X}{\pr{\alpha\cdot f}^-\,d\mu}\\
          & =\int_{X}{\pr{\pr{-\alpha}\cdot f^-}\,d\mu}-\int_{X}{\pr{\pr{-\alpha}\cdot f^+}\,d\mu}\\
          & =\pr{-\alpha}\cdot\int_{X}{f^-\,d\mu}-\pr{-\alpha}\cdot\int_{X}{f^+\,d\mu}\\
          & =\alpha\cdot\pr{\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}}=\alpha\cdot\int_{X}{f\,d\mu}.
        \end{split}
      \end{equation}

      In any case, $\alpha\cdot f\in\Leb{X}{\Sigma}{\mu}$ and:

      \begin{equation}
        \int_{X}{\alpha\cdot f\,d\mu}=\alpha\cdot\int_{X}{f\,d\mu}.
      \end{equation}
      \item Firstly, we will prove that $f+g$ is $\Sigma$-measurable. Let $a\in\mathbb{R}$. If $f,g\in\Leb{X}{\Sigma}{\mu}$, then $\bars{f},\bars{g}<+\infty$ $\mu$-almost everywhere in $X$. Let $N_1:=\set{\bars{f}=+\infty},N_2:=\set{\bars{g}=+\infty}\in\Sigma$. Then $\mu\pr{N_1}=0$, $\mu\pr{N_2}=0$ and $\mu\pr{N}=0$ where $N:=N_1\cup N_2\in\Sigma$. Furthermore, $\bars{f\pr{x}},\bars{g\pr{x}}<+\infty$ for all $x\in N^c$. Consequently, $\rest{\pr{f+g}}{N^c}\pr{x}=f\pr{x}+g\pr{x}\in\mathbb{R}$ for all $x\in N^c$. Then $\rest{\pr{f+g}}{N^c}$ is $\Sigma_{N^c}$-measurable. Thus:
      
      \begin{equation}
        \begin{split}
          \set{f+g<a} & =\set{f+g<a}\cap\pr{N\cup N^c}=\pr{\set{f+g<a}\cap N}\cup\pr{\set{f+g<a}\cap N^c}\\
          & =\set{\rest{\pr{f+g}}{N}<a}\cup\set{\rest{\pr{f+g}}{N^c}<a}\\
          &=\set{\rest{\pr{f+g}}{N}<a}\cup\set{\rest{f}{N^c}+\rest{g}{N^c}<a}.
        \end{split}
      \end{equation}

      Observe that $\set{\rest{\pr{f+g}}{N}<a}=\pr{\set{f=-\infty}\cap\set{g<+\infty}}\cup\pr{\set{f<+\infty}\cap\set{g=-\infty}}\cup N'\in\Sigma$, where $N'=\varnothing\in\Sigma$ if $a\leq 0$ and $N'=\set{\bars{f}=+\infty}\cap\set{\bars{g}=+\infty}\cap\set{f\cdot g=-\infty}\in\Sigma$ if $a>0$. Since $\set{\rest{f}{N^c}+\rest{g}{N^c}<a}\in\Sigma_{N^c}\subseteq\Sigma$, we have $\set{f+g<a}\in\Sigma$. Therefore, $f+g$ is $\Sigma$-measurable.

      Additionally:

      \begin{equation}
        \int_{X}{\pr{f+g}^\pm\,d\mu}\leq\int_{X}{\bars{f+g}\,d\mu}\leq\int_{X}{\bars{f}\,d\mu}+\int_{X}{\bars{g}\,d\mu}<+\infty.
      \end{equation}
      
      Consequently, $f+g\in\Leb{X}{\Sigma}{\mu}$. Since $f\pr{x}+g\pr{x}=\pr{f+g}^+\pr{x}-\pr{f+g}^-\pr{x}=\pr{f^+\pr{x}-f^-\pr{x}}+\pr{g^+\pr{x}-g^-\pr{x}}$ for all $x\in N^c$, we have $\pr{f+g}^+\pr{x}+f^-\pr{x}+g^-\pr{x}=\pr{f+g}^-\pr{x}+f^+\pr{x}+g^+\pr{x}$ for all $x\in N^c$. Integrating, we have:

      \begin{equation}
        \begin{split}
          & \int_{X}{\pr{f+g}^+\,d\mu}+\int_{X}{f^-\,d\mu}+\int_{X}{g^-\,d\mu} \\
          = & \int_{N^c}{\pr{f+g}^+\,d\mu}+\int_{N^c}{f^-\,d\mu}+\int_{N^c}{g^-\,d\mu}\\
          = & \int_{N^c}{\pr{f+g}^-\,d\mu}+\int_{N^c}{f^+\,d\mu}+\int_{N^c}{g^+\,d\mu}\\
          = & \int_{X}{\pr{f+g}^-\,d\mu}+\int_{X}{f^+\,d\mu}+\int_{X}{g^+\,d\mu}.
        \end{split}
      \end{equation}

      Therefore:

      \begin{equation}
        \begin{split}
          \int_{X}{\pr{f+g}\,d\mu} & =\int_{X}{\pr{f+g}^+\,d\mu}-\int_{X}{\pr{f+g}^-\,d\mu}\\
          & =\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}+\int_{X}{g^+\,d\mu}-\int_{X}{g^-\,d\mu}\\
          & =\int_{X}{f\,d\mu}+\int_{X}{g\,d\mu}.
        \end{split}
      \end{equation}
      \item Observe that $\chi_{A\cup B}=\chi_A+\chi_B-\chi_{A\cap B}$. Since $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\rest{f}{B}\in\Leb{B}{\Sigma_B}{\rest{\mu}{\Sigma_B}}$, then $\int_{A}{f^\pm\,d\mu},\int_{B}{f^\pm\,d\mu}<+\infty$. Thus:
      
      \begin{equation}
        \begin{split}
          \int_{A\cup B}{f^{\pm}\,d\mu} & =\int_{X}{f^\pm\cdot\chi_{A\cup B}\,d\mu}\\
          & =\int_{X}{f^\pm\cdot\chi_A\,d\mu}+\int_{X}{f^\pm\cdot\chi_B\,d\mu}-\int_{X}{f^\pm\cdot\chi_{A\cap B}\,d\mu}\\
          &=\int_{A}{f^\pm\,d\mu}+\int_{B}{f^\pm\,d\mu}-\int_{A\cap B}{f^\pm\,d\mu}<+\infty.
        \end{split}
      \end{equation}

      Consequently, $\rest{f}{A\cup B}\in\Leb{A\cup B}{\Sigma_{A\cup B}}{\rest{\mu}{\Sigma_{A\cup B}}}$ and:

      \begin{equation}
        \begin{split}
          \int_{A\cup B}{f\,d\mu} & =\int_{A\cup B}{f^+\,d\mu}-\int_{A\cup B}{f^-\,d\mu}=\pr{\int_{A}{f^+\,d\mu}+\int_{B}{f^+\,d\mu}-\int_{A\cap B}{f^+\,d\mu}}\\
          & -\pr{\int_{A}{f^-\,d\mu}+\int_{B}{f^-\,d\mu}-\int_{A\cap B}{f^-\,d\mu}}\\
          & =\int_{A}{f\,d\mu}+\int_{B}{f\,d\mu}-\int_{A\cap B}{f\,d\mu}.
        \end{split}
      \end{equation}
      \item Indeed, we have that:
      
      \begin{equation}
        \begin{split}
          \int_{\bigcup_{n=1}^{\infty}{A_n}}{f\,d\mu} & =\int_{\bigcup_{n=1}^{\infty}{A_n}}{f^+\,d\mu}-\int_{\bigcup_{n=1}^{\infty}{A_n}}{f^-\,d\mu}\\
          & =\sum_{n=1}^{\infty}{\int_{A_n}{f^+\,d\mu}}-\sum_{n=1}^{\infty}{\int_{A_n}{f^-\,d\mu}}\\
          & =\sum_{n=1}^{\infty}{\pr{\int_{A_n}{f^+\,d\mu}-\int_{A_n}{f^-\,d\mu}}}=\sum_{n=1}^{\infty}{\int_{A_n}{f\,d\mu}}.
        \end{split}
      \end{equation}
      \item Since $f,g\in\Leb{X}{\Sigma}{\mu}$, we have that $N_1:=\set{\bars{f}=+\infty},N_2:=\set{\bars{g}=+\infty}\in\Sigma$ are $\Sigma$-measurable sets such that $\mu\pr{N_1}=\mu\pr{N_2}=0$. Let $N:=N_1\cup N_2\in\Sigma$. It is easy to check that $\mu\pr{N}=0$ and $f\pr{x},g\pr{x}\in\mathbb{R}$ for all $x\in N^c$. 
      
      For all $x\in N^c$, we have:
      
      \begin{equation}
        \begin{split}
          \max\set{f\pr{x},g\pr{x}} & =\frac{f\pr{x}+g\pr{x}}{2}+\frac{\bars{f\pr{x}-g\pr{x}}}{2},\\
          \min\set{f\pr{x},g\pr{x}} & =-\max\set{-f\pr{x},-g\pr{x}}.
        \end{split}
      \end{equation}

      Thus $\rest{\max\set{f,g}}{N^c}\in\Leb{N^c}{\Sigma_{N^c}}{\rest{\mu}{\Sigma_{N^c}}}$ and, consequently, $\max\set{f,g}\in\Leb{X}{\Sigma}{\mu}$. Since $-f,-g\in\Leb{X}{\Sigma}{\mu}$, we have that $\max\set{-f,-g}\in\Leb{X}{\Sigma}{\mu}$ and, therefore, $\min\set{f,g}=-\max\set{-f,-g}\in\Leb{X}{\Sigma}{\mu}$.
      \item Observe that $\bars{f}=f^++f^-$. Then:
      
      \begin{equation}
        \int_{X}{\bars{f}\,d\mu}=\int_{X}{f^+\,d\mu}+\int_{X}{f^-\,d\mu}<+\infty.
      \end{equation}

      Therefore, $\bars{f}\in\Leb{X}{\Sigma}{\mu}$. Furthermore:

      \begin{equation}
        \bars{\int_{X}{f\,d\mu}}=\bars{\int_{X}{f^+\,d\mu}-\int_{X}{f^-\,d\mu}}\leq\int_{X}{f^+\,d\mu}+\int_{X}{f^-\,d\mu}=\int_{X}{\bars{f}\,d\mu}.
      \end{equation}
      \item Since $f$ is bounded on $A$, we have $\bars{f\pr{x}}\leq\sup_A{\bars{f}}$ for all $x\in A$. Integrating on $A$, we have:
      
      \begin{equation}
        \int_{A}{f^\pm\,d\mu}\leq\int_{A}{\bars{f}\,d\mu}=\int_{A}{\sup_{A}{\bars{f}}\,d\mu}=\mu\pr{A}\cdot\sup_{A}{\bars{f}}<+\infty.
      \end{equation}

      Therefore, $\rest{f}{A}\in\Leb{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and:

      \begin{equation}
        \bars{\int_{A}{f\,d\mu}}\leq\int_{A}{\bars{f}\,d\mu}\leq\int_{A}{\sup_A{\bars{f}}\,d\mu}=\mu\pr{A}\cdot\sup_{A}{\bars{f}}.
      \end{equation}
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space and $\mathbb{K}\in\set{\mathbb{Q},\mathbb{R}}$. We have proved above that the maps $\cdot:\pr{\alpha,f}\in\mathbb{K}\times\LebR{X}{\Sigma}{\mu}\mapsto\alpha\cdot f\in\LebR{X}{\Sigma}{\mu}$ and $+:\pr{f,g}\in\LebR{X}{\Sigma}{\mu}\times \LebR{X}{\Sigma}{\mu}\mapsto f+g\in\LebR{X}{\Sigma}{\mu}$ are well defined, that is to say, $\alpha\cdot f,f+g,\in\LebR{X}{\Sigma}{\mu}$ for all $\alpha\in\mathbb{K}$ and $f,g\in\LebR{X}{\Sigma}{\mu}$. Therefore, the triple $\pr{\LebR{X}{\Sigma}{\mu},+,\cdot}$ is a $\mathbb{K}$-vector space.
\end{remark}

\begin{theorem}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\overline{\mathbb{R}}$ and $f:X\rightarrow Y'$ a $\Sigma$-measurable function. The following are equivalent:

  \begin{enumerate}
    \item $f\in\Leb{X}{\Sigma}{\mu}$.
    \item $\bars{f}\in\Leb{X}{\Sigma}{\mu}$.
    \item There exists a Lebesgue integrable function $g\in\Leb{X}{\Sigma}{\mu}$ such that $\bars{f}\leq g$ $\mu$-almost everywhere in $X$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Trivial.
      \item[\boxed{2\Rightarrow3}] It suffices to take $g=\bars{f}$.
      \item[\boxed{3\Rightarrow1}] Since $f^+,f^-\leq\bars{f}\leq g$ $\mu$-almost everywhere in $X$, we have $\int_{X}{f^\pm\,d\mu}\leq\int_{X}{g\,d\mu}<+\infty$. Therefore, $f\in\Leb{X}{\Sigma}{\mu}$.
    \end{itemize}
  \end{proof}
\end{theorem}

\begin{definition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\mathbb{C}$ and $f:X\rightarrow Y'$ a $\Sigma$-measurable function. We say that $f$ is \textit{(Lebesgue) integrable (with respect to $\pr{X,\Sigma,\mu}$)} if $\Re\pr{f},\Im\pr{f}\in\LebR{X}{\Sigma}{\mu}$. In this case, we define the {(Lebesgue) integral of $f$ over $X$ with respect to $\mu$} as follows:

  \begin{equation}
    \int_{X}{f\,d\mu}:=\int_{X}{\Re\pr{f}\,d\mu}+\int_{X}{\Im\pr{f}\,d\mu}\cdot i.
  \end{equation}

  We define the following set:
  
  \begin{equation}
      \LebC{X}{\Sigma}{\mu}:=\set{f\middle|Y'\subseteq\mathbb{C},f\in\pr{Y'}^{X}\text{ Lebesgue integrable with respect to $\pr{X,\Sigma,\mu}$}}.
  \end{equation}
\end{definition}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A\in\Sigma$ and $f\in\LebC{X}{\Sigma}{\mu}$. By the next proposition, we know that $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$. We define the \textit{(Lebesgue) integral of $f$ over $A$ with respect to $\mu$ as follows:}

  \begin{equation}
    \int_{A}{f\,d\mu}:=\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}.
  \end{equation}

  Then:

  \begin{equation}
    \begin{split}
      \int_{A}{f\,d\mu} & =\int_{A}{\rest{f}{A}\,d\rest{\mu}{\Sigma_A}}=\int_{A}{\Re\pr{\rest{f}{A}}\,d\rest{\mu}{\Sigma_A}}+\int_{A}{\Im\pr{\rest{f}{A}}\,d\rest{\mu}{\Sigma_A}}\cdot i\\
      & =\int_{A}{\rest{\Re\pr{f}}{A}\,d\rest{\mu}{\Sigma_A}}+\int_{A}{\rest{\Im\pr{f}}{A}\,d\rest{\mu}{\Sigma_A}}\cdot i=\int_{A}{\Re\pr{f}\,d\mu}+\int_{A}{\Im\pr{f}\,d\mu}\cdot i\\
      & =\int_{X}{\Re\pr{f}\cdot\chi_A\,d\mu}+\int_{X}{\Im\pr{f}\cdot\chi_A\,d\mu}\cdot i\\
      &=\int_{X}{\Re\pr{f\cdot\chi_A}\,d\mu}+\int_{X}{\Im\pr{f\cdot\chi_A}\,d\mu}\cdot i=\int_{X}{f\cdot\chi_A\,d\mu}.
    \end{split}
  \end{equation}

  Therefore:

  \begin{equation}
    \int_{A}{f\,d\mu}=\int_{A}{\Re\pr{f}\,d\mu}+\int_{A}{\Im\pr{f}\,d\mu}\cdot i=\int_{X}{f\cdot\chi_A\,d\mu}.
  \end{equation}
\end{remark}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $A,B\in\Sigma$, $Y'\subseteq\mathbb{C}$, $\alpha,\beta\in\mathbb{C}$ and $f,g:X\rightarrow Y'$ two $\Sigma$-measurable functions. The following properties hold:

  \begin{enumerate}
    \item If $f\in\LebC{X}{\Sigma}{\mu}$, then $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$.
    \item If $f\in\LebC{X}{\Sigma}{\mu}$, then $\overline{f}\in\LebC{X}{\Sigma}{\mu}$.
    \item If $f=0$ $\mu$-almost everywhere in $A$ or $\mu\pr{A}=0$, then $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\int_{A}{f\,d\mu}=0$.
    \item If $N\in\Sigma$ is a $\Sigma$-measurable set such that $\mu\pr{N}=0$ and $\rest{f}{N^c}\in\LebC{N^c}{\Sigma_{N^c}}{\rest{\mu}{\Sigma_{N^c}}}$, then $f\in\LebC{X}{\Sigma}{\mu}$ and $\int_{X}{f\,d\mu}=\int_{N^c}{f\,d\mu}$.
    \item If $f,g\in\LebC{X}{\Sigma}{\mu}$, then $\alpha\cdot f+\beta\cdot g\in\LebC{X}{\Sigma}{\mu}$ and:
    
    \begin{equation}
      \int_{X}{\pr{\alpha\cdot f+\beta\cdot g}\,d\mu}=\alpha\cdot\int_{X}{f\,d\mu}+\beta\cdot\int_{X}{g\,d\mu}.
    \end{equation}
    \item If $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\rest{f}{B}\in\LebC{B}{\Sigma_{B}}{\rest{\mu}{\Sigma_B}}$, then $\rest{f}{C}\in\LebC{C}{\Sigma_{C}}{\rest{\mu}{\Sigma_C}}$ with $C:=A\cup B$ and:
    
    \begin{equation}
      \int_{A\cup B}{f\,d\mu}=\int_{A}{f\,d\mu}+\int_{B}{f\,d\mu}-\int_{A\cap B}{f\,d\mu}
    \end{equation}

    Particularly, if $A$ and $B$ are disjoint, then $\int_{A\cup B}{f\,d\mu}=\int_{A}{f\,d\mu}+\int_{B}{f\,d\mu}$.

    \item If $f\in\LebC{X}{\Sigma}{\mu}$ and $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of pairwise disjoint $\Sigma$-measurable sets, then:
    

    \begin{equation}
      \int_{\bigcup_{n=1}^{\infty}{A_n}}{f\,d\mu}=\sum_{n=1}^{\infty}{\int_{A_n}{f\,d\mu}}.
    \end{equation}
    \item If $f\in\LebC{X}{\Sigma}{\mu}$, then $\bars{f}\in\LebR{X}{\Sigma}{\mu}$. Furthermore:
    
    \begin{equation}
      \bars{\int_{X}{f\,d\mu}}\leq\int_{X}{\bars{f}\,d\mu}.
    \end{equation}
    \item If $\mu\pr{A}<+\infty$ and $f$ is bounded on $A$, then $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and:
    
    \begin{equation}
      \bars{\int_{A}{f\,d\mu}}\leq\int_{A}{\bars{f}\,d\mu}\leq\int_{A}{\sup_A{\bars{f}}\,d\mu}=\mu\pr{A}\cdot\sup_A\bars{f}.
    \end{equation}
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Since $f\in\LebC{X}{\Sigma}{\mu}$, we have $\Re\pr{f},\Im\pr{f}\in\LebR{X}{\Sigma}{\mu}$. Then $\rest{\Re\pr{f}}{A},\rest{\Im\pr{f}}{A}\in\LebR{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$. Therefore, $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$.
      \item Since $f\in\LebC{X}{\Sigma}{\mu}$, we have $\Re\pr{f},\Im\pr{f}\in\LebR{X}{\Sigma}{\mu}$ and, consequently, $\Re\pr{\overline{f}}=\Re\pr{f},\Im\pr{\overline{f}}=-\Im\pr{f}\in\LebR{X}{\Sigma}{\mu}$. Therefore, $\overline{f}\in\LebR{X}{\Sigma}{\mu}$.
      \item If $f=0$ $\mu$-almost everywhere in $A$, then $\Re\pr{f},\Im\pr{f}=0$ $\mu$-almost everywhere in $A$ and, consequently, $\rest{\Re\pr{f}}{A},\rest{\Im\pr{f}}{A}\in\LebR{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\int_{A}{\Re\pr{f}\,d\mu},\int_{A}{\Im\pr{f}\,d\mu}=0$. If $\mu\pr{A}=0$, $\rest{\Re\pr{f}}{A},\rest{\Im\pr{f}}{A}\in\LebR{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\int_{A}{\Re\pr{f}\,d\mu},\int_{A}{\Im\pr{f}\,d\mu}=0$.

      In any case, $\rest{\Re\pr{f}}{A},\rest{\Im\pr{f}}{A}\in\LebR{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\int_{A}{\Re\pr{f}\,d\mu},\int_{A}{\Im\pr{f}\,d\mu}=0$. Therefore, $f\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and:

      \begin{equation}
        \int_{A}{f\,d\mu}=\int_{A}{\Re\pr{f}\,d\mu}+\int_{A}{\Im\pr{f}\,d\mu}\cdot i=0+0\cdot i=0.
      \end{equation}
      \item Since $\rest{f}{N^c}\in\LebC{N^c}{\Sigma_{N^c}}{\rest{\mu}{\Sigma_{N^c}}}$, then $\rest{\Re\pr{f}}{N^c},\rest{\Im\pr{f}}{N^c}\in\LebC{N^c}{\Sigma_{N^c}}{\rest{\mu}{\Sigma_{N^c}}}$. As $\mu\pr{N}=0$ and $\rest{\Re\pr{f}}{N^c},\rest{\Im\pr{f}}{N^c}\in\LebC{N^c}{\Sigma_{N^c}}{\rest{\mu}{\Sigma_{N^c}}}$, then $\Re\pr{f},\Im\pr{f}\in\LebC{X}{\Sigma}{\mu}$, $\int_{X}{\Re\pr{f}\,d\mu}=\int_{N^c}{\Re\pr{f}\,d\mu}$ and $\int_{X}{\Im\pr{f}\,d\mu}=\int_{N^c}{\Im\pr{f}\,d\mu}$. Therefore, $f\in\LebC{X}{\Sigma}{\mu}$ and:
      
      \begin{equation}
        \int_{X}{f\,d\mu}=\int_{X}{\Re\pr{f}\,d\mu}-\int_{X}{\Im\pr{f}\,d\mu}\cdot i=\int_{N^c}{\Re\pr{f}\,d\mu}-\int_{N^c}{\Im\pr{f}\,d\mu}\cdot i=\int_{N^c}{f\,d\mu}.
      \end{equation}
      \item Let $u:=\Re\pr{f}$, $v:=\Im\pr{f}$, $u':=\Re\pr{g}$, $v':=\Im\pr{g}$ and, $a,b,c,d\in\mathbb{R}$ such that $\alpha=a+b\cdot i$ and $\beta=c+d\cdot i$. Then:
      
      \begin{equation}
        \begin{split}
          \alpha\cdot f+\beta\cdot g & =\pr{a+b\cdot i}\cdot\pr{u+v\cdot i}+\pr{c+d\cdot i}\cdot\pr{u'+v'\cdot i}\\
          & =\pr{\pr{a\cdot u-b\cdot v}+\pr{b\cdot u+a\cdot v}\cdot i}\\
          & +\pr{\pr{c\cdot u'-d\cdot v'}+\pr{c\cdot v'+d\cdot u'}\cdot i}\\
          & = \pr{a\cdot u-b\cdot v+c\cdot u'-d\cdot v'}+\pr{b\cdot u+a\cdot v+c\cdot v'+d\cdot u'}\cdot i.
        \end{split}
      \end{equation}

      Since $u,v\in\LebR{X}{\Sigma}{\mu}$, then $\Re\pr{\alpha\cdot f+\beta\cdot g},\Im\pr{\alpha\cdot f+\beta\cdot g}\in\LebR{X}{\Sigma}{\mu}$. Thus $\alpha\cdot f+\beta\cdot g\in\LebC{X}{\Sigma}{\mu}$ and:

      \begin{equation}
        \begin{split}
          \,\,& \int_{X}{\pr{\alpha\cdot f+\beta\cdot g}\,d\mu}\\
          =\,\, & \int_{X}{\pr{a\cdot u-b\cdot v+c\cdot u'-d\cdot v'}\,d\mu}+\int_{X}{\pr{b\cdot u+a\cdot v+c\cdot v'+d\cdot u'}\,d\mu}\cdot i\\
          =\,\, & \pr{a\cdot\int_{X}{u\,d\mu}-b\cdot\int_{X}{v\,d\mu}+c\cdot\int_{X}{u'\,d\mu}-d\cdot\int_{X}{v'\,d\mu}}\\
          +\,\, & \pr{b\cdot\int_{X}{u\,d\mu}+a\cdot\int_{X}{v\,d\mu}+c\cdot\int_{X}{v'\,d\mu}+d\cdot\int_{X}{u'\,d\mu}}\cdot i\\
          =\,\, & \pr{a\cdot\int_{X}{u\,d\mu}-b\cdot\int_{X}{v\,d\mu}}+\pr{b\cdot\int_{X}{u\,d\mu}+a\cdot\int_{X}{v\,d\mu}}\cdot i\\
          +\,\, & \pr{c\cdot\int_{X}{u'\,d\mu}-d\cdot\int_{X}{v'\,d\mu}}+\pr{d\cdot\int_{X}{u'\,d\mu}+c\cdot\int_{X}{v'\,d\mu}}\cdot i\\
          =\,\, & \pr{a+b\cdot i}\cdot\pr{\int_{X}{u\,d\mu}+\int_{X}{v\,d\mu}\cdot i} +\pr{c+d\cdot i}\cdot\pr{\int_{X}{u'\,d\mu}+\int_{X}{v'\,d\mu}\cdot i}\\
          =\,\, & \alpha\cdot\int_{X}{f\,d\mu}+\beta\cdot\int_{X}{g\,d\mu}.
        \end{split}
      \end{equation}
      \item Since $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\rest{f}{B}\in\LebC{B}{\Sigma_{B}}{\rest{\mu}{\Sigma_B}}$, we have $\rest{\Re\pr{f}}{A},\rest{\Im\pr{f}}{A}\in\LebR{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and $\rest{\Re\pr{f}}{B},\rest{\Im\pr{f}}{B}\in\LebR{B}{\Sigma_{B}}{\rest{\mu}{\Sigma_B}}$ and, consequently, $\rest{\Re\pr{f}}{A\cup B}$, $\rest{\Im\pr{f}}{A\cup B}\in\LebR{A\cup B}{\Sigma_{A\cup B}}{\rest{\mu}{\Sigma_{A\cup B}}}$ and:
      
      \begin{equation}
        \begin{split}
          \int_{A\cup B}{\Re\pr{f}\,d\mu}=\int_{A}{\Re\pr{f}\,d\mu}+\int_{B}{\Re\pr{f}\,d\mu}-\int_{A\cap B}{\Re\pr{f}\,d\mu},\\
          \int_{A\cup B}{\Im\pr{f}\,d\mu}=\int_{A}{\Im\pr{f}\,d\mu}+\int_{B}{\Im\pr{f}\,d\mu}-\int_{A\cap B}{\Im\pr{f}\,d\mu}.
        \end{split}
      \end{equation}

      Therefore, $f\in\LebC{A\cup B}{\Sigma_{A\cup B}}{\rest{\mu}{\Sigma_{A\cup B}}}$ and:

      \begin{equation}
        \begin{split}
          \int_{A\cup B}{f\,d\mu} & =\int_{A\cup B}{\Re\pr{f}\,d\mu}+\int_{A\cup B}{\Im\pr{f}\,d\mu}\cdot i\\
          &=\pr{\int_{A}{\Re\pr{f}\,d\mu}+\int_{B}{\Re\pr{f}\,d\mu}-\int_{A\cap B}{\Re\pr{f}\,d\mu}}\\
          &+\pr{\int_{A}{\Im\pr{f}\,d\mu}+\int_{B}{\Im\pr{f}\,d\mu}-\int_{A\cap B}{\Im\pr{f}\,d\mu}}\cdot i\\
          &=\pr{\int_{A}{\Re\pr{f}\,d\mu}+\int_{A}{\Im\pr{f}\,d\mu}\cdot i}+\pr{\int_{B}{\Re\pr{f}\,d\mu}+\int_{B}{\Im\pr{f}\,d\mu}\cdot i}\\
          &-\pr{\int_{A\cap B}{\Re\pr{f}\,d\mu}+\int_{A\cap B}{\Im\pr{f}\,d\mu}\cdot i}\\
          &=\int_{A}{f\,d\mu}+\int_{B}{f\,d\mu}-\int_{A\cap B}{f\,d\mu}.
        \end{split}
      \end{equation}
      \item Indeed, we have that:
      
      \begin{equation}
        \begin{split}
          \int_{\bigcup_{n=1}^{\infty}{A_n}}{f\,d\mu} & =\int_{\bigcup_{n=1}^{\infty}{A_n}}{\Re\pr{f}\,d\mu}+\int_{\bigcup_{n=1}^{\infty}{A_n}}{\Im\pr{f}\,d\mu}\cdot i\\
          & =\sum_{n=1}^{\infty}{\int_{A_n}{\Re\pr{f}\,d\mu}}+\pr{\sum_{n=1}^{\infty}{\int_{A_n}{\Im\pr{f}\,d\mu}}}\cdot i\\
          & =\sum_{n=1}^{\infty}{\int_{A_n}{\Re\pr{f}\,d\mu}}+\sum_{n=1}^{\infty}{\pr{\int_{A_n}{\Im\pr{f}\,d\mu}\cdot i}}\\
          & =\sum_{n=1}^{\infty}{\pr{\int_{A_n}{\Re\pr{f}\,d\mu}+\int_{A_n}{\Im\pr{f}\,d\mu}\cdot i}}=\sum_{n=1}^{\infty}{\int_{A_n}{f\,d\mu}}.
        \end{split}
      \end{equation}

      \item Since $f\in\LebC{X}{\Sigma}{\mu}$, we have $\Re\pr{f},\Im\pr{f}\in\LebR{X}{\Sigma}{\mu}$. As $\bars{f\pr{x}}\leq\bars{\Re\pr{f}\pr{x}}+\bars{\Im\pr{f}\pr{x}}$ for all $x\in X$ and $\bars{\Re\pr{f}},\bars{\Im\pr{f}}\in\LebR{X}{\Sigma}{\mu}$, we have $\bars{f}\in\LebR{X}{\Sigma}{\mu}$.
      
      Let $z:=\int_{X}{f\,d\mu}\in\mathbb{C}$. Then we can take $\alpha=e^{-i\cdot\text{Arg}\pr{z}}\in\mathbb{C}$ and see that $\bars{\alpha}=1$ and $\alpha\cdot z=\bars{z}$. Therefore:

      \begin{equation}
        \begin{split}
          \bars{\int_{X}{f\,d\mu}} & =\bars{\alpha}\cdot\bars{\int_{X}{f\,d\mu}}=\bars{\alpha\cdot\int_{X}{f\,d\mu}}=\bars{\int_{X}{\pr{\alpha\cdot f}\,d\mu}}\leq\int_{X}{\bars{\alpha\cdot f}\,d\mu}\\
          & =\int_{X}{\bars{f}\,d\mu}.
        \end{split}
      \end{equation}
      \item Since $f$ is bounded on $A$, we have $\bars{f\pr{x}}\leq\sup_A{\bars{f}}$ for all $x\in A$. Integrating on $A$, we have:
      
      \begin{equation}
        \int_{A}{\Re\pr{f}^{\pm}\,d\mu},\int_{A}{\Im\pr{f}^{\pm}\,d\mu}\leq\int_{A}{\bars{f}\,d\mu}=\int_{A}{\sup_{A}{\bars{f}}\,d\mu}=\mu\pr{A}\cdot\sup_{A}{\bars{f}}<+\infty.
      \end{equation}

      Then $\rest{\Re\pr{f}}{A},\rest{\Im\pr{f}}{A}\in\LebR{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$. Therefore, $\rest{f}{A}\in\LebC{A}{\Sigma_A}{\rest{\mu}{\Sigma_A}}$ and:

      \begin{equation}
        \bars{\int_{A}{f\,d\mu}}\leq\int_{A}{\bars{f}\,d\mu}\leq\mu\pr{A}\cdot\sup_{A}{\bars{f}}.
      \end{equation}
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{remark}
  Let $\pr{X,\Sigma,\mu}$ be a measure space and $\mathbb{K}\in\set{\mathbb{Q},\mathbb{R},\mathbb{C}}$. We have proved above that the maps $\cdot:\pr{\alpha,f}\in\mathbb{K}\times\LebC{X}{\Sigma}{\mu}\mapsto\alpha\cdot f\in\LebC{X}{\Sigma}{\mu}$ and $+:\pr{f,g}\in\LebC{X}{\Sigma}{\mu}\times \LebC{X}{\Sigma}{\mu}\mapsto f+g\in\LebC{X}{\Sigma}{\mu}$ are well defined, that is to say, $\alpha\cdot f,f+g,\in\LebC{X}{\Sigma}{\mu}$ for all $\alpha\in\mathbb{K}$ and $f,g\in\LebC{X}{\Sigma}{\mu}$. Therefore, the triple $\pr{\LebC{X}{\Sigma}{\mu},+,\cdot}$ is a $\mathbb{K}$-vector space.
\end{remark}

\begin{theorem}
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\mathbb{C}$ and $f:X\rightarrow Y'$ a $\Sigma$-measurable function. The following are equivalent:

  \begin{enumerate}
    \item $f\in\LebC{X}{\Sigma}{\mu}$.
    \item $\bars{f}\in\LebR{X}{\Sigma}{\mu}$.
    \item There exists a Lebesgue integrable function $g\in\Leb{X}{\Sigma}{\mu}$ such that $\bars{f}\leq g$ $\mu$-almost everywhere in $X$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Trivial.
      \item[\boxed{2\Rightarrow3}] It suffices to take $g=\bars{f}$.
      \item[\boxed{3\Rightarrow1}] Since $\bars{\Re\pr{f\pr{x}}},\bars{\Im\pr{f\pr{x}}}\leq\bars{f\pr{x}}$ for all $x\in X$ and $\bars{f}\leq g$ $\mu$-almost everywhere in $X$, then $\Re\pr{f}^\pm,\Im\pr{f}^\pm\leq g$ $\mu$-almost everywhere in $X$. Thus:
      
      \begin{equation}
        \int_{X}{\Re\pr{f}^\pm\,d\mu},\int_{X}{\Im\pr{f}^\pm\,d\mu}\leq\int_{X}{g\,d\mu}<+\infty.
      \end{equation}

      Consequently, $\Re\pr{f},\Im\pr{f}\in\LebR{X}{\Sigma}{\mu}$ and, therefore, $f\in\LebC{X}{\Sigma}{\mu}$.
    \end{itemize}
  \end{proof}
\end{theorem}

\section{Convergence theorems}

\begin{theorem}[\textbf{Monotone Convergence Theorem}]\label{Monotone Convergence Theorem}
  Let $\pr{X,\Sigma,\mu}$, be a measure space,  $Y',Y''\subseteq\overline{\mathbb{R}}$, $\set{f_n}_{n\in\mathbb{N}}\subseteq\pr{Y'}^X$ a sequence of $\Sigma$-measurable functions and $f:X\rightarrow Y''$ a $\Sigma$-measurable function. Suppose the following:
  
  \begin{itemize}
    \item For each $n\in\mathbb{N}$, $f_n\leq f_{n+1}$ $\mu$-almost everywhere in $X$.
    \item $f=\lim_{n\to\infty}{f_n}$ $\mu$-almost everywhere where the limit exists.
  \end{itemize}
  
  The following holds:

  \begin{enumerate}
    \item If $f_n\geq 0$ for all $n\in\mathbb{N}$ $\mu$-almost everywhere in $X$, then $\lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}$ exists and:
    
    \begin{equation}
      \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\int_{X}{f\,d\mu}.
    \end{equation}
    \item If $f_n\in\Leb{X}{\Sigma}{\mu}$ for all $n\in\mathbb{N}$, then $f\in\Leb{X}{\Sigma}{\mu}$ if and only if $\lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\sup_{n\in\mathbb{N}}{\int_{X}{f_n\,d\mu}}<+\infty$ and, in that case, we have that:
    
    \begin{equation}
      \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\int_{X}{f\,d\mu}.
    \end{equation}
  \end{enumerate}

  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item We define the following sets:
      
      \begin{equation}
        \begin{split}
          N_1 & :=\bigcup_{n=1}^{\infty}{\set{f_n>f_{n+1}}},\\
          N_2 & :=\bigcup_{n=1}^{\infty}{\set{f_n<0}},\\
          N_3 & :=\set{\liminf_{n\to\infty}{f_n}\neq\limsup_{n\to\infty}{f_n}},\\
          N_4 & :=\set{f\neq\liminf_{n\to\infty}{f_n}}\in\Sigma,\\
          N & :=N_1\cup N_2\cup N_3\cup N_4.
        \end{split}
      \end{equation}
      
      It is easy to check that $\set{f_n>f_{n+1}},\set{f_n<0},N_3,N_4\in\Sigma$ for all $n\in\mathbb{N}$ and, consequently, $N_1,N_2,N\in\Sigma$ as well. Furthermore, we know that $\mu\pr{\set{f_n>f_{n+1}}}=\mu\pr{\set{f_n<0}}=\mu\pr{N_3}=\mu\pr{N_4}=0$ for all $n\in\mathbb{N}$ and, consequently, by $\sigma$-subadditivity, we have $\mu\pr{N_1}=\mu\pr{N_2}=\mu\pr{N}=0$.

      Then, for all $x\in N^c$, $0\leq f_n\pr{x}\leq f_{n+1}\pr{x}$ for all $n\in\mathbb{N}$, $\lim_{n\to\infty}{f_n\pr{x}}$ exists and $f\pr{x}=\lim_{n\to\infty}{f_n\pr{x}}$. Thus $f_n\pr{x}\leq f_{n+1}\pr{x}\leq\sup_{n\in\mathbb{N}}{f_n\pr{x}}=\lim_{n\to\infty}{f_n\pr{x}}=f\pr{x}$ for all $x\in N^c$ and:

      \begin{equation}
        \int_{X}{f_n\,d\mu}\leq\int_{X}{f_{n+1}\,d\mu}\leq\int_{X}{f\,d\mu}.
      \end{equation}

      Consequently:

      \begin{equation}
        \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\sup_{n\in\mathbb{N}}{\int_{X}{f_n\,d\mu}}\leq\int_{X}{f\,d\mu}.
      \end{equation}

      If $\sup_{n\in\mathbb{N}}{\int_{X}{f_n\,d\mu}}=+\infty$, the equality holds. Suppose $\sup_{n\in\mathbb{N}}{\int_{X}{f_n\,d\mu}}<+\infty$. Let $c\in\pr{0,1}$ and $\varphi\in\mathcal{S}\pr{X,\Sigma}$ such that $0\leq\varphi\pr{x}\leq f\pr{x}$ for all $x\in X$. For each $n\in\mathbb{N}$, we define $E_n:=\set{x\middle|x\in N^c,f_n\pr{x}\geq c\cdot\varphi\pr{x}}=\set{f_n\geq c\cdot\varphi}\cap N^c\in\Sigma$. Since $f_n\pr{x}\leq f_{n+1}\pr{x}$ for all $x\in N^c$, then $E_n\subseteq E_{n+1}$ for all $n\in\mathbb{N}$. 

      Let's prove that $N^c=\bigcup_{n=1}^{\infty}{E_n}$. Let $x\in N^c$. If $f\pr{x}=0$, then $\varphi\pr{x}=0$ and $x\in\set{f_n\geq c\cdot\varphi}\cap N^c\subseteq E_n\subseteq\bigcup_{m=1}^{\infty}{E_m}$. Suppose $f\pr{x}>0$. Since $c\in\pr{0,1}$, then $c\cdot\varphi\pr{x}<f\pr{x}$. Using $\lim_{n\to\infty}{f_n\pr{x}}=f\pr{x}$, there exists $n_0=n_0\pr{x}\in\mathbb{N}$ such that $c\cdot\varphi\pr{x}<f_n\pr{x}\leq f\pr{x}$ for all $n\in\mathbb{N}$ with $n\geq n_0$. Then $x\in E_{n_0}\subseteq\bigcup_{m=1}^{\infty}{E_m}$. Therefore, $N^c=\bigcup_{n=1}^{\infty}{E_n}$.

      For each $n\in\mathbb{N}$, we have:

      \begin{equation}
        \int_{X}{f_n\,d\mu}\geq\int_{E_n}{f_n\,d\mu}\geq\int_{E_n}{\pr{c\cdot\varphi}\,d\mu}=c\cdot\int_{E_n}{\varphi\,d\mu}.
      \end{equation}

      Since the map $\nu_{c\cdot\varphi}:E\in\Sigma\mapsto\int_{E}{\pr{c\cdot\varphi}\,d\mu}\in\br{0,+\infty}$ is a measure on $\pr{X,\Sigma}$ and $E_n\subseteq E_{n+1}$ for all $n\in\mathbb{N}$, we have:

      \begin{equation}
        \begin{split}
          c\cdot\int_{X}{\varphi\,d\mu} & =\int_{X}{\pr{c\cdot\varphi}\,d\mu}=\int_{N^c}{\pr{c\cdot\varphi}\,d\mu}=\nu_{c\cdot\varphi}\pr{N^c}=\nu_{c\cdot\varphi}\pr{\bigcup_{n=1}^{\infty}{E_n}}\\
          & =\lim_{n\to\infty}{\nu_{c\cdot\varphi}\pr{E_n}}=\lim_{n\to\infty}{\int_{E_n}{\pr{c\cdot\varphi}\,d\mu}}\leq\lim_{n\to\infty}{\int_{X}{f_n}\,d\mu}.
        \end{split}
      \end{equation}

      Taking the limit when $c\to 1^-$, we obtained $\int_{X}{\varphi\,d\mu}\leq\lim_{n\to\infty}{\int_{X}{f_n}\,d\mu}$. Thus $\int_{X}{f\,d\mu}\leq\lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}$. Therefore:

      \begin{equation}
        \lim_{n\to\infty}{\int_{X}{f_n}\,d\mu}=\int_{X}{f\,d\mu}.
      \end{equation}
      \item We define the following sets:
      
      \begin{equation}
        \begin{split}
          N_1 & :=\bigcup_{n=1}^{\infty}{\set{f_n>f_{n+1}}},\\
          N_2 & :=\set{\liminf_{n\to\infty}{f_n}\neq\limsup_{n\to\infty}{f_n}},\\
          N_3 & :=\set{f\neq\liminf_{n\to\infty}{f_n}}\in\Sigma,\\
          N & :=N_1\cup N_2\cup N_3.
        \end{split}
      \end{equation}
      
      It is easy to check that $\set{f_n>f_{n+1}},N_2,N_3\in\Sigma$ for all $n\in\mathbb{N}$ and, consequently, $N_1,N_2,N\in\Sigma$ as well. Furthermore, we know that $\mu\pr{\set{f_n>f_{n+1}}}=\mu\pr{N_2}=\mu\pr{N_3}=0$ for all $n\in\mathbb{N}$ and, consequently, by $\sigma$-subadditivity, we have $\mu\pr{N_1}=\mu\pr{N}=0$.

      If $f\in\Leb{X}{\Sigma}{\mu}$, then, for all $x\in N^c$, $f_n\pr{x}\leq f_{n+1}\pr{x}$ for all $n\in\mathbb{N}$, $\lim_{n\to\infty}{f_n\pr{x}}$ exists and $f\pr{x}=\lim_{n\to\infty}{f_n\pr{x}}$. Thus $f_n\pr{x}\leq f_{n+1}\pr{x}\leq\sup_{n\in\mathbb{N}}{f_n\pr{x}}=\lim_{n\to\infty}{f_n\pr{x}}=f\pr{x}$ for all $x\in N^c$ and:

      \begin{equation}
        \int_{X}{f_n\,d\mu}\leq\int_{X}{f_{n+1}\,d\mu}\leq\int_{X}{f\,d\mu}.
      \end{equation}

      Therefore:

      \begin{equation}
        \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\sup_{n\in\mathbb{N}}{\int_{X}{f_n\,d\mu}}\leq\int_{X}{f\,d\mu}<+\infty.
      \end{equation}

      Now, suppose that $\lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\sup_{n\in\mathbb{N}}{\int_{X}{f_n\,d\mu}}<+\infty$. Consider the set $N':=\bigcup_{n=1}^{\infty}{\set{\bars{f_n}=+\infty}}\in\Sigma$. Since $f_n\in\Leb{X}{\Sigma}{\mu}$ for all $n\in\mathbb{N}$, we have that $\mu\pr{\set{\bars{f_n}=+\infty}}=0$ for all $n\in\mathbb{N}$ and, consequently, $\mu\pr{N'}=0$ by $\sigma$-subadditivity. For each $n\in\mathbb{N}$, let $g_n:\pr{N\cup N'}^c\rightarrow\mathbb{R}$ be defined as $g_n=f_n-f_1$. Observe that $\mu\pr{N\cup N'}=0$ by $\sigma$-subadditivity. Let $g:\pr{N\cup N'}^c\rightarrow\mathbb{R}$ be defined as $g:=f-f_1$. Then $\set{g_n}_{n\in\mathbb{N}}\subseteq\mathbb{R}^{\pr{N\cup N'}^c}$ is a sequence of $\Sigma_{\pr{N\cup N'}^c}$-measurable functions and $g$ is a $\Sigma_{\pr{N\cup N'}^c}$-measurable function such that:
      
      \begin{itemize}
        \item $g_n\pr{x}\leq g_{n+1}\pr{x}$ for all $x\in\pr{N\cup N'}^c$.
        \item $g\pr{x}=\lim_{n\to\infty}{g_n\pr{x}}$ for all $x\in\pr{N\cup N'}^c$.
        \item For each $n\in\mathbb{N}$, $g_n\pr{x}\geq 0$ for all $x\in\pr{N\cup N'}^c$.
      \end{itemize}
      
      Then:

      \begin{equation}
        \begin{split}
          \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}-\int_{X}{f_1\,d\mu} & =\lim_{n\to\infty}{\int_{\pr{N\cup N'}^c}{f_n\,d\mu}}-\int_{\pr{N\cup N'}^c}{f_1\,d\mu}\\
          & =\lim_{n\to\infty}{\pr{\int_{\pr{N\cup N'}^c}{f_n\,d\mu}-\int_{\pr{N\cup N'}^c}{f_1\,d\mu}}}\\
          & =\lim_{n\to\infty}{\int_{\pr{N\cup N'}^c}{g_n\,d\mu}}=\int_{\pr{N\cup N'}^c}{g\,d\mu}=\int_{X}{g\,d\mu}.
        \end{split}
      \end{equation}

      Thus, $g\in\Leb{X}{\Sigma}{\mu}$. Since $f\pr{x}=g\pr{x}+f_1\pr{x}$ for all $x\in\pr{N\cup N'}^c$, we have that $f\in\Leb{\pr{N\cup N'}^c}{\Sigma_{\pr{N\cup N'}^c}}{\rest{\mu}{\Sigma_{\pr{N\cup N'}^c}}}$ and, therefore, $f\in\Leb{X}{\Sigma}{\mu}$. Furthermore, we have:

      \begin{equation}
        \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\int_{X}{f\,d\mu}.
      \end{equation}
    \end{enumerate}
  \end{proof}
\end{theorem}

\begin{corollary}
  Let $\pr{X,\Sigma,\mu}$ be a measure space and $\set{f_n}_{n\in\mathbb{N}}\subseteq\overline{\mathbb{R}}^X$ a sequence of non-negative $\Sigma$-measurable functions. The following properties hold:
  \begin{enumerate}
    \item Beppo Levi's Theorem: $\int_{X}{\pr{\sum_{n=1}^{\infty}{f_n}}\,d\mu}=\sum_{n=1}^{\infty}{\int_{X}{f_n\,d\mu}}$.\label{Beppo Levi's Theorem}
    \item If $\sum_{n=1}^{\infty}{\int_{X}{f_n}<+\infty}$, then $\sum_{n=1}^{\infty}{f_n}\in\Leb{X}{\Sigma}{\mu}$. Futhermore, $\sum_{n=1}^{\infty}{f_n\pr{x}}$ exists and belongs to $\mathbb{R}$ $\mu$-almost everywhere in $X$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item It is a direct consequence of \hyperref[Monotone Convergence Theorem]{Monotone Convergence Theorem} applied to the sequence $\pr{\sum_{k=1}^n{f_k}}_{n\in\mathbb{N}}\subseteq\overline{\mathbb{R}}^X$ and the series of functions $\sum_{n=1}^{\infty}{f_n}:X\rightarrow\overline{\mathbb{R}}$.
      \item Trivial by the \hyperref[Beppo Levi's Theorem]{Beppo Levi's Theorem}.
    \end{enumerate}
  \end{proof}
\end{corollary}

\begin{theorem}[\rm\textbf{Fatou's Lemma}]
  Let $\pr{X,\Sigma,\mu}$ be a measure space and $\set{f_n}_{n\in\mathbb{N}}\subseteq\overline{\mathbb{R}}^X$ a sequence of non-negative $\Sigma$-measurable functions. Then:

  \begin{equation}
    \int_{X}{\liminf_{n\to\infty}{f_n}\,d\mu}=\liminf_{n\to\infty}{\int_{X}{f_n}\,d\mu}.
  \end{equation}
\end{theorem}

\begin{theorem}[\rm\textbf{Dominated Convergence Theorem}]
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y',Y''\subseteq\overline{\mathbb{R}}$, $\set{f_n}_{n\in\mathbb{N}}\subseteq\overline{\mathbb{R}}^X$ a sequence of $\Sigma$-measurable functions and, $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two $\Sigma$-measurable functions verifying:

  \begin{itemize}
    \item $g\in\Leb{X}{\Sigma}{\mu}$.
    \item $\bars{f_n}\leq g$ $\mu$-almost everywhere in $X$.
    \item Exists $\lim_{n\to\infty}{f_n}$ $\mu$-almost everywhere in $X$.
    \item $f=\lim_{n\to\infty}{f_n}$ $\mu$-almost everywhere where the limit exists.
  \end{itemize}

  Then $f\in\Leb{X}{\Sigma}{\mu}$ and:

  \begin{equation}
    \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\int_{X}{f\,d\mu}.
  \end{equation}
\end{theorem}

\begin{theorem}[\rm\textbf{Dominated Convergence Theorem on $\mathbb{C}$}]
  Let $\pr{X,\Sigma,\mu}$ be a measure space, $Y'\subseteq\mathbb{C}$ and $Y''\subseteq\overline{\mathbb{R}}$, $\set{f_n}_{n\in\mathbb{N}}\subseteq\mathbb{C}^X$ a sequence of $\Sigma$-measurable functions and, $f:X\rightarrow Y'$ and $g:X\rightarrow Y''$ two $\Sigma$-measurable functions verifying:

  \begin{itemize}
    \item $g\in\Leb{X}{\Sigma}{\mu}$.
    \item $\bars{f_n}\leq g$ $\mu$-almost everywhere in $X$.
    \item Exists $\lim_{n\to\infty}{f_n}$ $\mu$-almost everywhere in $X$.
    \item $f=\lim_{n\to\infty}{f_n}$ $\mu$-almost everywhere where the limit exists.
  \end{itemize}

  Then $f\in\Leb{X}{\Sigma}{\mu}$ and:

  \begin{equation}
    \lim_{n\to\infty}{\int_{X}{f_n\,d\mu}}=\int_{X}{f\,d\mu}.
  \end{equation}
\end{theorem}

\begin{proposition}
  Let $\pr{X,\Sigma,\mu}$ be a measure space. The following properties hold:

  \begin{itemize}
    \item If $\set{f_n}_{n\in\mathbb{N}}\subseteq\overline{\mathbb{R}}^X$ is a sequence of $\Sigma$-measurable functions such that:
    
    \begin{equation}
      \sum_{n=1}^{\infty}{\int_{X}{\bars{f_n}\,d\mu}}<+\infty,
    \end{equation}

    then $\sum_{n=1}^{\infty}{f_n}$ converges absolutely $\mu$-almost everywhere in $X$. Furthermore, if $S:X\rightarrow\overline{\mathbb{R}}$ is a $\Sigma$-measurable function such that $S=\sum_{n=1}^{\infty}{f_n}$ $\mu$-almost everywhere where the sum exists, then $S\in\Leb{X}{\Sigma}{\mu}$ and:

    \begin{equation}
      \sum_{n=1}^{\infty}{\int_{X}{f_n}\,d\mu}=\int_{X}{S\,d\mu}.
    \end{equation}
    \item If $\set{f_n}_{n\in\mathbb{N}}\subseteq\mathbb{C}^X$ is a sequence of $\Sigma$-measurable functions such that:
    
    \begin{equation}
      \sum_{n=1}^{\infty}{\int_{X}{\bars{f_n}\,d\mu}}<+\infty,
    \end{equation}

    then $\sum_{n=1}^{\infty}{f_n}$ converges absolutely $\mu$-almost everywhere in $X$. Furthermore, if $S:X\rightarrow\mathbb{C}$ is a $\Sigma$-measurable function such that $S=\sum_{n=1}^{\infty}{f_n}$ $\mu$-almost everywhere where the sum exists, then $S\in\LebC{X}{\Sigma}{\mu}$ and:

    \begin{equation}
      \sum_{n=1}^{\infty}{\int_{X}{f_n}\,d\mu}=\int_{X}{S\,d\mu}.
    \end{equation}
  \end{itemize}
\end{proposition}
