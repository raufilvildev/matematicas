\chapter{Introduction to Measure Theory}

\section{Measurable spaces}

\begin{definition}
  Let $X$ be a set. A \emph{$\sigma$-algebra on $X$} is a collection $\Sigma\subseteq\mathcal{P}\pr{X}$ of subsets of $X$ satisfying the following conditions:

  \begin{itemize}
    \item $\varnothing\in\Sigma$.
    \item If $A\in\Sigma$, then   $A^c:=X\setminus A\in\Sigma$.
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of sets in $\Sigma$, then $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma$.
  \end{itemize}

  The ordered pair $\pr{X,\Sigma}$ is called a \emph{measurable space}. The elements of $\Sigma$ are called \textit{measurable sets}.
\end{definition}

\begin{example}
  \rm Let $X$ be a set. Some examples of $\sigma$-algebras on $X$ are the following:

  \begin{itemize}
    \item Trivial (or indiscrete) $\sigma$-algebra: $\Sigma:=\set{\varnothing,X}$.
    \item Discrete $\sigma$-algebra: $\Sigma:=\mathcal{P}\pr{X}$.
    \item Countable-cocountable $\sigma$-algebra: $\Sigma:=\set{A\middle|A\subseteq X,A\text{ or }A^c\text{ is countable}}$.
  \end{itemize}
\end{example}

\begin{proposition}
  Let $\pr{X,\Sigma}$ be a measurable space. 

  \begin{itemize}
    \item[1.] $X\in\Sigma$.
    \item[2.] If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of sets in $\Sigma$, then $\bigcap_{n=1}^{\infty}{A_n}\in\Sigma$.
    \item[3.] If $A_1,\dots,A_n\in\Sigma$ with $n\in\mathbb{N}$, then $\bigcup_{k=1}^{n}{A_k},\bigcap_{k=1}^{n}{A_k}\in\Sigma$.
    \item[4.] If $A,B\in\Sigma$, then $A\setminus B\in\Sigma$.
  \end{itemize}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[1.] This is immediate, since $X:=\varnothing^c$.
      \item[2.] For each $n\in\mathbb{N}$, since $A_n\in\Sigma$, we have $A_n^c\in\Sigma$. Then $\bigcup_{n=1}^{\infty}{A_n^c}\in\Sigma$. By De Morgan's law, $\bigcap_{n=1}^{\infty}{A_n}=\pr{\bigcup_{n=1}^{\infty}{A_n^c}}^c\in\Sigma$.
      \item[3.] Given $m\in\mathbb{N}$, we define:
      
      \begin{equation}
        A'_m=\begin{cases}
          A_m & \text{if $m\leq n$}\\
          \varnothing & \text{if $m>n$}
        \end{cases}\hspace{0.5cm}A''_m=\begin{cases}
          A_m & \text{if $m\leq n$}\\
          X & \text{if $m>n$}
        \end{cases}
      \end{equation}

      Then $\set{A'_m}_{m\in\mathbb{N}},\set{A''_m}_{m\in\mathbb{N}}\subseteq\Sigma$. Therefore, $\bigcup_{k=1}^{n}{A_k}=\bigcup_{m=1}^{\infty}{A'_m},\bigcap_{k=1}^{n}{A_k}=\bigcap_{m=1}^{\infty}{A''_m}\in\Sigma$.
      \item[4.] Since $B\in\Sigma$, we have $B^c\in\Sigma$. Therefore, $A\setminus B=A\cap B^c\in\Sigma$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\mathscr{S}$ a family of $\sigma$-algebras on a set $X$. Then $\Sigma:=\bigcap{\mathscr{S}}\subseteq\mathcal{P}\pr{X}$ is a $\sigma$-algebra on $X$.
  \begin{proof}[\rm\textbf{Proof}]
    Since $\varnothing\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$, we have $\varnothing\in\Sigma$. 
    
    Let $A\in\Sigma$. Then $A\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Thus $A^c\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Therefore, $A^c\in\Sigma$. 
    
    Let $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ be a countable collection of sets in $\Sigma$. Then $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of sets in $\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Thus $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma'$ for all $\Sigma'\in\mathscr{S}$. Therefore $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma$.

    In conclusion, $\Sigma$ is a $\sigma$-algebra on $X$.
  \end{proof}
\end{proposition}

\begin{definition}
  Let $X$ be a set and $\mathcal{S}\subseteq\mathcal{P}\pr{X}$ a collection of subsets of $X$. We define the \textit{$\sigma$-algebra generated by $\mathcal{S}$ on $X$} as follows:

  \begin{equation}
    \sigma\pr{X,\mathcal{S}}:=\bigcap\set{\Sigma\middle|\Sigma\subseteq\mathcal{P}\pr{X},\mathcal{S}\subseteq\Sigma, \Sigma\text{ is a $\sigma$-algebra on $X$}}.
  \end{equation}
\end{definition}

\begin{remark}
  If $\pr{X,\Sigma}$ is a measurable space and $\mathcal{S}\subseteq\Sigma$, then $\sigma\pr{X,\mathcal{S}}\subseteq\Sigma$ by the definition of $\sigma\pr{X,\mathcal{S}}$.
\end{remark}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space.The \textit{Borel $\sigma$-algebra on $\pr{X,\mathcal{T}}$} is defined by $\mathscr{B}\pr{X,\mathcal{T}}:=\sigma\pr{X,\mathcal{T}}$. The ordered pair $\pr{X,\mathscr{B}\pr{X,\mathcal{T}}}$ is called a \textit{Borel measurable space}.

  If $\pr{X,\mathcal{T}}=\pr{\mathbb{K}^n,\mathcal{T}_e\pr{\mathbb{K}^n}}$ where $\mathbb{K}\in\set{\mathbb{R},\mathbb{C}}$ and $n\in\mathbb{N}$, we write $\mathscr{B}\pr{\mathbb{K}^n}:=\mathscr{B}\pr{\mathbb{K}^n,\mathcal{T}_e\pr{\mathbb{K}^n}}$.
\end{definition}

\begin{proposition}
  \rm Let $\pr{X,\Sigma}$ be a measurable space and $S\subseteq X$. The collection $\Sigma_S:=\set{A\cap S\middle|A\in\Sigma}\subseteq\mathcal{P}\pr{S}$ is a $\sigma$-algebra on $S$ called the \textit{$\sigma$-algebra induced by $S$ on $\pr{X,\Sigma}$}.
  \begin{proof}[\rm\textbf{Proof}]
    Since $\varnothing=\varnothing\cap S$, we have $\varnothing\in\Sigma_S$. 
    
    Let $A'\in\Sigma_S$. Then there exists $A\in\Sigma$ such that $A'=A\cap S$. Thus $S\setminus A'=S\setminus\pr{A\cap S}=S\cap\pr{A\cap S}^c=S\cap A^c$. Since $A^c\in\Sigma$, we have $S\setminus A'\in\Sigma_S$. 
    
    Let $\set{A'_n}_{n\in\mathbb{N}}\subseteq\Sigma_S$ be a countable collection of sets in $\Sigma_S$. Then there exists a countable collection of sets $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ in $\Sigma$ such that $A'_n=A_n\cap S$. Since $\bigcup_{n=1}^{\infty}{A_n}\in\Sigma$, we have $\bigcup_{n=1}^{\infty}{A'_n}=\bigcup_{n=1}^{\infty}{\pr{A_n\cap S}}=\pr{\bigcup_{n=1}^{\infty}{A_n}}\cap S\in\Sigma_S$.

    Therefore, $\Sigma_S$ is a $\sigma$-algebra on $S$.
  \end{proof}
\end{proposition}

\section{Measure spaces}

\begin{definition}
  \rm Let $\pr{X,\Sigma}$ be a measurable space. A \emph{(positive) measure on $\pr{X,\Sigma}$} is a map $\mu:\Sigma\rightarrow\br{0,+\infty}$ satisfying the following conditions:

  \begin{itemize}
    \item $\mu\pr{\varnothing}=0$.
    \item If $\set{A_n}_{n\in\mathbb{N}}\subseteq\Sigma$ is a countable collection of pairwise disjoint sets in $\Sigma$, then $\mu\pr{\bigcup_{n=1}^{\infty}{A_n}}=\sum_{n=1}^{\infty}{\mu\pr{A_n}}$.
  \end{itemize}

  The triple $\pr{X,\Sigma,\mu}$ is called a \emph{measure space}.
\end{definition}
