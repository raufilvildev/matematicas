\chapter{Introduction to General Topology}

\section{Topological spaces}

\begin{definition}
  A \emph{topology on a set $X$ }is a collection $\mathcal{T}\subseteq\mathcal{P}\pr{X}$ of subsets of $X$ satisfying the following properties:

  \begin{enumerate}
      \item $\varnothing,X\in\mathcal{T}$.
      \item If $G_1,G_2\in\mathcal{T}$, then $G_1\cap G_2\in\mathcal{T}$.
      \item If $\mathcal{G}\subseteq\mathcal{T}$, then $\bigcup{\mathcal{G}}\in\mathcal{T}$.
  \end{enumerate}

  The ordered pair $\pr{X,\mathcal{T}}$ is called a \emph{topological space}. The elements of $\mathcal{T}$ are called \emph{open sets}.
\end{definition}

\begin{remark}
  Observe that if $\pr{X,\mathcal{T}}$ is a topological space, then the finite intersection of open sets is an open set. This can be easily proved by induction.
\end{remark}

\begin{example}
  Let $X$ be a set. Some examples of topologies on $X$ are the following:
  
  \begin{itemize}
      \item \textbf{Discrete topology}: It is defined as the topology on $X$ which consists of all subsets of $X$, that is to say:

      \begin{equation}
          \mathcal{T}_{\text{dis}}:=\mathcal{P}\pr{X}.
      \end{equation}

      It is straightforward to verify that $\mathcal{T}_{\text{dis}}$ is a topology on $X$, because $\varnothing,X\subseteq X$ and, the union and intersection of any arbitrary collection of subsets of $X$ is a subset of $X$.
      \item \textbf{Indiscrete or trivial topology}: It is defined as the topology which consists of $\varnothing$ and $X$, that is to say:
      
      \begin{equation}
          \mathcal{T}_{\text{ind}}:=\set{\varnothing,X}
      \end{equation}

      One readily checks that $\mathcal{T}_{\text{ind}}$ is a topology on $X$, because $\varnothing,X\in\mathcal{T}_{\text{ind}}$ and $\varnothing\cup X=X,\varnothing\cap X=\varnothing\in\mathcal{T}$.
      \item \textbf{Sierpinski's topology}: If $X=\set{0,1}$, this topology on $X$ is defined as $\mathcal{T}_{S}:=\set{\varnothing,\set{0},X}$.
      
      It is straightforward to verify that $\mathcal{T}_{S}$ is a topology on $X$.
      \item \textbf{Cofinite topology}: It is defined as the topology on $X$ which consists of $\varnothing$ and those subsets of $X$ whose complement in $X$ is finite, that is to say:

      \begin{equation}
          \mathcal{T}_{\text{cof}}:=\set{G\middle|G\subseteq X, G=\varnothing\text{ or }X\setminus G\text{ is finite}}.
      \end{equation}

      Clearly, $\varnothing,X\in\mathcal{T}_{\text{cof}}$. 
      
      Let $G_1,G_2\in\mathcal{T}_{\text{cof}}$. If $G_1\cap G_2=\varnothing$, then there is nothing to prove. Suppose that $G_1\cap G_2\neq\varnothing$. Then $X\setminus\pr{G_1\cap G_2}=\pr{X\setminus G_1}\cup\pr{X\setminus G_2}$. Since the finite union of finite sets is a finite set, we deduce that $G_1\cap G_2\in\mathcal{T}_{\text{cof}}$. 
      
      Lastly, let $\mathcal{G}\subseteq\mathcal{T}_{\text{cof}}$. If $\bigcup{\mathcal{G}}=\varnothing$, then there is nothing to prove. Suppose there exists a non-empty set $G_0\in\mathcal{G}$. Then $X\setminus\pr{\bigcup{\mathcal{G}}}=\bigcap_{G\in\mathcal{G}}{\pr{X\setminus G}}\subseteq X\setminus G_0$. As $X\setminus G_0$ is finite, we deduce that $X\setminus\pr{\bigcup_{G\in\mathcal{G}}{G}}$ is finite too. Thus $\bigcup{{\mathcal{G}}}\in\mathcal{T}_{\text{cof}}$. 
      
      Therefore, $\mathcal{T}_{\text{cof}}$ is a topology on $X$.
      \item \textbf{Cocountable topology}: It is defined as the topology on $X$ which consists of $\varnothing$ and those subsets of $X$ whose complement in $X$ is countable, that is to say:
      
      \begin{equation}
          \mathcal{T}_{\text{cc}}:=\set{G\middle|G\subseteq X, G=\varnothing\text{ or }X\setminus G\text{ is countable}}.
      \end{equation}

      The proof is analogous to the one we did above.
      \item \textbf{Included point topology}: Suppose $X\neq\varnothing$. Given $p\in X$, this topology is defined as the topology on $X$ which consists of $\varnothing$ and all the subsets of $X$ containing $p$, that is to say: 
      
      \begin{equation}
        \mathcal{T}_p^{\text{inc}}:=\set{G\middle|G\subseteq X, G=\varnothing\text{ or }p\in G}.
      \end{equation}
      
      It is immediate that $\mathcal{T}_p^{\text{inc}}$ is a topology on $X$, because $\varnothing,X\in\mathcal{T}_p^{\text{inc}}$ and, the union and intersection of any arbitrary collection of subsets of $X$ containing $p$ is a subset of $X$ containing $p$.
      \item \textbf{Excluded point topology}: Suppose $X\neq\varnothing$. Given $p\in X$, this topology is defined as the topology on $X$ which consists of $X$ and all the subsets of $X$ not containing $p$, that is to say: 
      
      \begin{equation}
        \mathcal{T}_p^{\text{exc}}:=\set{G\middle|G\subseteq X, G=X\text{ or }p\notin G}.
      \end{equation}
      
      It is immediate that $\mathcal{T}_p^{\text{exc}}$ is a topology on $X$, because $\varnothing,X\in\mathcal{T}_p^{\text{exc}}$ and, the union and intersection of any arbitrary collection of subsets of $X$ not containing $p$ is a subset of $X$ not containing $p$.
  \end{itemize}
\end{example}

In general, the arbitrary intersection of open sets in a topological space is not an open set.

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. We say that a subset of $X$ is $G_{\delta}$ if it is the intersection of a countable collection of open sets.
\end{definition}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. A subset of $X$ is called \emph{closed} (with respect to $\pr{X,\mathcal{T}}$) if the complement in $X$ is an open set. The collection of all closed sets is denoted by $\mathcal{F}\pr{X,\mathcal{T}}$.
\end{definition}

\begin{example}
  Let $X$ be a set. Some examples of closed sets in different topologies are the following:
  \begin{itemize}
      \item \textbf{Discrete topology}: $\mathcal{F}\pr{X,\mathcal{T}_{\text{dis}}}=\mathcal{T}_{\text{dis}}=\mathcal{P}\pr{X}$.
      \item \textbf{Indiscrete or trivial topology}: $\mathcal{F}\pr{X,\mathcal{T}_{\text{ind}}}=\mathcal{T}_{\text{ind}}=\set{\varnothing,X}$.
      \item \textbf{Sierpinski's topology}: If $X=\set{0,1}$, $\mathcal{F}\pr{X,\mathcal{T}_S}=\set{\varnothing,\set{1},X}$.
      \item \textbf{Cofinite topology}: $\mathcal{F}\pr{X,\mathcal{T}_{\text{cof}}}=\set{F\middle|F\subseteq X,F=X\text{ or }F\text{ is finite}}$.
      \item \textbf{Cocountable topology}: $\mathcal{F}\pr{X,\mathcal{T}_{\text{cc}}}=\set{F\middle|F\subseteq X,F=X\text{ or }F\text{ is countable}}$.
      \item \textbf{Included point topology}: If $X\neq\varnothing$, given $p\in X$, $\mathcal{F}\pr{X,\mathcal{T}_p^{\text{inc}}}=\mathcal{T}_p^{\text{exc}}$.
      \item \textbf{Excluded point topology}: If $X\neq\varnothing$, given $p\in X$, $\mathcal{F}\pr{X,\mathcal{T}_p^{\text{exc}}}=\mathcal{T}_p^{\text{inc}}$.
  \end{itemize}
\end{example}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. The following properties hold:

  \begin{enumerate}
      \item $\varnothing,X\in\mathcal{F}\pr{X,\mathcal{T}}$.
      \item If $F_1,F_2\in\mathcal{F}\pr{X,\mathcal{T}}$, then $F_1\cup F_2\in\mathcal{F}\pr{X,\mathcal{T}}$.
      \item If $\mathcal{C}\subseteq\mathcal{F}\pr{X,\mathcal{T}}$, then $\bigcap{\mathcal{C}}\in\mathcal{F}\pr{X,\mathcal{T}}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    Trivial by the definitions of topology and closed set, and De Morgan's laws.
  \end{proof}
\end{proposition}

\begin{remark}
  Observe that if $\pr{X,\mathcal{T}}$ is a topological space, then the finite union of closed sets is a closed set. This can be easily proved by induction.
\end{remark}

In general, the arbitrary union of closed sets in a topological space is not a closed set.

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. We say that a subset of $X$ is $F_{\sigma}$ if it is the union of a countable collection of closed sets.
\end{definition}


\begin{definition}
  Let $X$ be a set and $\mathcal{T},\mathcal{T}'\subseteq\mathcal{P}\pr{X}$ be two topologies on $X$. We say:

  \begin{itemize}
    \item \emph{$\mathcal{T}$ is finer than $\mathcal{T}'$} if $\mathcal{T}'\subseteq\mathcal{T}$. If the containment is strict, we say that \emph{$\mathcal{T}$ is strictly finer than $\mathcal{T}'$}.
    \item \emph{$\mathcal{T}$ is coarser than $\mathcal{T}'$} if $\mathcal{T}\subseteq\mathcal{T}'$. If the containment is strict, we say that \emph{$\mathcal{T}$ is strictly coarser than $\mathcal{T}'$}.
    \item \emph{$\mathcal{T}$ and $\mathcal{T}'$ are comparable} if $\mathcal{T}\subseteq\mathcal{T}'$ or $\mathcal{T'}\subseteq\mathcal{T}$. Otherwise, we say \emph{$\mathcal{T}$ and $\mathcal{T}'$ are incomparable}.
  \end{itemize}
\end{definition}

\begin{proposition}
  The arbitrary intersection of topologies on a set $X$ is a topology on that set. 
  \begin{proof}[\rm\textbf{Proof}]
    Let $\mathscr{T}\subseteq\mathcal{P}\pr{\mathcal{P}\pr{X}}$ be a collection of topologies on $X$ and $\mathcal{T}:=\bigcap{\mathscr{T}}\subseteq\mathcal{P}\pr{X}$. Since $\varnothing,X\in\mathcal{T}'$ for all $\mathcal{T}'\in\mathscr{T}$, then $\varnothing,X\in\mathcal{T}$. Let $G_1,G_2\in\mathcal{T}$ and $\mathcal{G}\subseteq\mathcal{T}$. Then $G_1,G_2\in\mathcal{T}'$ and $\mathcal{G}\subseteq\mathcal{T}'$ for all $\mathcal{T}'\in\mathscr{T}$. Thus $G_1\cap G_2,\bigcup{\mathcal{G}}\in\mathcal{T}'$ for all $\mathcal{T}'\in\mathscr{T}$. Consequently, $G_1\cap G_2,\bigcup{\mathcal{G}}\in\mathcal{T}$. Therefore, $\mathcal{T}$ is a topology on $X$.
  \end{proof}
\end{proposition}

\begin{definition}
  Let $X$ be a set and $\mathcal{A}$ a collection of subsets of $X$. The \emph{topology generated by $\mathcal{A}$} is defined as the smallest topology containing $\mathcal{A}$:

  \begin{equation}
    \mathcal{T}\pr{\mathcal{A}}=\bigcap{\set{\mathcal{T}\middle|\mathcal{T}\subseteq\mathcal{P}\pr{X},\mathcal{T}\text{ is a topology on $X$},\mathcal{A}\subseteq\mathcal{T}}}.
  \end{equation}
\end{definition}

\begin{remark}
  It is easy to check that if $X$ is a set and $\mathcal{A}$ a collection of subsets of $X$, then $\mathcal{A}\subseteq\mathcal{T}\pr{\mathcal{A}}$.
\end{remark}

\section{Bases}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. A \emph{basis for $\mathcal{T}$} is a collection $\mathcal{B}\subseteq\mathcal{T}$ of open sets in $\mathcal{T}$ such that for any $G\in\mathcal{T}$, there exists a basis set $\mathcal{B}'=\mathcal{B}'\pr{G}\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}'}$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $\mathcal{B}\subseteq\mathcal{T}$ be a basis for $\mathcal{T}$. Then $\mathcal{T}=\mathcal{T}\pr{\mathcal{B}}$.
  \begin{proof}[\rm\textbf{Proof}]
    Since $\mathcal{B}\subseteq\mathcal{T}$, then $\mathcal{T}\pr{\mathcal{B}}\subseteq\mathcal{T}$ by the definition of $\mathcal{T}\pr{\mathcal{B}}$. Given $G\in\mathcal{T}$, there exists $\mathcal{B}'\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}'}$. As $\mathcal{B}\subseteq\mathcal{T}\pr{\mathcal{B}}$, then $G\in\mathcal{T}\pr{\mathcal{B}}$. Thus $\mathcal{T}\subseteq\mathcal{T}\pr{\mathcal{B}}$. Therefore, $\mathcal{T}=\mathcal{T}\pr{\mathcal{B}}$.
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $X$ be a set and $\mathcal{B}\subseteq\mathcal{P}\pr{X}$ be a collection of subsets of $X$. The following are equivalent:

  \begin{enumerate}
    \item $\mathcal{B}$ is a basis for some topology on $X$.
    \item The following properties hold:
    \begin{itemize}
      \item[(i)] $X=\bigcup{\mathcal{B}}$.
      \item[(ii)] For any pair of non-disjoint basis sets $B_1,B_2\in\mathcal{B}$, if $x\in B_1\cap B_2$, there exists a basis set $B_3=B_3\pr{x}\in\mathcal{B}$ such that $x\in B_3\subseteq B_1\cap B_2$.
    \end{itemize}
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Let $\mathcal{T}$ be the topology on $X$ for which $\mathcal{B}$ is a basis. Since $X\in\mathcal{T}$, there exists $\mathcal{B}'\subseteq\mathcal{B}$ such that $X=\bigcup{\mathcal{B}'}\subseteq\bigcup{\mathcal{B}}\subseteq X$ and, consequently, $X=\bigcup{\mathcal{B}}$. Let $B_1,B_2\in\mathcal{B}$ be a pair of non-disjoint basis sets and $x\in B_1\cap B_2$. Since $\mathcal{B}\subseteq\mathcal{T}$, then $B_1\cap B_2\in\mathcal{T}$. Thus there exists $\mathcal{B}'\subseteq\mathcal{B}$ such that $x\in B_1\cap B_2=\bigcup{\mathcal{B}'}$. Hence, there exists a basis set $B_3\in\mathcal{B}'\subseteq\mathcal{B}$ such that $x\in B_3\subseteq B_1\cap B_2$.
      \item[\boxed{2\Rightarrow1}] Let $\mathcal{T}:=\set{\bigcup{\mathcal{B}'}\middle|\mathcal{B}'\subseteq\mathcal{B}}\subseteq\mathcal{P}\pr{X}$. Let's prove that $\mathcal{T}$ is a topology on $X$. Firstly, $\varnothing=\bigcup{\varnothing},X=\bigcup{\mathcal{B}}\in\mathcal{T}$. 
      
      Let $G_1,G_2\in\mathcal{T}$. If $G_1$ and $G_2$ are a pair of disjoint open sets, then there is nothing to prove. Suppose $G_1\cap G_2\neq\varnothing$. Let $\mathcal{B}',\mathcal{B}''\subseteq\mathcal{B}$ such that $G_1=\bigcup{\mathcal{B}'}$ and $G_2=\bigcup{\mathcal{B}''}$. Then $G_1\cap G_2=\bigcup_{B'\in\mathcal{B}',B''\in\mathcal{B}''}{\pr{B'\cap B''}}$. Given $x\in G_1\cap G_2$, there exists a basis set $B'_x\in\mathcal{B}'$ and $B''_x\in\mathcal{B}''$ such that $x\in B'_x\cap B''_x$. By hypothesis, there exists a basis set $B_x\in\mathcal{B}$ such that $x\in B_x\subseteq B'_x\cap B''_x$. Then $G_1\cap G_2=\bigcup_{x\in G_1\cap G_2}{\set{x}}\subseteq\bigcup_{x\in G_1\cap G_2}{B_x}\subseteq G_1\cap G_2$ and, consequently, $G_1\cap G_2=\bigcup_{x\in G_1\cap G_2}{B_x}$. Let $\widetilde{B}:=\set{B_x\middle|x\in G_1\cap G_2}\subseteq\mathcal{B}$. Then $G_1\cap G_2=\bigcup{\widetilde{B}}\in\mathcal{T}$. 

      Let $\mathcal{G}\subseteq\mathcal{T}$. For each $G\in\mathcal{G}$, there exists $\mathcal{B}_G\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}_G}$. Let $\mathcal{B}':=\bigcup\set{\mathcal{B}_G\middle|G\in\mathcal{G}}\subseteq\mathcal{B}$. Then $\bigcup{\mathcal{G}}=\bigcup{\mathcal{B}'}\in\mathcal{T}$.

      In conclusion, $\mathcal{B}$ is a basis for some topology on $X$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $\mathcal{B}\subseteq\mathcal{T}$ be a collection of open sets in $\mathcal{T}$. The following are equivalent:

  \begin{enumerate}
    \item $\mathcal{B}$ is a basis for $\mathcal{T}$.
    \item For any open set $G\in\mathcal{T}$, if $x\in G$, there exists a basis set $B=B\pr{x}\in\mathcal{B}$ such that $x\in B\subseteq G$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Let $G\in\mathcal{T}$ and $x\in G$. Since $\mathcal{B}$ is a basis for $\mathcal{T}$, there exists $\mathcal{B}'\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}'}$. As $x\in G=\bigcup{\mathcal{B}'}$, there exists $B\in\mathcal{B}'\subseteq\mathcal{B}$ such that $x\in B\subseteq G$.
      \item[\boxed{2\Rightarrow1}] Let $G\in\mathcal{T}$. Given $x\in G$, there exists a basis set $B_x\in\mathcal{B}$ such that $x\in B_x\subseteq G$. Then $G=\bigcup_{x\in G}{\set{x}}\subseteq\bigcup_{x\in G}{B_x}\subseteq G$ and, consequently, $G=\bigcup_{x\in G}{B_x}$. Thus $\mathcal{B}':=\set{B_x\middle|x\in G}\subseteq\mathcal{B}$ and $G=\bigcup{\mathcal{B}'}$. Therefore, $\mathcal{B}$ is a basis for $\mathcal{T}$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space, $\mathcal{B}\subseteq\mathcal{T}$ be a basis for $\mathcal{T}$ and $G\subseteq X$. Then $G\in\mathcal{T}$ if and only if, for each $x\in G$, there exists a basis set $B=B\pr{x}\in\mathcal{B}$ such that $x\in B\subseteq G$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Trivial by the previous proposition.
      \item[\boxed{\Leftarrow}] If $x\in G$, there exists a basis set $B_x\in\mathcal{B}$ such that $x\in B_x\subseteq G$. Then $G=\bigcup_{x\in G}{\set{x}}\subseteq\bigcup_{x\in G}{B_x}\subseteq G$ and, therefore, $G=\bigcup_{x\in G}{B_x}\in\mathcal{T}$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{example}
  Some examples of basis for a topology on a set are the following:

  \begin{itemize}
    \item \textbf{Discrete topology}: If $X$ is any set, then $\mathcal{B}:=\set{\set{x}\middle|x\in X}$ is a basis for the discrete topology. It is enough to notice that, for all $A\subseteq X$, we have $A=\bigcup_{x\in A}{\set{x}}$ and $\set{x}\in\mathcal{T}_{\text{dis}}$ for all $x\in X$.
    \item \textbf{Euclidean topology on $\mathbb{R}$}: We define the following collection of subsets of $\mathbb{R}$:

    \begin{equation}
        \mathcal{B}:=\set{\pr{a,b}\middle|a,b\in\mathbb{R},a<b}\subseteq\mathcal{P}\pr{\mathbb{R}}.
    \end{equation}

    Let $x\in\mathbb{R}$. We define $B_x:=\pr{x-1,x+1}\in\mathcal{B}$. Clearly $x\in B_x$, $\mathbb{R}=\bigcup_{x\in\mathbb{R}}{\set{x}}\subseteq\bigcup_{x\in\mathbb{R}}{B_x}\subseteq\bigcup{\mathcal{B}}\subseteq\mathbb{R}$ and, consequently, $\mathbb{R}=\bigcup{\mathcal{B}}$. Let $B_1,B_2\in\mathcal{B}$ be a pair of non-disjoint sets and, $a,b,a',b'\in\mathbb{R}$ with $a<b$ and $a'<b'$ such that $B_1=\pr{a,b}$ and $B_2=\pr{a',b'}$. Then $B_1\cap B_2=\pr{\max\set{a,a'},\min\set{b,b'}}\in\mathcal{B}$. Therefore, $\mathcal{B}$ is a basis for the topology on $\mathbb{R}$.
    
    We define the \emph{Euclidean topology on $\mathbb{R}$} as the following topology on $\mathbb{R}$:
    
    \begin{equation}
      \mathcal{T}_{e}\pr{\mathbb{R}}:=\mathcal{T}\pr{\set{\pr{a,b}\middle|a,b\in\mathbb{R},a<b}}.
    \end{equation}
  \end{itemize}
\end{example}

\begin{example}
  Let $\pr{X,\leq}$ be a totally ordered set. Given $a,b\in X$ with $a<b$, we define the following subsets of $X$:

  \begin{enumerate}
    \item \textbf{Open interval}: $\pr{a,b}:=\set{x\in X\middle|a<x<b}$.
    \item \textbf{Semiopen interval}: $\bpr{a,b}:=\set{x\in X\middle|a\leq x<b}, \pbr{a,b}:=\set{x\in X\middle|a<x\leq b}$.
    \item \textbf{Closed interval}: $\br{a,b}=\set{x\in X\middle|a\leq x\leq b}$.
  \end{enumerate}

  Using these sets, we can define the following topologies on $X$:

  \begin{itemize}
      \item \textbf{Order topology}: Let $\mathcal{B}_{\leq}\subseteq\mathcal{P}\pr{X}$ be a collection of subsets of $X$ such that they are one of the following types:

      \begin{itemize}
        \item[(i)] Open intervals in $X$.
        \item[(ii)] Semiopen intervals in $X$ of the form $\bpr{a_0,b}$ where $b\in X$ and $a_0$ is the minimum of $X$ if it exists.
        \item[(iii)] Semiopen intervals in $X$ of the form $\pbr{a,b_0}$ where $a\in X$ and $b_0$ is the maximum of $X$ if it exists.
      \end{itemize}

      Since the intersection of non-disjoint pair of sets of $\mathcal{B}_{\leq}$ is an element of $\mathcal{B}_{\leq}$, then $\mathcal{B}_{\leq}$ is a basis for a topology on $X$ called the \emph{order topology} and we will denote it by $\mathcal{T}_{\leq}$. We further denote $X_{\leq}$ the topological space $\pr{X,\mathcal{T}_{\leq}}$.

      This definition is based on the Euclidean topology on $\mathbb{R}$. In fact, $\pr{\mathbb{R},\mathcal{T}_e\pr{\mathbb{R}}}=\mathbb{R}_{\leq}$.

      \item \textbf{Lower bound topology and Sorgenfrey's line}: Let $\mathcal{B}_{\ell}\subseteq\mathcal{P}\pr{X}$ be a collection of subsets of $X$ such that they are one of the following types:

      \begin{itemize}
        \item[(i)] Semiopen intervals in $X$ of the form $\bpr{a,b}$ where $a,b\in X$ and $a<b$.
        \item[(ii)] The closed interval in $X$ of the form $\br{a,b_0}$ where $b_0$ is the maximum of $X$ if it exists.
      \end{itemize}

      Since the intersection of non-disjoint pair of sets of $\mathcal{B}_{\ell}$ is an element of $\mathcal{B}_{\ell}$, then $\mathcal{B}_{\ell}$ is a basis for a topology on $X$ called the \emph{lower bound topology} and is denoted by $\mathcal{T}_{\ell}$. The topological space $\pr{X,\mathcal{T}_{\ell}}$ is denoted by $X_{\ell}$. If $X=\mathbb{R}$, the resulting space is called \emph{Sorgenfrey's line}.
      
      \item \textbf{Upper bound topology}: Let $\mathcal{B}_{u}\subseteq\mathcal{P}\pr{X}$ be a collection of subsets of $X$ such that they are one of the following types:

      \begin{itemize}
        \item[(i)] Semiopen intervals in $X$ of the form $\pbr{a,b}$ where $a,b\in X$ and $a<b$.
        \item[(ii)] The closed interval in $X$ of the form $\br{a_0,b}$ where $a_0$ is the minimum of $X$ if it exists.
      \end{itemize}

      Since the intersection of non-disjoint pair of sets of $\mathcal{B}_u$ is an element of $\mathcal{B}_u$, then $\mathcal{B}_u$ is a basis for a topology on $X$ called the \emph{upper bound topology} and is denoted by $\mathcal{T}_u$. The topological space $\pr{X,\mathcal{T}_u}$ is denoted by $X_u$.
  \end{itemize}
\end{example}

\begin{proposition}
  Let $X$ be a set and $\mathcal{T},\mathcal{T}'\subseteq\mathcal{P}\pr{X}$ be two topologies on $X$ with bases $\mathcal{B}$ and $\mathcal{B}'$ respectively. Then $\mathcal{T}$ is finer than $\mathcal{T}'$ if and only if for all $x\in X$ and $B'\in\mathcal{B}'$ with $x\in B'$, there exists a basis set $B\in\mathcal{B}$ such that $x\in B\subseteq B'$.

  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
        \item[\boxed{\Rightarrow}] Let $x\in X$ and $B'\in\mathcal{B}'$ with $x\in B'$. Since $\mathcal{T}'\subseteq\mathcal{T}$, then $B'\in\mathcal{B}$. It suffices to take $B:=B'$.
        \item[\boxed{\Leftarrow}] Let $G\in\mathcal{T}'$. Then there exists $\mathcal{B}''\subseteq\mathcal{B'}$ such that $G=\bigcup{\mathcal{B}''}$. Given $x\in G$, there exists a basis set $B''\in\mathcal{B}''$ such that $x\in B''$. By hypothesis, there exists a basis set $B_x\in\mathcal{B}$ such that $x\in B_x\subseteq B''$. Then $G=\bigcup_{x\in G}{\set{x}}\subseteq\bigcup_{x\in G}{B_x}\subseteq\bigcup_{B''\in\mathcal{B}''}{B''}=G$, that is to say, $G=\bigcup_{x\in G}{B_x}\in\mathcal{T}$. Thus $\mathcal{T}'\subseteq\mathcal{T}$. Therefore, $\mathcal{T}$ is finer than $\mathcal{T}'$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{corollary}
  Let $X$ be a set and $\mathcal{T},\mathcal{T}'\subseteq\mathcal{P}\pr{X}$ be two topologies on $X$ with bases $\mathcal{B}$ and $\mathcal{B}'$ respectively. Then $\mathcal{T}=\mathcal{T}'$ if and only if the following properties hold:

  \begin{enumerate}
      \item For all $x\in X$ and $B'\in\mathcal{B}'$ with $x\in B'$, there exists a basis set $B\in\mathcal{B}$ such that $x\in B\subseteq B'$.
      \item For all $x\in X$ and $B\in\mathcal{B}$ with $x\in B$, there exists a basis set $B'\in\mathcal{B}'$ such that $x\in B'\subseteq B$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    It is a direct consequence of the previous proposition.
  \end{proof}
\end{corollary}

\begin{definition}
  Let $X$ be a set. A \emph{subbasis for a topology on $X$} is a collection $\mathcal{S}\subseteq\mathcal{P}\pr{X}$ that covers $X$. If $\mathcal{T}\subseteq\mathcal{P}\pr{X}$ is a topology on $X$ and $\mathcal{S}\subseteq\mathcal{P}\pr{X}$ is a subbasis for a topology on $X$ such that $\mathcal{T}=\mathcal{T}\pr{{\mathcal{S}}}$, we say that $\mathcal{S}$ is a \emph{subbasis for $\mathcal{T}$}.
\end{definition}

\begin{proposition}
  Let $X$ be a set and $\mathcal{S}$ be a subbasis for a topology on $X$. We define $\mathcal{B}_{\mathcal{S}}\subseteq\mathcal{P}\pr{X}$ as the following collection of subsets of $X$:

  \begin{equation}
    \mathcal{B}_{\mathcal{S}}:=\set{\bigcap_{k=1}^{n}{S_k}\middle|n\in\mathbb{N},S_1,\dots,S_n\in\mathcal{S}}.
  \end{equation}

  Then $\mathcal{B}_{\mathcal{S}}$ is a basis for the topology $\mathcal{T}\pr{\mathcal{S}}$, that is to say, any open set in $\mathcal{T}\pr{\mathcal{S}}$ is arbitrary union of finite intersections of sets in $\mathcal{S}$.
  \begin{proof}[\rm\textbf{Proof}]
    It is immediate to check that $\mathcal{S}\subseteq\mathcal{B}_{\mathcal{S}}$ and, consequently, $\mathcal{T}\pr{\mathcal{S}}\subseteq\mathcal{T}\pr{\mathcal{B}_{\mathcal{S}}}$. Since $\mathcal{S}\subseteq\mathcal{T}\pr{\mathcal{S}}$, we have that $\mathcal{B}_{\mathcal{S}}\subseteq\mathcal{T}\pr{\mathcal{S}}$ and, consequently, $\mathcal{T}\pr{\mathcal{B}_{\mathcal{S}}}\subseteq\mathcal{T}\pr{\mathcal{S}}$. Thus $\mathcal{T}\pr{\mathcal{S}}=\mathcal{T}\pr{\mathcal{B}_{\mathcal{S}}}$.

    Let's prove that $\mathcal{B}_{\mathcal{S}}$ is a basis for a topology. Observe that $X=\bigcup{\mathcal{S}}$. Let $B_1,B_2\in\mathcal{B}_{\mathcal{S}}$ be a pair of non-disjoint basis sets. Let $S_1,\dots,S_n,S'_1,\dots,S'_m\in\mathcal{S}$ with $n,m\in\mathbb{N}$ such that $B_1=\bigcap_{k=1}^{n}{S_k}$ and $B_2=\bigcap_{l=1}^{m}{S'_l}$. Then $B_1\cap B_2=\bigcap_{k=1}^{n}{\bigcap_{l=1}^{m}{\pr{S_k\cap S'_l}}}\in\mathcal{B}_{\mathcal{S}}$. Thus, given $x\in B_1\cap B_2$, there exists a basis set $B_3:=B_1\cap B_2\in\mathcal{B}_{\mathcal{S}}$ such that $x\in B_3\subseteq B_1\cap B_2$. Therefore, $\mathcal{B}_{\mathcal{S}}$ is a basis for a topology.

    Using that $\mathcal{T}\pr{\mathcal{S}}=\mathcal{T}\pr{\mathcal{B}_{\mathcal{S}}}$, we deduce that $\mathcal{B}_{\mathcal{S}}$ is a basis for $\mathcal{T}\pr{\mathcal{S}}$.
  \end{proof}
\end{proposition}

\begin{example}
  Let $\pr{X,\leq}$ be a totally ordered set. Given $a,b\in X$, we define the following sets:

  \begin{itemize}
    \item \textbf{Open rays}: $\pr{a,+\infty}:=\set{x\middle|x>a}$ and $\pr{-\infty,b}:=\set{x\middle|x<b}$.
    \item \textbf{Closed rays}: $\bpr{a,+\infty}:=\set{x\middle|x\geq a}$ and $\pbr{-\infty,b}:=\set{x\middle|x\leq b}$.
  \end{itemize}

  Here, the symbols $-\infty$ and $+\infty$ do not denote any element of $X$. For example, in $\pr{\br{0,1},\leq}$, $\pr{-\infty,1}=\bpr{0,1}$. Observe the following:

  \begin{itemize}
    \item If there exists $\min\pr{X}:=a_0$, then $\pr{-\infty,b}=\bpr{a_0,b}$ and $\pbr{-\infty,b}=\br{a_0,b}$.
    \item If there exists $\max\pr{X}:=b_0$, then $\pr{a,+\infty}=\pbr{a,b_0}$ and $\bpr{a,+\infty}=\br{a,b_0}$.
  \end{itemize}
  
  Consider the following collection of sets in $\mathcal{P}\pr{X}$:
  
  \begin{equation}
    \begin{split}
      \mathcal{S}_{\leq} & := \set{\pr{a,+\infty}\middle|a\in X}\cup\set{\pr{-\infty,b}\middle|b\in X}\subseteq\mathcal{P}\pr{X},\\
      \mathcal{S}_{\ell} & := \set{\bpr{a,+\infty}\middle|a\in X}\cup\set{\pr{-\infty,b}\middle|b\in X}\subseteq\mathcal{P}\pr{X},\\
      \mathcal{S}_{u} & := \set{\pr{a,+\infty}\middle|a\in X}\cup\set{\pbr{-\infty,b}\middle|b\in X}\subseteq\mathcal{P}\pr{X}.
    \end{split}
  \end{equation}
  
  Then:
  
  \begin{itemize}
    \item $\mathcal{S}_{\leq}$ is a subbasis for the order topology.
    \item $\mathcal{S}_{\ell}$ is a subbasis for the lower bound topology.
    \item $\mathcal{S}_{u}$ is a subbasis for the upper bound topology.
  \end{itemize}
\end{example}

\section{Topological subspaces}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $Y\subseteq X$. Then the collection $\mathcal{T}_Y:=\set{G\cap Y\middle|G\in\mathcal{T}}\subseteq\mathcal{P}\pr{Y}$ of subsets in $Y$ is a topology on $Y$.
  \begin{proof}[\rm\textbf{Proof}]
    It is immediate that $\varnothing=\varnothing\cap Y,Y=X\cap Y\in\mathcal{T}_Y$. 
    
    Let $G_1',G'_2\in\mathcal{T}_Y$. Then there exists two open sets $G_1,G_2\in\mathcal{T}$ such that $G'_1=G_1\cap Y$ and $G'_2=G_2\cap Y$. Thus $G_1\cap G_2\in\mathcal{T}$ and $G'_1\cap G'_2=\pr{G_1\cap G_2}\cap Y\in\mathcal{T}_Y$. 
    
    Let $\mathcal{G}'\subseteq\mathcal{T}_Y$. Given $G'\in\mathcal{G}'$, there exists an open set $G\in\mathcal{T}$ such that $G'=G\cap Y$. Let $\mathcal{G}=\set{G\middle|G\cap Y\in\mathcal{G}'}$. Then $\bigcup{\mathcal{G}}\in\mathcal{T}$ and $\bigcup{\mathcal{G}'}=\bigcup_{G\in\mathcal{G}}{\pr{G\cap Y}}=\pr{\bigcup{\mathcal{G}}}\cap Y\in\mathcal{T}_Y$. Therefore, $\mathcal{T}_Y$ is a topology on $Y$.
  \end{proof}
\end{proposition}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $Y\subseteq X$. We define the \emph{topological subspace induced by $Y$ (with respect to $\pr{X,\mathcal{T}}$)} as the topological space $\pr{Y,\mathcal{T}_Y}$ with $\mathcal{T}_Y:=\set{G\cap Y\middle|G\in\mathcal{T}}\subseteq\mathcal{P}\pr{Y}$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $Y\subseteq X$. If $\mathcal{B}$ is a basis for $\mathcal{T}$, then $\mathcal{B}_Y:=\set{B\cap Y\middle|B\in\mathcal{B}}$ is a basis for $\mathcal{T}_Y$.
  \begin{proof}[\rm\textbf{Proof}]
    Let $G'\in\mathcal{T}_Y$. Then there exists an open set $G\in\mathcal{T}$ such that $G'=G\cap Y$. Thus, there exists $\mathcal{B}'\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}'}$. Let $\mathcal{B}'_Y:=\set{B\cap Y\middle|B\in\mathcal{B}'}\subseteq\mathcal{B}_Y$. Then $G'=G\cap Y=\pr{\bigcup{\mathcal{B}'}}\cap Y=\bigcup{\mathcal{B}'_Y}$. Therefore, $\mathcal{B}_Y$ is a basis for $\mathcal{T}_Y$.
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $Y\subseteq X$. If $Y\in\mathcal{T}$, then $\mathcal{T}_Y\subseteq\mathcal{T}$.
  \begin{proof}[\rm\textbf{Proof}]
  It is trivial, because the finite intersection of open sets in $\mathcal{T}$ is an open set in $\mathcal{T}$.
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space, $Y\subseteq X$ and $F'\subseteq Y$. Then $F'\in\mathcal{F}\pr{Y,\mathcal{T}_Y}$ is a closed set if and only if there exists a closed set $F\in\mathcal{F}\pr{X,\mathcal{T}}$ such that $F'=F\cap Y$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
        \item[\boxed{\Rightarrow}] Since $F'\in\mathcal{F}\pr{Y,\mathcal{T}_Y}$, then $Y\setminus F'\in\mathcal{T}_Y$. Let $G\in\mathcal{T}$ be the open set such that $Y\setminus F'=G\cap Y$. Then $F'=Y\setminus\pr{G\cap Y}=Y\setminus G=\pr{X\setminus G}\cap Y$. It suffices to notice that $F:=X\setminus G\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed set.
        \item[\boxed{\Leftarrow}] Observe that $Y\setminus F'=Y\setminus F=\pr{X\setminus F}\cap Y$. Since $F\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed set, we have $X\setminus F\in\mathcal{T}$. Then $Y\setminus F'\in\mathcal{T}_Y$ is an open set. Therefore, $F'\in\mathcal{F}\pr{Y,\mathcal{T}_Y}$ is a closed set.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space, $Y\subseteq X$ and $F'\subseteq Y$. If $Y\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed set and $F'\in\mathcal{F}\pr{Y,\mathcal{T}_Y}$ is a closed set, then $F'\in\mathcal{F}\pr{X,\mathcal{T}}$.
  \begin{proof}[\rm\textbf{Proof}]
    Trivial because the arbitrary intersection of closed sets with respect to $\pr{X,\mathcal{T}}$ is a closed set with respect to that topological space.
  \end{proof}
\end{proposition}

\section{Finite product topology}

\begin{proposition}
  Let $d\in\mathbb{N}$ and $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$ be $d$ topological spaces. Then $\mathcal{B}=\set{G_1\times\cdots\times G_d\middle|G_1\in\mathcal{T}_1,\dots,G_d\in\mathcal{T}_d}$ is a basis for a topology on $X_1\times\cdots\times X_d$.
  \begin{proof}[\rm\textbf{Proof}]
    Firstly, since $X_1\times\cdots\times X_d\in\mathcal{B}$, then $X_1\times\cdots\times X_d=\bigcup{\mathcal{B}}\subseteq X_1\times\cdots\times X_d$ and, consequently, $X_1\times\cdots\times X_d=\bigcup{\mathcal{B}}$. 
    
    Let $B_1,B_2\in\mathcal{B}$ be a pair of non-disjoint basis sets and $x:=\pr{x_1,\dots,x_d}\in B_1\cap B_2$. Let $G_{1,1},G_{1,2}\in\mathcal{T}_1,\dots,G_{d,1},G_{d,2}\in\mathcal{T}_d$ be open sets such that $B_1=G_{1,1}\times\cdots\times G_{d,1}$ and $B_2=G_{1,2}\times\cdots\times G_{d,2}$. Then $B_1\cap B_2=\pr{G_{1,1}\cap G_{1,2}}\times\cdots\times\pr{G_{d,1}\cap G_{d,2}}$. Since $G_{1,1}\cap G_{1,2}\in\mathcal{T}_1,\dots,G_{d,1}\cap G_{d,2}\in\mathcal{T}_d$, then $B_3:=B_1\cap B_2\in\mathcal{B}$ and, consequently, $x\in B_3\subseteq B_1\times B_2$. 
    
    Therefore, $\mathcal{B}$ is a basis for a topology on $X_1\times\cdots\times X_d$.
  \end{proof}
\end{proposition}

\begin{definition}
  Let $d\in\mathbb{N}$ and $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$ be $d$ topological spaces. The \emph{product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$} is defined as the topology $\mathcal{T}_{X_1\times\cdots\times X_d}\subseteq\mathcal{P}\pr{X_1\times\cdots\times X_d}$ on $X_1\times\cdots\times X_d$ with basis:

  \begin{equation}
    \mathcal{B}=\set{G_1\times\cdots\times G_d\middle|G_1\in\mathcal{T}_1,\dots,G_d\in\mathcal{T}_d}.
  \end{equation}
\end{definition}

\begin{proposition}
  Let $d\in\mathbb{N}$ and $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$ be $d$ topological spaces with bases $\mathcal{B}_1,\dots,\mathcal{B}_d$ respectively. Then $\mathcal{B}:=\mathcal{B}_1\times\cdots\times\mathcal{B}_d$ is a basis for the product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$.
  \begin{proof}[\rm\textbf{Proof}]
    Let $\widetilde{\mathcal{B}}:=\set{G_1\times\cdots\times G_d\middle|G_1\in\mathcal{T}_1,\dots,G_d\in\mathcal{T}_d}$ the standard basis for the product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$. Given $G\in\mathcal{T}_{X_1\times\cdots\times X_d}$, there exists $\mathcal{G}\subseteq\widetilde{\mathcal{B}}$ such that $G=\bigcup{\mathcal{G}}$. Given $G'\in\mathcal{G}$, there exists $d$ open sets $G'_1\in\mathcal{T}_1,\dots,G'_d\in\mathcal{T}_d$ such that $G'=G'_1\times\cdots\times G'_d$ and, consequently, there exists $\mathcal{B}_{G',1}\subseteq\mathcal{B}_1,\dots,\mathcal{B}_{G',d}\subseteq\mathcal{B}_d$ such that $G'_1=\bigcup{\mathcal{B}_{G',1}},\dots,G'_d=\bigcup{\mathcal{B}_{G',d}}$. Thus:

    \begin{equation}
      G=\bigcup_{G'\in\mathcal{G}}{\pr{\pr{\bigcup{\mathcal{B}_{G',1}}}\times\cdots\times\pr{\bigcup{\mathcal{B}_{G',d}}}}}=\bigcup_{G'\in\mathcal{G}}{\bigcup_{B'_1\in\mathcal{B}_{G',1}}{\cdots\bigcup_{B'_d\in\mathcal{B}_{G',d}}{B'_1\times\cdots\times B'_d}}}.
    \end{equation}

    Let $\mathcal{B}':=\set{B'_1\times\cdots\times B'_d\middle|G'\in\mathcal{G},B'_1\in\mathcal{B}'_{G',1},\dots,B'_d\in\mathcal{B}'_{G',d}}\subseteq\mathcal{B}$. Then $G=\bigcup{\mathcal{B}'}$. Therefore, $\mathcal{B}$ is a basis for the product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$.
  \end{proof}
\end{proposition}

\begin{example}
  Given $d\in\mathbb{N}$, we define the \emph{Euclidean topology on $\mathbb{R}^d$} as the following topology on $\mathbb{R}^d$:

  \begin{equation}
      \mathcal{T}_e\pr{\mathbb{R}^d}:=\mathcal{T}\pr{\set{\prod_{k=1}^{d}{\pr{a_k,b_k}}\middle|\forall k\in\set{1,\dots,d},a_k,b_k\in\mathbb{R},a_k<b_k}}.
  \end{equation}

  Observe that $\mathcal{T}_e\pr{\mathbb{R}^d}$ is the product topology with respect to $\pr{\mathbb{R},\mathcal{T}_e\pr{\mathbb{R}}}$ $d$ times.
\end{example}


Given $d\in\mathbb{N}$, $d$ sets $X_1,\dots,X_d$ and $k\in\set{1,\dots,d}$, we define the map $\pi_k:X_1\times\cdots\times X_d\rightarrow X_k$ called \emph{projection on the $k$-th component} as $\pi_k\pr{x_1,\dots,x_{k-1},x_k,x_{k+1},\dots,x_d}=x_k$.

\begin{remark}
  Given $d\in\mathbb{N}$, $d$ topological spaces $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$, $k\in\set{1,\dots,d}$ and $G\in\mathcal{T}_k$, observe that $\pi_k^{-1}\pr{G}=X_1\times\cdots\times X_{k-1}\times G\times X_{k+1}\times\cdots\times X_d$.
\end{remark}

\begin{proposition}\label{Finite product topology subbasis}
  Let $d\in\mathbb{N}$ and $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$ be $d$ topological spaces. The following collection of subsets of $X_1\times\cdots\times X_d$ is a subbasis for the product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$:

  \begin{equation}
      \mathcal{S}=\set{\pi^{-1}_k\pr{G}\middle|k\in\set{1,\dots,d},G\in\mathcal{T}_k}\subseteq\mathcal{P}\pr{X_1\times\cdots\times X_d}.
  \end{equation}
  \begin{proof}[\rm\textbf{Proof}]
    Clearly, $\mathcal{S}$ is a subbasis for a topology, because $X_1\times\cdots\times X_d\in\mathcal{S}$. Consider the collection $\mathcal{B}:=\set{G_1\times\cdots\times G_d\middle|G_1\in\mathcal{T}_1,\dots,G_d\in\mathcal{T}_d}$ to be the standard basis for the product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$. Since $\mathcal{S}\subseteq\mathcal{B}\subseteq\mathcal{T}\pr{\mathcal{B}}$, then $\mathcal{T}\pr{\mathcal{S}}\subseteq\mathcal{T}\pr{\mathcal{B}}$. For each $k\in\set{1,\dots,d}$, let $G_k\in\mathcal{T}_k$. Then $\pi_1^{-1}\pr{G_1}\cap\dots\cap\pi_d^{-1}\pr{G_d}=G_1\times\cdots\times G_d$. Thus $\mathcal{B}\subseteq\mathcal{B}_{\mathcal{S}}:=\set{\bigcap_{l=1}^{n}{S_l}\middle|n\in\mathbb{N},S_1,\dots,S_n\in\mathcal{S}}$. Hence, $\mathcal{T}\pr{\mathcal{B}}\subseteq\mathcal{T}\pr{\mathcal{B}_\mathcal{S}}=\mathcal{T}\pr{\mathcal{S}}$. Consequently, $\mathcal{T}\pr{\mathcal{B}}=\mathcal{T}\pr{\mathcal{S}}$. Therefore, $\mathcal{S}$ is a subbasis for the product topology with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$.
  \end{proof}
\end{proposition}

\section{Neighborhoods}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $x\in X$. A neighborhood of $x$ with respect to $\pr{X,\mathcal{T}}$ is a set $N\subseteq X$ for which there exists an open set $G=G\pr{x,N}\in\mathcal{T}$ such that $x\in G\subseteq N$. The set of all neighborhoods of $x$ with respect to $\pr{X,\mathcal{T}}$ is denoted by $\mathcal{N}_x\pr{X,\mathcal{T}}$.
\end{definition}

\begin{remark}
  Given a topological space $\pr{X,\mathcal{T}}$ and an open set $G\in\mathcal{T}$, we have that $G\in\mathcal{N}_x\pr{X,\mathcal{T}}$ for all $x\in G$.
\end{remark}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space with basis $\mathcal{B}\subseteq\mathcal{T}$ for $\mathcal{T}$, $x\in X$ and $N\subseteq X$ with $x\in N$. Then $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ if and only if there exists a basis set $B=B\pr{x,N}\in\mathcal{B}$ such that $x\in B\subseteq N$.
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] Let $G\in\mathcal{T}$ such that $x\in G\subseteq N$. Then there exists $\mathcal{B}'\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}'}$. Thus there exists a basis set $B\in\mathcal{B}'\subseteq\mathcal{B}$ such that $x\in B\subseteq G\subseteq N$.
      \item[\boxed{\Leftarrow}] Trivial.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space, $x\in X$ and $Y\subseteq X$. The following properties hold:

  \begin{enumerate}
    \item If $N,M\in\mathcal{N}_x\pr{X,\mathcal{T}}$, then $N\cap M\in\mathcal{N}_x\pr{X,\mathcal{T}}$.
    \item If $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ and $N\subseteq M$, then $M\in\mathcal{N}_x\pr{X,\mathcal{T}}$.
    \item If $N,M\in\mathcal{N}_x\pr{X,\mathcal{T}}$, then $N\cup M\in\mathcal{N}_x\pr{X,\mathcal{T}}$.
    \item For every $N'\subseteq X$ and $x\in Y$, $N'\in\mathcal{N}_x\pr{Y,\mathcal{T}_Y}$ is a neighborhood of $x$ with respect to the topological subspace $\pr{Y,\mathcal{T}_Y}$ if and only if there exists a neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$ with respect to $\pr{X,\mathcal{T}}$ such that $N'=N\cap Y$.
  \end{enumerate}

  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $G_1,G_2\in\mathcal{T}$ be two open sets such that $x\in G_1\subseteq N$ and $x\in G_2\subseteq M$. Since $G_1\cap G_2\in\mathcal{T}$, we have that $x\in G_1\cap G_2\subseteq N\cap M$. Therefore, $N\cap M\in\mathcal{N}_x\pr{X,\mathcal{T}}$.
      \item Let $G\in\mathcal{T}$ be an open set such that $x\in G\subseteq N$. Since $N\subseteq M$, then $x\in G\subseteq M$. Therefore $M\in\mathcal{N}_x\pr{X,\mathcal{T}}$.
      \item Trivial by the previous property.
      \item \begin{itemize}
        \item[\boxed{\Rightarrow}] Since $N'\in\mathcal{N}_x\pr{Y,\mathcal{T}_Y}$ is a neighborhood of $x$ with respect to the topological subspace $\pr{Y,\mathcal{T}_Y}$, there exists an open set $G'\in\mathcal{T}_Y$ such that $G'\subseteq N'$. Let $G\in\mathcal{T}$ be an open set such that $G'=G\cap Y$. Let $N:=N'\cup G$. Since $G\in\mathcal{T}$ is an open set containing $x$ such that $G\subseteq N$, then $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ is a neighborhood of $x$ with respect to $\pr{X,\mathcal{T}}$. Furthermore, $N\cap Y=\pr{N'\cup G}\cap Y=N'\cup\pr{G\cap Y}=N'$.
        \item[\boxed{\Leftarrow}] Let $G\in\mathcal{T}$ be an open set containing $x$ such that $G\subseteq N$. Then $G\cap Y\subseteq N\cap Y=N'$ and $x\in G':=G\cap Y\in\mathcal{T}_Y$. Therefore, $N'\in\mathcal{N}_x\pr{Y,\mathcal{T}_Y}$ is a neighborhood of $x$ with respect to $\pr{Y,\mathcal{T}_Y}$.
      \end{itemize}
    \end{enumerate}
  \end{proof}
\end{proposition}

\section{Position of a point with respect to a set}

\subsection{Interior and exterior of a set}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. A point $x\in A$ is said to be an \emph{interior point of $A$ (with respect to $\pr{X,\mathcal{T}}$)} if $A\in\mathcal{N}_x\pr{X,\mathcal{T}}$. We define the \emph{interior of $A$ (with respect to $\pr{X,\mathcal{T}}$)} as the following set:

  \begin{equation}
    \interior\pr{A}:=\set{x\middle|x\in A,x\text{ is an interior point of $A$}}.
  \end{equation}

  Another commonly used notation is $A^\circ$.
\end{definition}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. An interior point of $X\setminus A$ is called an \emph{exterior point of $A$ (with respect to $\pr{X,\mathcal{T}}$)}. We define the \emph{exterior of $A$ (with respect to $\pr{X,\mathcal{T}}$)} as the following set:

  \begin{equation}
    \ext\pr{A}:=\set{x\middle|x\in X\setminus A,x\text{ is an exterior point of $A$}}=\interior\pr{X\setminus A}.
  \end{equation}
\end{definition}

\begin{remark}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. Observe that $\interior\pr{A}\subseteq A$ and $\ext\pr{A}\subseteq X\setminus A$. Furthermore, $\interior\pr{A}\cap\ext\pr{A}=\varnothing$.
\end{remark}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space with basis $\mathcal{B}\subseteq\mathcal{T}$ for $\mathcal{T}$, $A\subseteq X$ and $x\in A$. The following are equivalent:

  \begin{enumerate}
    \item $x\in\interior\pr{A}$.
    \item There exists an open set $G=G\pr{x}\in\mathcal{T}$ such that $x\in G\subseteq A$.
    \item There exists a basis set $B=B\pr{x}\in\mathcal{B}$ such that $x\in B\subseteq A$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Trivial by definition of neighborhood.
      \item[\boxed{2\Rightarrow3}] Let $\mathcal{B}'\subseteq\mathcal{B}$ such that $G=\bigcup{\mathcal{B}'}$. Since $x\in G$, there exists a basis set $B\in\mathcal{B}'\subseteq\mathcal{B}$ such that $x\in B\subseteq G\subseteq A$.
      \item[\boxed{3\Rightarrow1}] Trivial.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. Then:
  
  \begin{equation}
    \interior\pr{A}=\bigcup{\set{G\middle|G\in\mathcal{T},G\subseteq A}}. 
  \end{equation}
  
  Consequently, $\interior\pr{A}\in\mathcal{T}$ is an open set.
  \begin{proof}[\rm\textbf{Proof}]
    Let $x\in\interior\pr{A}$. Then there exists an open set $G\in\mathcal{T}$ such that $x\in G\subseteq A$. Thus $x\in\bigcup{\set{G\middle|G\in\mathcal{T},G\subseteq A}}$. Hence $\interior\pr{A}\subseteq\bigcup{\set{G\middle|G\in\mathcal{T},G\subseteq A}}$. 
    
    If $x\in\bigcup{\set{G\middle|G\in\mathcal{T},G\subseteq A}}$, there exists an open set $G\in\mathcal{T}$ such that $x\in G\subseteq A$. Then $x\in\interior\pr{A}$. Consequently, $\bigcup{\set{G\middle|G\in\mathcal{T},G\subseteq A}}\subseteq\interior\pr{A}$. 
    
    Therefore, $\interior\pr{A}=\bigcup{\set{G\middle|G\in\mathcal{T},G\subseteq A}}$.
  \end{proof}
\end{proposition}

\begin{remark}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. By the previous proposition, $\interior\pr{A}$ is the largest open set in $\mathcal{T}$ contained in $A$.
\end{remark}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space, $A,B,Y\subseteq X$. The following properties hold:

  \begin{enumerate}
    \item $A\in\mathcal{T}$ if and only if $\interior\pr{A}=A$.
    \item $\interior\pr{\interior\pr{A}}=\interior\pr{A}$.
    \item If $A\subseteq B$, then $\interior\pr{A}\subseteq\interior\pr{B}$.
    \item $\interior\pr{A}\cap\interior\pr{B}=\interior\pr{A\cap B}$.
    \item $\interior\pr{A}\cup\interior\pr{B}\subseteq\interior\pr{A\cup B}$.
    \item If $A\subseteq Y$, then $\interior\pr{A}\cap Y\subseteq\interior_Y\pr{A}$, where $\interior_Y\pr{A}$ denotes the interior of $A$ with respect to the topological subspace $\pr{Y,\mathcal{T}_Y}$. If $Y\in\mathcal{T}$, the equality holds.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Trivial by the previous proposition and the fact that an open set is neighborhood of all of its points.
      \item Trivial, because $\interior{A}\in\mathcal{T}$.
      \item Let $x\in\interior\pr{A}$. Then there exists an open set $G\in\mathcal{T}$ such that $x\in G\subseteq A$. Since $A\subseteq B$, we have that $x\in G\subseteq B$. Thus $x\in\interior\pr{B}$. Therefore, $\interior\pr{A}\subseteq\interior\pr{B}$.
      \item Let $x\in\interior\pr{A}\cap\interior\pr{B}$. Then there exists two open sets $G_1,G_2\in\mathcal{T}$ such that $x\in G_1\subseteq A$ and $x\in G_2\subseteq B$. Thus $x\in G_1\cap G_2\subseteq A\cap B$. Hence $x\in\interior\pr{A\cap B}$. Consequently, $\interior\pr{A}\cap\interior\pr{B}\subseteq\interior\pr{A\cap B}$. 
      
      Let $x\in\interior\pr{A\cap B}$. Then there exists an open set $G\in\mathcal{T}$ such that $x\in G\subseteq A\cap B\subseteq A,B$. Thus $x\in\interior\pr{A}$ and $x\in\interior\pr{B}$, or, in other words, $x\in\interior\pr{A}\cap\interior\pr{B}$. Hence, $\interior\pr{A\cap B}\subseteq\interior\pr{A}\cap\interior\pr{B}$. 
      
      Therefore, $\interior\pr{A}\cap\interior\pr{B}=\interior\pr{A\cap B}$.
      \item Let $x\in \interior\pr{A}\cup\interior\pr{B}$. Then there exists an open set $G\in\mathcal{T}$ such that $x\in G\subseteq A$ or $x\in G\subseteq B$. Particularly, $x\in G\subseteq A\cup B$. Then $x\in\interior\pr{A\cup B}$. Therefore, $\interior\pr{A}\cup\interior\pr{B}\subseteq\interior\pr{A\cup B}$.
      \item Let $x\in\interior\pr{A}\cap Y$. Then there exists an open set $G\in\mathcal{T}$ such that $x\in G\subseteq A$ and $x\in Y$. Then $x\in G\cap Y\subseteq A\cap Y=A$. Since $G\cap Y\in\mathcal{T}_Y$, we have $x\in\interior_Y\pr{A}$. Therefore, $\interior\pr{A}\cap Y\subseteq\interior_Y\pr{A}$.
      
      Suppose $Y\in\mathcal{T}$. Then $\mathcal{T}_Y\subseteq\mathcal{T}$ and, consequently, $\interior_Y\pr{A}\cap Y\in\mathcal{T}$ and $\interior_Y\pr{A}\cap Y\subseteq A$. Since $\interior\pr{A}=\bigcup\set{G\middle|G\in\mathcal{T},G\subseteq A}$, we have $\interior_Y\pr{A}\cap Y\subseteq\interior\pr{A}$. 
      
      Therefore, if $Y\in\mathcal{T}$, then $\interior\pr{A}\cap Y=\interior_Y\pr{A}$.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{example}
  Let $\pr{X,\mathcal{T}}$ be a topological space. Some examples of the interior of sets are the following:

  \begin{itemize}
    \item \textbf{Discrete topology}: For every $A\subseteq X$, $\interior\pr{A}=A$ with respect to $\pr{X,\mathcal{T}_{\text{dis}}}$ because $A\in\mathcal{T}_{\text{dis}}$.
    \item \textbf{Cofinite topology}: Let $a,b\in\mathbb{R}$ with $a<b$. Then $\interior\pr{\pr{a,b}}=\varnothing$ with respect to $\pr{\mathbb{R},\mathcal{T}_{\text{cof}}}$, because if $G\in\mathcal{T}_{\text{cof}}$ is such that $G\subseteq\pr{a,b}$ then $\mathbb{R}\setminus\pr{a,b}\subseteq\mathbb{R}\setminus G$ which is absurd, since $\mathbb{R}\setminus G$ should be a finite set and $\mathbb{R}\setminus\pr{a,b}$ is an uncountable set.
    \item \textbf{Euclidean topology}: Let $a,b\in\mathbb{R}$ with $a<b$ and $I\subseteq\mathbb{R}$ be a bounded interval with extremes $a$ and $b$. Then $\interior\pr{I}=\pr{a,b}$ with respect to $\pr{\mathbb{R},\mathcal{T}_e\pr{\mathbb{R}}}$, because any open interval containing $a$ or $b$ is not contained in $I$ and, consequently, $a,b\notin\interior\pr{I}$. Furthermore, since $I\cap\pr{\mathbb{R}\setminus\mathbb{Q}}\neq\varnothing$ for any interval $I\subseteq\mathbb{R}$, we have $\interior\pr{\mathbb{Q}}=\varnothing$ with respect to $\pr{\mathbb{R},\mathcal{T}_e\pr{\mathbb{R}}}$.
  \end{itemize}
\end{example}

\subsection{Closure and boundary of a set}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. A point $x\in X$ is said to be an \emph{adherence point of $A$  (with respect to $\pr{X,\mathcal{T}}$)} if, for all $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$, $N\cap A\neq\varnothing$. We define the \emph{adherence or closure of $A$ (with respect to $\pr{X,\mathcal{T}}$)} as the following set:
  
  \begin{equation}
      \overline{A}:=\set{x\middle|x\in X,\text{$x$ is an adherence point of $A$}}.
  \end{equation}

  Another commonly used notation is $\cl\pr{A}$.
\end{definition}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. A point $x\in X$ is said to be a \emph{boundary or frontier point of $A$  (with respect to $\pr{X,\mathcal{T}}$)} if, for every neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$, $N\cap A\neq\varnothing$ and $N\cap\pr{X\setminus A}\neq\varnothing$. We define the \emph{boundary or frontier of $A$ (with respect to $\pr{X,\mathcal{T}}$)} as the following set:
  
  \begin{equation}
    \bd\pr{A}:=\set{x\middle|x\in X,\text{$x$ is a boundary point of $A$}}.
  \end{equation}

  Another commonly used notations are $\fr\pr{A}$ and $\partial A$.
\end{definition}

\begin{remark}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. Then $\interior\pr{A}\subseteq A\subseteq\overline{A}$ and $\bd\pr{A}=\overline{A}\cap\overline{X\setminus A}$. Therefore, $\bd\pr{A}=\bd\pr{X\setminus A}$.
\end{remark}

\begin{proposition}\label{Equivalent definitions for the closure of a set}
  Let $\pr{X,\mathcal{T}}$ be a topological space with basis $\mathcal{B}\subseteq\mathcal{T}$ for $\mathcal{T}$, $A\subseteq X$ and $x\in X$. The following are equivalent:

  \begin{enumerate}
    \item $x\in\overline{A}$.
    \item For every open set $G\in\mathcal{T}$ containing $x$, $G\cap A\neq\varnothing$.
    \item For every basis set $B\in\mathcal{B}$ containing $x$, $B\cap A\neq\varnothing$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Since $G\in\mathcal{T}$ is an open set containing $x$, we have $G\in\mathcal{N}_x\pr{X,\mathcal{T}}$. Therefore, $G\cap A\neq\varnothing$.
      \item[\boxed{2\Rightarrow3}] Trivial.
      \item[\boxed{3\Rightarrow1}] Let $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$. Then there exists a basis set $B\in\mathcal{B}$ such that $x\in B\subseteq N$. Since $B\cap A\neq\varnothing$ and $B\cap A\subseteq N\cap A$, we have $N\cap A\neq\varnothing$. Therefore, $x\in\overline{A}$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{corollary}
  Let $\pr{X,\mathcal{T}}$ be a topological space with basis $\mathcal{B}\subseteq\mathcal{T}$ for $\mathcal{T}$, $A\subseteq X$ and $x\in X$. The following are equivalent:

  \begin{enumerate}
    \item $x\in\bd\pr{A}$.
    \item For every open set $G\in\mathcal{T}$ containing $x$, $G\cap A\neq\varnothing$ and $G\cap\pr{X\setminus A}\neq\varnothing$.
    \item For every basis set $B\in\mathcal{B}$ containing $x$, $B\cap A\neq\varnothing$ and $B\cap\pr{X\setminus A}\neq\varnothing$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \rm Trivial, because of the previous proposition and the fact that $\bd\pr{A}=\overline{A}\cap\overline{X\setminus A}$.
  \end{proof}
\end{corollary}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. Then:
  
  \begin{equation}
    \overline{A}=\bigcap{\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}. 
  \end{equation}
  
  Consequently, $\overline{A},\bd\pr{A}\in\mathcal{F}\pr{X,\mathcal{T}}$ are closed sets.
  \begin{proof}[\rm\textbf{Proof}]
    Let $x\in\overline{A}$. By contradiction, suppose there exists a closed set $F\in\mathcal{F}\pr{X,\mathcal{T}}$ with $A\subseteq F$ such that $x\notin F$. Then $x\in X\setminus F$. Since $G:=X\setminus F\in\mathcal{T}$ is an open set, then $G\cap A\neq\varnothing$. We have reached a contradiction, because $A\subseteq F$ and, consequently, $G\cap A=\varnothing$. Thus $x\in\bigcap{\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}$. Hence, $\overline{A}\subseteq\bigcap{\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}$.

    Let $x\in\bigcap{\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}$. By contradiction, suppose $x\notin\overline{A}$. Then there exists an open set $G\in\mathcal{T}$ containing $x$ such that $G\cap A=\varnothing$. Thus $F:=X\setminus G\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed and $A\subseteq F$. Hence $x\in F$. We have reached a contradiction, because $x\in G$. Consequently, $x\in\overline{A}$. As a result, $\bigcap{\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}\subseteq\overline{A}$.

    Therefore, $\overline{A}=\bigcap{\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}$.
  \end{proof}
\end{proposition}

\begin{remark}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. By the previous proposition, $\overline{A}$ is the smallest closed set in $\mathcal{F}\pr{X,\mathcal{T}}$ containing $A$.
\end{remark}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A,B,Y\subseteq X$. The following properties hold:
  \begin{enumerate}
    \item $A\in\mathcal{F}\pr{X,\mathcal{T}}$ if and only if $A=\overline{A}$.
    \item $\overline{\overline{A}}=\overline{A}$.
    \item If $A\subseteq B$, then $\overline{A}\subseteq\overline{B}$.
    \item $\overline{A\cup B}=\overline{A}\cup\overline{B}$.
    \item $\overline{A\cap B}\subseteq\overline{A}\cap\overline{B}$.
    \item $\interior\pr{A}=X\setminus\overline{X\setminus A}$, $\overline{A}=X\setminus\interior\pr{X\setminus A}$ and $\ext\pr{A}=X\setminus\overline{A}$.
    \item If $A\subseteq Y$, then $\overline{A}^Y=\overline{A}\cap Y$, where $\overline{A}^Y$ is the closure of $A$ with respect to the topological subspace $\pr{Y,\mathcal{T}_Y}$.
    \item $\bd\pr{A}=\overline{A}\setminus\interior\pr{A}$ and, consequently, $\bd\pr{A}\cap\interior\pr{A}=\bd\pr{A}\cap\ext\pr{A}=\varnothing$.
    \item $X=\interior\pr{A}\cup\bd\pr{A}\cup\ext\pr{A}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Trivial by the previous proposition.
      \item Trivial by the previous property.
      \item Let $x\in\overline{A}$. Given $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$, we know that $N\cap A\neq\varnothing$. Since $A\subseteq B$, we have $N\cap A\subseteq N\cap B$. Thus $N\cap B\neq\varnothing$ and, consequently, $x\in\overline{B}$. Therefore, $\overline{A}\subseteq\overline{B}$.
      \item Let $x\in\overline{A\cup B}$. Given $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$, we know that $N\cap\pr{A\cup B}\neq\varnothing$. Necessarily, $N\cap A\neq\varnothing$ or $N\cap B\neq\varnothing$ or, in other words, $x\in\overline{A}$ or $x\in\overline{B}$. Thus $\overline{A\cup B}\subseteq\overline{A}\cup\overline{B}$.
      
      Since $A,B\subseteq A\cup B$, we have $\overline{A},\overline{B}\subseteq\overline{A\cup B}$. Then $\overline{A}\cup\overline{B}\subseteq\overline{A\cup B}$.
      
      Therefore, $\overline{A\cup B}=\overline{A}\cup\overline{B}$.
      \item Let $x\in\overline{A\cap B}$. Given $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$, we know that $N\cap\pr{A\cap B}\neq\varnothing$. Since $N\cap\pr{A\cap B}\subseteq N\cap A,N\cap B$, we have $N\cap A\neq\varnothing$ and $N\cap B\neq\varnothing$. Then $x\in\overline{A}$ and $x\in\overline{B}$, and, consequently, $x\in\overline{A}\cap\overline{B}$. Therefore, $\overline{A\cap B}\subseteq\overline{A}\cap\overline{B}$.
      \item It suffices to prove that $\interior\pr{A}=X\setminus\overline{X\setminus A}$. Let $x\in\interior\pr{A}$. Then there exists a neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$ such that $N\subseteq A$. Then $N\cap\pr{X\setminus A}\neq\varnothing$. Consequently, $x\notin\overline{X\setminus A}$ or, in other words, $x\in X\setminus\overline{X\setminus A}$. Thus $\interior\pr{A}\subseteq X\setminus\overline{X\setminus A}$.
      
      Let $x\in X\setminus\overline{X\setminus A}$. Then $x\notin\overline{X\setminus A}$. Thus, there exists a neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$ such that $N\cap\pr{X\setminus A}=\varnothing$. Hence $x\in N\subseteq A$ and, consequently, $x\in\interior\pr{A}$. As a result, $X\setminus\overline{X\setminus A}\subseteq\interior\pr{A}$.

      Therefore, $\interior\pr{A}=X\setminus\overline{X\setminus A}$.
      \item Indeed, we have:
      
      \begin{equation}
        \begin{split}
          \overline{A}^Y & =\bigcap\set{F'\middle|F'\in\mathcal{F}\pr{Y,\mathcal{T}_Y},A\subseteq F'}=\bigcap\set{F\cap Y\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F\cap Y}\\
          & =\bigcap\set{F\cap Y\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}=\pr{\bigcap\set{F\middle|F\in\mathcal{F}\pr{X,\mathcal{T}},A\subseteq F}}\cap Y\\
          & =\overline{A}\cap Y.
        \end{split}
      \end{equation}
      \item Observe that $\bd\pr{A}=\overline{A}\cap\overline{X\setminus A}=\overline{A}\cap\pr{X\setminus\text{int}\pr{A}}=\overline{A}\setminus\interior\pr{A}$.
      \item Indeed, we have:
      
      \begin{equation}
        \interior\pr{A}\cup\bd\pr{A}\cup\ext\pr{A}=\interior\pr{A}\cup\pr{\overline{A}\setminus\interior\pr{A}}\cup\pr{X\setminus\overline{A}}=\overline{A}\cup\pr{X\setminus\overline{A}}=X.
      \end{equation}
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{example}
  Let $\pr{X,\mathcal{T}}$ be a topological space. Some examples of the closure of sets are the following:

  \begin{itemize}
    \item \textbf{Discrete topology}: For every $A\subseteq X$, $\overline{A}=A$ with respect to $\pr{X,\mathcal{T}_{\text{dis}}}$ because $A\in\mathcal{F}\pr{X,\mathcal{T}_{\text{dis}}}$.
    \item \textbf{Cofinite topology}: Let $a,b\in\mathbb{R}$ with $a<b$. Then $\overline{\pr{a,b}}=\varnothing$ with respect to $\pr{\mathbb{R},\mathcal{T}_{\text{cof}}}$, because if every closed set containing $\pr{a,b}$ must be finite with respect to $\pr{\mathbb{R},\mathcal{T}_{\text{cof}}}$ which is impossible.
    \item \textbf{Euclidean topology}: Let $a,b\in\mathbb{R}$ with $a<b$ and $I\subseteq\mathbb{R}$ be a bounded interval with extremes $a$ and $b$. Then $\overline{I}=\br{a,b}$ with respect to $\pr{\mathbb{R},\mathcal{T}_e\pr{\mathbb{R}}}$, because for any $x\notin\br{a,b}$, there exists an open interval which does not intersects with $\br{a,b}$ and, consequently, with $I$ as well. Furthermore, $\overline{\mathbb{Q}}=\mathbb{R}$, because for any $x\in\mathbb{R}\setminus\mathbb{Q}$ and any open interval $I\subseteq\mathbb{R}$ containing $x$, $I\cap\mathbb{Q}\neq\varnothing$.
  \end{itemize}
\end{example}

\subsection{Derived set}
\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. A point $x\in X$ is said to be an \emph{accumulation point of $A$ (with respect to $\pr{X,\mathcal{T}}$)} if for every neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$, $N\cap\pr{A\setminus\set{x}}\neq\varnothing$ or, in other words, $x\in\overline{A\setminus\set{x}}$. We define the \emph{derived set of $A$ (with respect to $\pr{X,\mathcal{T}}$)} as the following set:

  \begin{equation}
    A'=\set{x\middle|x\in X, x\text{ is an accumulation point of $A$}}.
  \end{equation}

  Another commonly used notation is $\text{der}\pr{A}$. If $x\in A\setminus A'$, we say that $x$ is an \emph{isolated point of $A$}.
\end{definition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A,B\subseteq X$. The following properties hold:
  \begin{enumerate}
    \item $\overline{A}=A\cup A'$.
    \item $A\in\mathcal{F}\pr{X,\mathcal{T}}$ if and only if $A'\subseteq A$.
    \item If $A\subseteq B$, then $A'\subseteq B'$.
    \item $\pr{A\cup B}'=A'\cup B'$.
    \item $\pr{A\cap B}'\subseteq A'\cap B'$. 
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item It is straightforward to check that $A\cup A'\subseteq\overline{A}$. 
      
      Let $x\in\overline{A}$. If $x\in A$, then $x\in A\cup A'$ trivially. Suppose that $x\notin A$. Then $A=A\setminus\set{x}$ and $x\in\overline{A}=\overline{A\setminus\set{x}}$. Thus $x\in A'$. Hence, $x\in A\cup A'$. As a result, $\overline{A}\subseteq A\cup A'$. 
      
      Therefore, $\overline{A}=A\cup A'$.
      \item $A\in\mathcal{F}\pr{X,\mathcal{T}}$ if and only if $A=\overline{A}=A\cup A'$ which is equivalent to $A'\subseteq A$.
      \item Let $x\in A'$. Then $x\in\overline{A\setminus\set{x}}$. Since $A\subseteq B$, then $\overline{A\setminus\set{x}}\subseteq\overline{B\setminus\set{x}}$. Thus $x\in\overline{B\setminus\set{x}}$. Consequently, $x\in B'$. Therefore, $A'\subseteq B'$.
      \item Observe the following:

      \begin{equation}
        \overline{\pr{A\cup B}\setminus\set{x}}=\overline{\pr{A\setminus\set{x}}\cup\pr{B\setminus\set{x}}}=\overline{A\setminus\set{x}}\cup\overline{B\setminus\set{x}}.
      \end{equation}
      
      Then $x\in \pr{A\cup B}'$ if and only if $x\in A'\cup B'$. Therefore, $\pr{A\cup B}'=A'\cup B'$.
      \item Since $A\cap B\subseteq A,B$, we have $\pr{A\cap B}'\subseteq A',B'$. Therefore, $\pr{A\cap B}'\subseteq A'\cap B'$.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{example}
  Let $\pr{X,\mathcal{T}}$ be a topological space. Some examples of accumulation points of sets are the following:

  \begin{itemize}
    \item \textbf{Discrete topology}: Let $A\subseteq X$ and $x\in A$. Then $x\notin A'$ with respect to $\pr{X,\mathcal{T}_{\text{dis}}}$, because $\set{x}\in\mathcal{T}_{\text{disc}}$, $x\in\set{x}$ and $\pr{A\setminus\set{x}}\cap\set{x}=\varnothing$.
    \item \textbf{Cofinite topology}: Assume $X=\mathbb{R}$. Given $x\in\mathbb{R}$, let $G\in\mathcal{T}_{\text{cof}}$ be an open set such containing $x$. Since $X\setminus G$ is a finite set, necessarily, $\pr{\mathbb{N}\setminus\set{x}}\cap G\neq\varnothing$. Therefore, $x\in\mathbb{N}'$ is an accumulation point of $\mathbb{N}$ with respect to $\pr{\mathbb{R},\mathcal{T}_{\text{cof}}}$ for all $x\in\mathbb{R}$ and, therefore, $\mathbb{N}'=\mathbb{R}$ with respect to $\pr{\mathbb{R},\mathcal{T}_{\text{cof}}}$.
    \item \textbf{Cocountable topology}: Assume $X=\mathbb{R}$. Let $A:=\set{\frac{1}{n}\middle|n\in\mathbb{N}}\subseteq\mathbb{R}$. Observe that $X\setminus A\in\mathcal{T}_{\text{cc}}$, because $A$ is a countable set. Furthermore, $0\in X\setminus A$. Since $A\cap\pr{X\setminus A}=\varnothing$, $0$ is not an accumulation point of $A$ with respect to $\pr{\mathbb{R},\mathcal{T}_{\text{cc}}}$.
  \end{itemize}
\end{example}

\section{Density}

\begin{definition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $A\subseteq X$. We say that \emph{$A$ is dense (with respect to $\pr{X,\mathcal{T}}$)} if $\overline{A}=X$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space with basis $\mathcal{B}\subseteq\mathcal{T}$ for $\mathcal{T}$ and $A\subseteq X$. The following are equivalent:

  \begin{enumerate}
      \item $A$ is dense.
      \item For every non-empty open set $G\in\mathcal{T}$, $G\cap A\neq\varnothing$.
      \item For every non-empty basis set $B\in\mathcal{B}$, $B\cap A\neq\varnothing$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    It is direct consequence of \hyperref[Equivalent definitions for the closure of a set]{Proposition~\ref*{Equivalent definitions for the closure of a set}}.
  \end{proof}
\end{proposition}

\begin{example}
  Let $\pr{X,\mathcal{T}}$ be a topological space. Some examples of dense sets with respect to $\pr{X,\mathcal{T}}$ are the following:

  \begin{itemize}
    \item \textbf{Discrete topology}: If $X\neq\varnothing$, the only dense set with respect to $\pr{X,\mathcal{T}_{\text{dis}}}$ is $X$.
    \item \textbf{Indiscrete topology}: If $X\neq\varnothing$, any non-empty subset of $X$ is a dense set with respect to $\pr{X,\mathcal{T}_{\text{ind}}}$.
    \item \textbf{Cofinite topology}: Any infinite set is dense with respect to $\pr{X,\mathcal{T}_{\text{cof}}}$.
    \item \textbf{Cocountable topology}: Any uncountable set is dense with respect to $\pr{X,\mathcal{T}_{\text{cc}}}$.
    \item \textbf{Euclidean topology, Sorgenfrey's line and upper bound topology}: $\mathbb{Q}$ and $\mathbb{R}\setminus\mathbb{Q}$ are dense with respect to $\pr{\mathbb{R},\mathcal{T}_e\pr{\mathbb{R}}}$, $\mathbb{R}_{\ell}$ and $\mathbb{R}_u$.
  \end{itemize}
\end{example}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space, $A,B\subseteq X$ with $A$ dense with respect to $\pr{X,\mathcal{T}}$, $Y\in\mathcal{T}$, $d\in\mathbb{N}$, $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$ be $d$ topological spaces and $A_1,\dots,A_d$ be $d$ dense sets with respect to $\pr{X_1,\mathcal{T}_1},\dots,\pr{X_d,\mathcal{T}_d}$ respectively. The following properties hold:

  \begin{enumerate}
      \item If $A\subseteq B$, then $B$ is dense with respect to $\pr{X,\mathcal{T}}$.
      \item $A\cap Y$ is dense with respect to $\pr{Y,\mathcal{T}_Y}$.
      \item $A_1\times\cdots\times A_d$ is dense with respect to $\pr{X_1\times\cdots\times X_d,\mathcal{T}_{X_1\times\cdots\times X_d}}$.
  \end{enumerate}
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $G\in\mathcal{T}$ be a non-empty open set. Then $G\cap A\neq\varnothing$. Since $A\subseteq B$, we have $G\cap B\neq\varnothing$. Therefore, $B$ is dense with respect to $\pr{X,\mathcal{T}}$.
      \item Let $G'\in\mathcal{T}_Y$ be a non-empty open set. Then there exists an open set $G\in\mathcal{T}$ such that $G'=G\cap Y$. Since $Y\in\mathcal{T}$, then $G'=G\cap Y\in\mathcal{T}$. Thus $G'\cap\pr{A\cap Y}=\pr{G\cap Y}\cap A\neq\varnothing$. Therefore, $A\cap Y$ is dense with respect to $\pr{Y,\mathcal{T}_Y}$.
      \item Let $G_1\in\mathcal{T}_1,\dots,G_d\in\mathcal{T}_d$ be $d$ open sets. Then $A_k\cap G_k\neq\varnothing$ for all $k\in\set{1,\dots,d}$. Thus $\pr{A_1\times\cdots\times A_d}\cap\pr{G_1\times\cdots\times G_d}=\pr{A_1\cap G_1}\times\cdots\times\pr{A_d\cap G_d}\neq\varnothing$. Therefore, $A_1\times\cdots\times A_d$ is dense with respect to $\pr{X_1\times\cdots\times X_d,\mathcal{T}_{X_1\times\cdots\times X_d}}$.
    \end{enumerate}
  \end{proof}
\end{proposition}
