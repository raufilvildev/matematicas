\section{Separation axioms}

\begin{definition}
  A topological space $\pr{X,\mathcal{T}}$ is \textit{$T_0$ or Kolmogorov} if for any pair of points $x,y\in X$, there exists a neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$ such that $y\notin N$ or there exists a neighborhood $M\in\mathcal{N}_y\pr{X,\mathcal{T}}$ of $y$ such that $x\notin M$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $\mathcal{B}\subseteq\mathcal{T}$ a basis for $\mathcal{T}$. The following are equivalent:

  \begin{enumerate}
    \item $\pr{X,\mathcal{T}}$ is Kolmogorov.
    \item For every pair of distinct points $x,y\in X$, there exists an open set $G\in\mathcal{T}$ such that $x\in G$ and $y\notin G$, or, $x\notin G$ and $y\in G$.
    \item For every pair of distinct points $x,y\in X$, there exists a basis set $B\in\mathcal{B}$ such that $x\in B$ and $y\notin B$, or, $x\notin B$ and $y\in B$.
  \end{enumerate}

  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Suppose there exists a neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ (resp. $M\in\mathcal{N}_y\pr{X,\mathcal{T}}$) of $x$ (resp. $y$) such that $x\in N$ and $y\notin N$ (resp. $x\notin M$ and $y\in M$). Then there exists an open set $G\in\mathcal{T}$ containing $x$ (resp. $y$) such that $G\subseteq N$ (resp. $G\subseteq M$). Since $y\notin N$ (resp. $x\notin M$) and $G\subseteq N$ (resp. $G\subseteq M$), we know that $y\notin G$ (resp. $x\notin G$).
      \item[\boxed{2\Rightarrow3}] Suppose there exists an open set $G\in\mathcal{T}$ containing $x$ (resp. $y$) such that $x\in G$ and $y\notin G$ (reso, $x\notin G$ and $y\in G$). Then there exists a basis set $B\in\mathcal{B}$ containing $x$ (resp. $y$) such that $B\subseteq G$. Since $y\notin G$ (resp. $x\notin G$) and $B\subseteq G$, then $y\notin B$ (resp. $x\notin B$).
      \item[\boxed{3\Rightarrow1}] Trivial.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  The following properties hold:

  \begin{enumerate}
    \item If $\pr{X,\mathcal{T}}$ is a Kolmogorov space and $A\subseteq X$, the subspace $\pr{A,\mathcal{T}_A}$ is a Kolmogorov space.
    \item If $\set{\pr{X_i,\mathcal{T}_i}}_{i\in I}$ is an indexed collection of Kolmogorov spaces, then $\pr{\prod_{i\in I}{X_i},\mathcal{T}_{\prod_{i\in I}{X_i}}}$ is a Kolmogorov space.
  \end{enumerate}
  
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{enumerate}
      \item Let $x,y\in A$ be two distinct points. Then there exists $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ (resp. $M\in\mathcal{N}_y\pr{X,\mathcal{T}}$) of $x$ (resp. $y$) such that $y\notin N$ (resp. $x\notin M$). Observe that $N\cap A\in\mathcal{N}_x\pr{A,\mathcal{T}_A}$ (resp. $N\cap A\in\mathcal{N}_y\pr{A,\mathcal{T}_A}$) is a neighborhood of $x$ (resp. $y$). Therefore, $\pr{A,\mathcal{T}_A}$ is a Kolmogorov space. 
      \item Let $x:=\pr{x_i}_{i\in I},y:=\pr{y_i}_{i\in I}\in\prod_{i\in I}{X_i}$ be two distinct points. Then there exists $i_0\in I$ such that $x_{i_0}\neq y_{i_0}$. Since $\pr{X_{i_0},\mathcal{T}_{i_0}}$ is a Kolmogorov space, there exists an open set $G_{i_0}\in\mathcal{T}_{i_0}$ containing $x$ (resp. $y$) such that $y\notin G_{i_0}$ (resp. $x\notin G_{i_0}$). Let $G=\prod_{i\in I}{G_i}$ where $G_i=X_i$ for all $i\in I\setminus\set{i_0}$. Then $G\in\mathcal{T}_{\prod_{i\in I}{X_i}}$ is an open set containing $x$ (resp. $y$) such that $y\notin G$ (resp. $x\notin G$). Therefore, $\pr{\prod_{i\in I}{X_i},\mathcal{T}_{\prod_{i\in I}{X_i}}}$ is a Kolmogorov space.
    \end{enumerate}
  \end{proof}
\end{proposition}

\begin{definition}
  A topological space $\pr{X,\mathcal{T}}$ is \textit{$T_1$ or Fréchet} if for any pair of distinct points $x,y\in X$, there exists a neighborhood $N_1\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$ and a neighborhood $N_2\in\mathcal{N}_y\pr{X,\mathcal{T}}$ of $y$ such that $x\in N_1\setminus N_2$ and $y\in N_2\setminus N_1$.
\end{definition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space and $\mathcal{B}\subseteq\mathcal{T}$ a basis for $\mathcal{T}$. The following are equivalent:

  \begin{enumerate}
    \item $\pr{X,\mathcal{T}}$ is Fréchet.
    \item For every pair of distinct points $x,y\in X$, there exists a pair of open sets $G_1,G_2\in\mathcal{T}$ such that $x\in G_1\setminus G_2$ and $y\in G_2\setminus G_1$.
    \item For every pair of distinct points $x,y\in X$,  there exists a pair of basis sets $B_1,B_2\in\mathcal{B}$ such that $x\in B_1\setminus B_2$ and $y\in B_2\setminus B_1$.
  \end{enumerate}
  
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Let $x,y\in X$ be two distinct points. Then there exists $N_1\in\mathcal{N}_x\pr{X,\mathcal{T}}$ be a neighborhood of $x$ and $N_2\in\mathcal{N}_y\pr{X,\mathcal{T}}$ be a neighborhood of $y$ such that $x\in N_1\setminus N_2$ and $y\in N_2\setminus N_1$. Then there exists two open sets $G_1,G_2\in\mathcal{T}$ such that $G_1$ contains $x$, $G_1\subseteq N_1$, $G_2$ contains $y$ and $G_2\subseteq N_2$. Since $x\notin N_2$ and $G_2\subseteq N_2$, then $x\notin G_2$. Furthermore, as $y\notin N_1$ and $G_1\subseteq N_1$, then $y\notin G_1$. Therefore, $x\in G_1\setminus G_2$ and $y\in G_2\setminus G_1$.
      \item[\boxed{2\Rightarrow3}] Let $x,y\in X$ be two distinct points and, $G_1,G_2\in\mathcal{T}$ be two open sets such that $x\in G_1\setminus G_2$ and $y\in G_2\setminus G_1$. Then there exists two basis sets $B_1,B_2\in\mathcal{B}$ such that $B_1$ contains $x$, $B_1\subseteq G_1$, $B_2$ contains $y$ and $B_2\subseteq G_2$. Since $x\notin G_2$ and $B_2\subseteq G_2$, then $x\notin B_2$. Furthermore, as $y\notin G_1$ and $B_1\subseteq G_1$, then $y\notin B_1$. Therefore, $x\in B_1\setminus B_2$ and $y\in B_2\setminus B_1$.
      \item[\boxed{3\Rightarrow1}] Trivial.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a topological space. The following are equivalent:

  \begin{enumerate}
    \item $\pr{X,\mathcal{T}}$ is a Fréchet space.
    \item For every $x\in X$, $\set{x}\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed set.
    \item For every finite subset $A\subseteq X$ of $X$, $A\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed set.
  \end{enumerate}

  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{1\Rightarrow2}] Let $x\in X$. Given $y\in X$, there exists an open set $G_y\in\mathcal{T}$ such that $x\notin G_y$. Then $X\setminus\set{x}=\bigcup_{y\in X\setminus\set{x}}{G_y}\in\mathcal{T}$ is an open set. Therefore, $\set{x}\in\mathcal{F}\pr{X,\mathcal{T}}$ is a closed set.
      \item[\boxed{2\Rightarrow3}] Trivial, because a finite set is finite union of singletons which by hypothesis are closed with respect to $\pr{X,\mathcal{T}}$.
      \item[\boxed{3\Rightarrow1}] Let $x,y\in X$ two distinct points. Since $\set{x},\set{y}\in\mathcal{F}\pr{X,\mathcal{T}}$ are closed sets, then $G_1:=X\setminus\set{y},G_2:=X\setminus\set{x}\in\mathcal{T}$ are open sets. Furthermore, $x\in G_1\setminus G_2$ and $y\in G_2\setminus G_1$. Therefore, $\pr{X,\mathcal{T}}$ is a Fréchet space.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{proposition}
  Let $\pr{X,\mathcal{T}}$ be a Fréchet space, $x\in X$ and $A\subseteq X$. Then $x\in A'$ if and only if for every neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$, the set $N\cap A$ has infinite points.
  
  \begin{proof}[\rm\textbf{Proof}]
    \hfill
    \begin{itemize}
      \item[\boxed{\Rightarrow}] By contradiction, suppose there exists a neighborhood $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ of $x$, such that $N\cap A$ is a finite set. Since $\pr{X,\mathcal{T}}$ is a Fréchet space, we know that $N\cap A\in\mathcal{F}\pr{X,\mathcal{T}}$ is closed. Particularly, $F:=\pr{N\cap A}\setminus\set{x}\in\mathcal{F}\pr{X,\mathcal{T}}$ is closed. Let $G:=X\setminus F\in\mathcal{T}$. Observe that $x\in G$ and, consequently, $G\in\mathcal{N}_x\pr{X,\mathcal{T}}$. Thus $M:=N\cap G\in\mathcal{N}_x\pr{X,\mathcal{T}}$ is a neighborhood of $x$. However:
      
      \begin{equation}
        M\cap A=\pr{N\cap G}\cap A=\pr{N\cap\pr{X\setminus F}}\cap A=\pr{N\cap\pr{X\setminus\pr{\pr{N\cap A}\setminus\set{x}}}}\cap A=\set{x}.
      \end{equation}
    
      Consequently, $\pr{M\cap A}\setminus\set{x}=\varnothing$. We have reached a contradiction, because $M\in\mathcal{N}_x\pr{X,\mathcal{T}}$ and $\pr{M\cap A}\setminus\set{x}=\varnothing$. Therefore, $x\in A'$.
      \item[\boxed{\Leftarrow}] Let $N\in\mathcal{N}_x\pr{X,\mathcal{T}}$ be a neighborhood of $x$. Since $N\cap A$ has infinite points, then $\pr{N\cap A}\setminus\set{x}\neq\varnothing$. Therefore, $x\in A'$.
    \end{itemize}
  \end{proof}
\end{proposition}

\begin{definition}

\begin{proposition}
  If $\pr{X,\mathcal{T}}$ is a Hausdorff space and $\set{x_n}_{n\in\mathbb{N}}\subseteq X$ is a convergent sequence in $X$, then there exists a unique $x\in X$ such that $\lim_{n\to\infty}{x_n}=x$.

  \begin{proof}[\rm\textbf{Proof}]
    By contradiction, suppose there exists two distinct points $x,y\in X$ such that $\lim_{n\to\infty}{x_n}=x$ and $\lim_{n\to\infty}{x_n}=y$. Let $G_x,G_y\in\mathcal{T}$ be a pair of disjoint open sets such that $x\in G_x$ and $y\in G_y$. Then there exists $n_1,n_2\in\mathbb{N}$ such that $x_n\in G_x$ for all $n\in\mathbb{N}$ with $n\geq n_1$ and $x_n\in G_y$ for all $n\in\mathbb{N}$ with $n\geq n_2$. Let $n_0:=\max\set{n_1,n_2}\in\mathbb{N}$. Then $x_n\in G_1\cap G_2$ for all $n\in\mathbb{N}$ with $n\geq n_0$. We have reached a contradiction, because $G_1\cap G_2=\varnothing$. Therefore, $x=y$.
  \end{proof}
\end{proposition}
